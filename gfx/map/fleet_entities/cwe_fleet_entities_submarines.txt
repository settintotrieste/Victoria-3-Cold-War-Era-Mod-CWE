ship_type_submarine_1 = {
    ship_type = ship_type_submarine_1
	weapon_type = turret
	hull_type = steel
	motion_intensity = 0.7
	entity = "submarine_base_entity"

	locator = {
		name = loc_foam
		position = { 0.0 0.03 0.0 }
	}

	# Positions of damage decals
	damage_decal_positions = {
		{ -0.1 0.04 1.5 0.0 }
		{ 0.0 0.05 1.0 0.0 }
		{ -0.0 0.055 1.5 0.0 }
		{ 0.05 0.045 1.2 0.0 }
	}

	# VFX entity for states: 'damaged_light' and 'damaged_heavy'
	vfx_entity = {
		entity = "vfx_ship_hull_submarine_damaged_entity"
		attach = yes  								# Controls whether VFX should follow ship animation or not
	}

	# VFX entity for states: 'damage' and 'damage_weak'
	vfx_entity = {
		entity = "vfx_ship_submarine_damage_entity"
		attach = yes
	}

	# VFX entity for states: 'idle', 'moving', 'move_local', and 'sink'
	vfx_entity = {
		entity = "vfx_submarine_foam_entity"
		locator = loc_foam
	}

	# VFX entity for states: 'miss'
	vfx_entity = {
		entity = "vfx_ship_water_miss_entity"
		locator = loc_foam
	}

	# Armour
	attach = {
		locator = root
		slot_type = ship_mod_slot_armor
		modification = { entity = "hull_submarine_light_entity" }
		modification = { entity = "hull_submarine_medium_entity" }
		modification = { entity = "hull_submarine_heavy_entity" }
	}

}
