### Investment Treaties

# Lemur Concession
treaty_name_name_concession = {
	trigger = {
	    any_scope_article_option = {
	    	has_type = foreign_investment_rights
	    }
	    NOT = {
		    any_scope_article_option = {
		    	has_type = foreign_investment_rights
		    	count >= 2
		    }
	    }
	    NOT = {
		    any_scope_article_option = {
		    	OR = {
		    		has_type = alliance
		    		has_type = defensive_pact
		    	}
		    }
	    }
	}
	weight = {
		value = 200
	}
}

		value = 10000 # All treaties between the Holy See and countries are Concordats
	}
