### RULER IS A KING ###

gov_regency_empire = {
	transfer_of_power = hereditary

	male_ruler = "RULER_TITLE_REGENT"
	female_ruler = "RULER_TITLE_REGENT"

	male_heir = "RULER_TITLE_EMPEROR"
	female_heir = "RULER_TITLE_EMPRESS"

	possible = { is_a_condominium_gov = no
	
		#Head of State is a Monarch
		has_law_or_variant = law_type:law_HOS_monarch
		
		#Special exceptions only
		OR = {
			
			#From culture
			country_has_primary_culture = cu:japanese
			country_has_primary_culture = cu:amhara
			country_has_primary_culture = cu:chinese
			country_has_primary_culture = cu:vietnamese
			country_has_primary_culture = cu:korean
			country_has_primary_culture = cu:north_german
			country_has_primary_culture = cu:south_german
			country_has_primary_culture = cu:russian
			country_has_primary_culture = cu:indian
			country_has_primary_culture = cu:persian
			
			#From country tag
			c:CAR ?= this
			c:TUR ?= this
			c:BRZ ?= this
			c:MEX ?= this
			c:HAI ?= this
			
		}
		
		#Head of Government is Monarch or Shared
		OR = {
			has_law_or_variant = law_type:law_HOG_none
			has_law_or_variant = law_type:law_HOG_president_prime_minister
		}
		
		#Not a PU or dominion
		shares_same_monarch = no
		
		#Not a colony or SAR
		NOR = {
			is_subject_type = subject_type_semi_autonomous_colony
			#is_subject_type = subject_type_mandate
			is_subject_type = subject_type_special_administrative_region
		}
		
		has_gov_regency = yes
		
	}

	on_government_type_change = {
		change_to_regency = yes
	}
	on_post_government_type_change = {
		post_change_to_regency = yes
	}
}
