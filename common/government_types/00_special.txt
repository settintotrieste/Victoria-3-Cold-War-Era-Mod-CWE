#### Mandatory Governments ####

gov_mandate_administration_dictatorial_dynastic = {
	transfer_of_power = hereditary

	male_ruler = "RULER_TITLE_GOVERNOR_GENERAL"
	female_ruler = "RULER_TITLE_GOVERNOR_GENERAL"
	
	possible = {
		
		#Applies to all dictatorial mandates
		is_subject_type = subject_type_mandate
		
		#Legislature is not fully elected
		OR = { 
			has_law_or_variant = law_type:law_appointed_legislature
			has_law_or_variant = law_type:law_technocratic_legislature
		}
		
		#Dynastic Politics
		ruler ?= { has_variable = dynastic_ruler_enabled }
	}
	
	on_government_type_change = {
		change_to_dictatorial = yes
	}
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
	
}

gov_mandate_administration_dictatorial = {
	transfer_of_power = dictatorial

	male_ruler = "RULER_TITLE_GOVERNOR_GENERAL"
	female_ruler = "RULER_TITLE_GOVERNOR_GENERAL"
	
	possible = {
		
		#Applies to all dictatorial mandates
		is_subject_type = subject_type_mandate
		
		#Legislature is not fully elected
		OR = { 
			has_law_or_variant = law_type:law_appointed_legislature
			has_law_or_variant = law_type:law_technocratic_legislature
		}
		
		#Dynastic Politics
		NOT = { ruler ?= { has_variable = dynastic_ruler_enabled } }
		
	}
	
	on_government_type_change = {
		change_to_dictatorial = yes
	}
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
	
}
