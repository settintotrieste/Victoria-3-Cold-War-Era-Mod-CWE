#### Normal Presidential Republics ####

gov_presidential_dictatorship_dynastic = {
	transfer_of_power = hereditary

	male_ruler = "RULER_TITLE_PRESIDENT"
	female_ruler = "RULER_TITLE_PRESIDENT"
	
	possible = { is_a_condominium_gov = no
		#Head of State is a President
		has_law_or_variant = law_type:law_HOS_president
		
		#Legislature is not elected
		has_law_or_variant = law_type:law_appointed_legislature
		
		#Head of Government is President or Shared
		OR = {
			has_law_or_variant = law_type:law_HOG_none
			has_law_or_variant = law_type:law_HOG_president_prime_minister
		}
		
		#Not a colony or SAR
		NOR = {
			is_subject_type = subject_type_semi_autonomous_colony
			#is_subject_type = subject_type_mandate
			is_subject_type = subject_type_special_administrative_region
		}
		
		#Dynastic Politics
		ruler ?= { has_variable = dynastic_ruler_enabled }
	}
	
	on_government_type_change = {
		change_to_dictatorial = yes
	}
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
	
}
