				}
			}
		}
		
	}

}

amendment_rotating_presidency = {
	parent = law_democratic_legislature

	allowed_laws = {
		law_HOS_president
		law_HOS_religious_leader
		law_HOS_nationalist_leader
		law_HOS_communist_leader
	}

	modifier = {
		state_loyalists_from_political_movements_mult = 0.10
	}

	possible = {
	
		#Presidential system
		OR = { 
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOS_nationalist_leader
			has_law_or_variant = law_type:law_HOS_religious_leader
			has_law_or_variant = law_type:law_HOS_communist_leader
		}
		
		#Not for Prime Ministers
		NOT = { has_law_or_variant = law_type:law_HOG_prime_minister }
		
		#Has Term Limit
		has_law_or_variant = law_type:law_yes_term_limit
		
	}
	
	would_sponsor = {
	
		scope:approval > 0
	
	}

	can_repeal = { 
		government_has_legislative_super_majority = yes
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 1.0

	# Effect will trigger when amendment is added to an active law 
	on_activate = {
		
		#Ruler will oppose this
		ruler ?= {
			interest_group = { save_scope_as = ig_scope }
			owner = {
				add_radicals = {
					value = large_radicals
					interest_group = scope:ig_scope
				}
			}
		}
		
	}

	# Effect will trigger when amendment is repealed from an active law 
	on_deactivate = {
		
		#Opposition will oppose this
		every_interest_group = {
			limit = { is_in_government = no }
			save_scope_as = ig_scope
			owner = {
				add_radicals = {
					value = large_radicals
					interest_group = scope:ig_scope
				}
			}
		}
		
	}

}

amendment_corporate_state = {
	parent = law_HOS_nationalist_leader

	allowed_laws = {
		law_HOS_president
		law_HOS_monarch
		law_HOS_nationalist_leader
	}

	modifier = {
		country_legitimacy_govt_size_add = 2
		country_legitimacy_ideological_incoherence_mult = -0.25
		country_authority_add = 100
		interest_group_ig_industrialists_pol_str_mult = 0.15
		interest_group_ig_armed_forces_pol_str_mult = 0.15
		interest_group_ig_intelligentsia_pol_str_mult = 0.15
	}
	
	possible = {
	
		#Legislature is not fully elected
		OR = { 
			has_law_or_variant = law_type:law_appointed_legislature
			has_law_or_variant = law_type:law_technocratic_legislature
		}
		
	}

	would_sponsor = {
		scope:approval > 0
		OR = { 
			is_interest_group_type = ig_industrialists
			is_interest_group_type = ig_armed_forces
			is_interest_group_type = ig_intelligentsia
		}
	}
	
	ai_will_revoke = {
		ruler ?= {
			law_stance = {
				law = law_type:law_appointed_legislature
				value < neutral
			}
			interest_group ?= {
				law_stance = {
					law = law_type:law_appointed_legislature
					value < neutral
				}
			}
		}
	}

	can_repeal = { 
		government_has_legislative_super_majority = yes
		legitimacy >= legitimacy_to_repeal_amendment
	}
	
	amendment_activism_multiplier = 1.0

	# Effect will trigger when amendment is added to an active law 
	on_activate = {
		
		add_loyalists = {
			value = large_radicals
			interest_group = ig:ig_industrialists
		}
		add_loyalists = {
			value = large_radicals
			interest_group = ig:ig_armed_forces
		}
		add_loyalists = {
			value = large_radicals
			interest_group = ig:ig_intelligentsia
		}
		
	}

	# Effect will trigger when amendment is repealed from an active law 
	on_deactivate = {
		
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_industrialists
		}
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_armed_forces
		}
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_intelligentsia
		}
		
	}

}
