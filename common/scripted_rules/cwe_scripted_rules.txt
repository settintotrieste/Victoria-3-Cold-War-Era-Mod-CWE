	}

# Checks if any diplomatic play is possible against the target country
# Root = Country
# scope:target_country = target of a diplomatic play
REPLACE_OR_CREATE:can_start_diplomatic_plays_against = {

	# Not allowed to attack subjects of play participants
	NOT = {
		scope:target_country.top_overlord ?= {
			is_diplomatic_play_participant_with = root
		}
	}
	NOT = {
		scope:target_country.top_overlord ?= {
			is_diplomatic_play_involved_with = root
		}
	}
	
	# Might be disallowed by power bloc
	NOT = {
		custom_tooltip = {
			text = "power_bloc_disallow_war_bool"
			is_in_same_power_bloc = scope:target_country
			power_bloc ?= {
				modifier:power_bloc_disallow_war_bool = yes
			}
			# Conflicts between Overlord and (their own) Subject are always permitted despite Power Bloc special conditions
			NOR = {
				is_direct_subject_of = scope:target_country
				scope:target_country = {
					is_direct_subject_of = root
				}
			}
		}
	}
	NOT = {
		# Disallow plays targeting your indirect subjects
		scope:target_country = {
			is_indirect_subject_of = root
		}
	}
	
	#Has interest in region
	OR = {
		has_strategic_region_interest_tier = {
			strategic_region = scope:region
			value >= interest_tier_effect_can_start_diplomatic_plays_min_tier
		}
		is_direct_subject_of = scope:target_country
		scope:target_country = { is_direct_subject_of = root }
	}
	
	#Not has a non-aggression pact
	NOT = { 
		any_scope_treaty = { 
			binds = scope:target_country 
			any_scope_article = { has_type = non_aggression_pact } 
		}
	}
	
}

# Checks if Country can join the target country's side in a diplomatic play
# Root = Country
# scope:target_country = country of the side to join to
# scope:enemy_country = enemy country of target_country to which we could fight against
REPLACE_OR_CREATE:can_join_side_in_diplomatic_play = {
	# Might be disallowed by power bloc
	NOT = {
		custom_tooltip = {
			text = "POWER_BLOC_CANNOT_SIDE_AGAINST_OVERLORD"
			is_in_same_power_bloc = scope:target_country
			is_in_same_power_bloc = scope:enemy_country
			power_bloc ?= {
				modifier:power_bloc_disallow_war_bool = yes
			}
			# Conflicts between Overlord and (their own) Subject are always permitted despite Power Bloc special conditions
			# We can join the overlord side but not the subject's one
			scope:target_country = {
				is_direct_subject_of = scope:enemy_country
			}
		}
	}
	NOT = {
		# Disallow plays targeting your indirect subjects
		scope:enemy_country = {
			is_indirect_subject_of = root
		}
	}
	
	OR = {
		AND = {
			has_strategic_region_interest_tier = {
				strategic_region = scope:region
				value >= interest_tier_effect_can_join_diplomatic_plays_min_tier
			}
			can_go_to_war = yes
		}
		AND = {
			is_neutral = no
			OR = {
				has_treaty_alliance_with = { TARGET = scope:target_country }
				AND = {
					has_treaty_defensive_pact_with = { TARGET = scope:target_country }
					hidden_trigger = {
						scope:target_country = {
							scope:initiator != this
						}
					}
				}
				AND = {
					scope:target_country = {
						has_treaty_guarantee_independence_with = { TARGET = root }
					}
					hidden_trigger = {
						scope:target_country = {
							scope:initiator != this
						}
					}
				}
				scope:target_country = {
					has_treaty_support_independence_with = { TARGET = root }
				}
			}
		}
		AND = {
			has_strategic_region_interest_tier = {
				strategic_region = scope:region
				value >= interest_tier_effect_can_join_diplomatic_plays_min_tier
			}
			is_neutral = no
			any_diplomatic_play = {
				initiator = scope:enemy_country
				target = scope:target_country
				has_an_aggressive_war_goal = yes
			}
			any_diplomatic_play = {
				initiator = scope:target_country
				target = scope:enemy_country
				has_an_aggressive_war_goal = no
			}
		}
		is_direct_subject_of = scope:enemy_country
		scope:enemy_country = { is_direct_subject_of = root }
		is_direct_subject_of = scope:target_country
		scope:target_country = { is_direct_subject_of = root }
		scope:enemy_country = {
			has_modifier = condominium_modifier
			has_diplomatic_pact = { who = ROOT type = condominium is_initiator = no }
		}
		scope:target_country = {
			has_modifier = condominium_modifier
			has_diplomatic_pact = { who = ROOT type = condominium is_initiator = no }
		}
	}
	
	#Not has a non-aggression pact
	NOT = { 
		any_scope_treaty = { 
			binds = scope:enemy_country 
			any_scope_article = { has_type = non_aggression_pact } 
		} 
	}
	
	#Not demilitarised
	NOT = { has_law_or_variant = law_type:law_peasant_levies }
	
}

	hidden_trigger = {
		#has_dlc_feature = power_bloc_features
	}

		hidden_trigger = {
			used_principle_slots >= 3
		}
		AND = {
			power_bloc_rank <= 5
			num_power_bloc_members >= 5
		}
	}
}

# Evaluated when checking if a country can threaten naval hostilities in a treaty
# root = threatening country
# scope:other_country = threatened country
REPLACE_OR_CREATE:can_threaten_naval_hostilities = {
	navy_size >= 10

	custom_tooltip = {
		text = "THREATEN_NAVAL_HOSTILITIES_NAVY_MUST_BE_STRONGER"
		naval_combat_power > scope:other_country.naval_combat_power
	}

	custom_tooltip = {
		text = "THREATEN_NAVAL_HOSTILITIES_NOT_LANDLOCKED"
		scope:other_country = {
			any_scope_state = {
				is_coastal = yes
			}
		}
	}
	
	can_go_to_war = yes

	"relations(scope:other_country)" < relations_threshold:cordial
	NOR = {
		is_diplomatic_play_ally_of = scope:other_country
		any_scope_treaty = {
			binds = scope:other_country
			any_scope_article = {
				OR = {
					has_type = alliance
					has_type = defensive_pact
					has_type = guarantee_independence
				}
			}
		}
		is_direct_subject_of = scope:other_country
		scope:other_country = { is_direct_subject_of = root }
		any_scope_treaty = { 
			binds = scope:other_country 
			any_scope_article = { has_type = non_aggression_pact } 
		}
	}
}
