INJECT:ai_strategy_default = {

	diplomatic_play_support = {

		#Base value - Reputation
		add = {
			value = "scope:country.modifier:country_diplomatic_reputation_add"
			desc = "DIPLOMATIC_REPUTATION_REASON"
		}

		### Value of ENEMY ###
		
		### War Difficulty for THIS or TARGET ###
		
		#Don't support TARGET if THIS is not near TARGET
		if = {
			limit = {
				NOR = {
					has_strategic_adjacency = scope:country
					is_adjacent_to_country = scope:country
				}
			}
			add = {
				value = -100
				desc = "DIPLOMATIC_NOT_GEOGRAPHICALLY_CLOSE_TARGET_REASON"
			}
		}
		
		#Don't support TARGET if THIS is not near ENEMY
		if = {
			limit = {
				NOR = {
					has_strategic_adjacency = scope:enemy_country
					is_adjacent_to_country = scope:enemy_country
				}
			}
			add = {
				value = -100
				desc = "DIPLOMATIC_NOT_GEOGRAPHICALLY_CLOSE_ENEMY_REASON"
			}
		}
		
		### Political Alignment with ENEMY ###
		
		#Don't support TARGET if ENEMY is of similar culture to THIS
		if = {
			limit = {
				any_primary_culture = {
					OR = { 
						is_primary_culture_of = scope:enemy_country
						shares_non_heritage_trait_with_any_primary_culture = scope:enemy_country
					}
				}
			}
			add = {
				value = -50
				desc = "DIPLOMATIC_CULTURALLY_CLOSE_WITH_ENEMY_REASON"
			}
		}
		
		### Political Alignment with TARGET ###
		
		#Don't support TARGET if THIS is not of similar culture
		if = {
			limit = {
				NOT = { 
					any_primary_culture = {
						OR = { 
							is_primary_culture_of = scope:country
							shares_non_heritage_trait_with_any_primary_culture = scope:country
						}
					}
				}
			}
			add = {
				value = -25
				desc = "DIPLOMATIC_NOT_CULTURALLY_CLOSE_TARGET_REASON"
			}
		}
		
		#Different Ruling Coalition Ideologies - Common Enemy
		if = {
			limit = {
				OR = { 
					conservative_ruling_coalition = yes
					radical_right_wing_ruling_coalition = yes
				}
				scope:enemy_country = {
					OR = { 
						conservative_ruling_coalition = yes
						radical_right_wing_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_SAME_RULING_COALITION_IDEOLOGY_AS_ENEMY_REASON"
			}
		}
		else_if = {
			limit = {
				OR = { 
					liberal_ruling_coalition = yes
					progressive_ruling_coalition = yes
				}
				scope:enemy_country = {
					OR = { 
						liberal_ruling_coalition = yes
						progressive_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_SAME_RULING_COALITION_IDEOLOGY_AS_ENEMY_REASON"
			}
		}
		else_if = {
			limit = {
				OR = { 
					communist_ruling_coalition = yes
					socialist_ruling_coalition = yes
				}
				scope:enemy_country = {
					OR = { 
						communist_ruling_coalition = yes
						socialist_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_SAME_RULING_COALITION_IDEOLOGY_AS_ENEMY_REASON"
			}
		}
		
		#Different Ruling Coalition Ideologies - with TARGET
		if = {
			limit = {
				OR = { 
					conservative_ruling_coalition = yes
					radical_right_wing_ruling_coalition = yes
				}
				scope:country = {
					NOR = { 
						conservative_ruling_coalition = yes
						radical_right_wing_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_DIFFERENT_RULING_COALITION_IDEOLOGY_AS_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				OR = { 
					liberal_ruling_coalition = yes
					progressive_ruling_coalition = yes
				}
				scope:country = {
					NOR = { 
						liberal_ruling_coalition = yes
						progressive_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_DIFFERENT_RULING_COALITION_IDEOLOGY_AS_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				OR = { 
					communist_ruling_coalition = yes
					socialist_ruling_coalition = yes
				}
				scope:country = {
					NOR = { 
						communist_ruling_coalition = yes
						socialist_ruling_coalition = yes
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_DIFFERENT_RULING_COALITION_IDEOLOGY_AS_TARGET_REASON"
			}
		}
		
		### Relative Military Strength of THIS vs ENEMY ###
		
		#Don't support TARGET if THIS is militarily weaker than ENEMY
		if = {
			limit = {
				NOT = { has_modifier = nuclear_capability }
				scope:enemy_country = { has_modifier = nuclear_capability }
			}
			add = {
				value = -100
				desc = "DIPLOMATIC_ENEMY_HAS_NUKES_WHILE_WE_DONT_REASON"
			}
		}
		if = {
			limit = {
				OR = { 
					has_modifier = nuclear_capability
					scope:country = { has_modifier = nuclear_capability }
				}
				scope:enemy_country = { has_modifier = nuclear_capability }
			}
			add = {
				value = -50
				desc = "DIPLOMATIC_MUTUALLY_ASSURED_DESTRUCTION_WITH_ENEMY_REASON"
			}
		}
		
		### Relative Military Strength of TARGET vs ENEMY ###

		#Don't support TARGET if TARGET is militarily weaker than ENEMY
		if = {
			limit = {
				NOT = { has_modifier = nuclear_capability }
				scope:country = { 
					NOT = { has_modifier = nuclear_capability }
					scope:enemy_country = { has_modifier = nuclear_capability }
				}
			}
			add = {
				value = -100
				desc = "DIPLOMATIC_ENEMY_HAS_NUKES_WHILE_TARGET_DONT_REASON"
			}
		}
		if = {
			limit = {
				scope:country = {
					OR = { 
						has_modifier = nuclear_capability
						scope:country = { has_modifier = nuclear_capability }
					}
					scope:enemy_country = { has_modifier = nuclear_capability }
				}
			}
			add = {
				value = -50
				desc = "DIPLOMATIC_MUTUALLY_ASSURED_DESTRUCTION_WITH_ENEMY_REASON"
			}
		}
		
		#Trade laws reduce desire to join wars
		if = {
			limit = { 
				has_law_or_variant = law_type:law_mercantilism
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_TRADE_LAWS_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				has_law_or_variant = law_type:law_free_trade
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_TRADE_LAWS_THIS_REASON"
			}
		}

			limit = { 
				has_law_or_variant = law_type:law_protectionism

			add = {
				value = -5

			}
		}

		if = {
			limit = { 
				has_law_or_variant = law_type:law_peasant_levies
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_ARMY_MODEL_LAWS_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				has_law_or_variant = law_type:law_national_militia
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_ARMY_MODEL_LAWS_THIS_REASON"
			}
		}
		if = {
			limit = {
				ruler ?= { 
					interest_group = {
						law_stance = {
							law = law_type:law_nationalist_fp
							value >= approve
						}
					}
				}
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_FOREIGN_POLICY_LAWS_STANCE_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				ruler ?= { 
					interest_group = {
						law_stance = {
							law = law_type:law_regionalist_fp
							value >= approve
						}
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_FOREIGN_POLICY_LAWS_STANCE_THIS_REASON"
			}
		}

	}

}
