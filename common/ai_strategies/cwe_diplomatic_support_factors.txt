					country_rank = rank_value:minor_power
				}
				NOT = { has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes } }
			}
			add = {
				value = -15
				desc = "DIPLOMATIC_LOWER_RANK_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				scope:country = { country_rank < root.country_rank }
				scope:country = {
					country_rank = rank_value:insignificant_power
				}
				NOT = { has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes } }
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_LOWER_RANK_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				scope:country = { country_rank < root.country_rank }
				scope:country = {
					country_rank = rank_value:unrecognized_major_power
				}
				NOT = { has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes } }
			}
			add = {
				value = -25
				desc = "DIPLOMATIC_LOWER_RANK_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				scope:country = { country_rank < root.country_rank }
				scope:country = {
					country_rank = rank_value:unrecognized_regional_power
				}
				NOT = { has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes } }
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_LOWER_RANK_TARGET_REASON"
			}
		}
		else_if = {
			limit = {
				scope:country = { country_rank < root.country_rank }
				scope:country = {
					country_rank = rank_value:unrecognized_power
				}
				NOT = { has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes } }
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_LOWER_RANK_TARGET_REASON"
			}
		}
		
		#Infamy of THIS reduces urge to join wars
		if = {
			limit = { 
				infamy = infamy_threshold:infamous
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_INFAMY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				infamy = infamy_threshold:notorious
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_INFAMY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				infamy >= infamy_threshold:pariah
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_INFAMY_THIS_REASON"
			}
		}
		
		#Trade laws reduce desire to join wars
		if = {
			limit = { 
				has_law_or_variant = law_type:law_mercantilism
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_TRADE_LAWS_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				has_law_or_variant = law_type:law_free_trade
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_TRADE_LAWS_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				has_law_or_variant = law_type:law_protectionism
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_TRADE_LAWS_THIS_REASON"
			}
		}
		
		#Legitimacy reduce desire to join wars
		if = {
			limit = { 
				government_legitimacy < 20
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_LEGITIMACY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				government_legitimacy < 40
			}
			add = {
				value = -8
				desc = "DIPLOMATIC_LEGITIMACY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				government_legitimacy < 60
			}
			add = {
				value = -6
				desc = "DIPLOMATIC_LEGITIMACY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				government_legitimacy < 80
			}
			add = {
				value = -4
				desc = "DIPLOMATIC_LEGITIMACY_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				government_legitimacy < 100
			}
			add = {
				value = -2
				desc = "DIPLOMATIC_LEGITIMACY_THIS_REASON"
			}
		}
		
		### Isolationism Tendencies (INTRAstate) ###
		
		# AI is more neutral if leader is democratic
		if = {
			limit = { 
				ruler ?= { has_ideology = ideology:ideology_liberal }
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_RULER_PACIFISM_REASON"
			}
		}
		if = {
			limit = { 
				ruler ?= { has_ideology = ideology:ideology_democratic }
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_RULER_PACIFISM_REASON"
			}
		}
		if = {
			limit = { 
				ruler ?= { has_ideology = ideology:ideology_elitist }
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_RULER_PACIFISM_REASON"
			}
		}
		
		# AI is more neutral based on laws
		if = {
			limit = { 
				has_law_or_variant = law_type:law_peasant_levies
			}
			add = {
				value = -20
				desc = "DIPLOMATIC_ARMY_MODEL_LAWS_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				has_law_or_variant = law_type:law_national_militia
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_ARMY_MODEL_LAWS_THIS_REASON"
			}
		}
		if = {
			limit = {
				ruler ?= { 
					interest_group = {
						law_stance = {
							law = law_type:law_nationalist_fp
							value >= approve
						}
					}
				}
			}
			add = {
				value = -10
				desc = "DIPLOMATIC_FOREIGN_POLICY_LAWS_STANCE_THIS_REASON"
			}
		}
		else_if = {
			limit = { 
				ruler ?= { 
					interest_group = {
						law_stance = {
							law = law_type:law_regionalist_fp
							value >= approve
						}
					}
				}
			}
			add = {
				value = -5
				desc = "DIPLOMATIC_FOREIGN_POLICY_LAWS_STANCE_THIS_REASON"
			}
		}

		#Protecting Condominiums & Fighting rebelling condominiums
		if = {
			limit = { 	
				scope:enemy_country = { has_modifier = condominium_modifier }
				has_diplomatic_pact = { who = scope:enemy_country type = condominium is_initiator = yes }
				scope:country = { has_diplomatic_pact = { who = scope:enemy_country type = condominium is_initiator = yes } }
			}
			add = {
				value = 500
				desc = "DIPLOMATIC_CONDOMINIUM_REBELLING_REASON"
			}
		}
		if = {
			limit = { 
				scope:country = { has_modifier = condominium_modifier }
				has_diplomatic_pact = { who = scope:country type = condominium is_initiator = yes }
				NOT = { has_diplomatic_pact = { who = scope:enemy_country type = condominium is_initiator = yes } }
			}
			add = {
				value = 500
				desc = "DIPLOMATIC_CONDOMINIUM_TARGETED_REASON"
			}
		}

	}
