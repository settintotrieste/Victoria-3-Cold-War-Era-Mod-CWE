ig_trade_unions = {
	color = hsv{ 0 0.56 0.58 }
	texture = "gfx/interface/icons/ig_icons/trade_unions.dds"
	layer = "revolution_dynamic_trade_unions"
	index = 7
	
	ideologies = {
	
		#Economic Ideologies
		ideology_far_left_economy 			#General Economic Beliefs 
		ideology_manufacturing_focus 		#Economic Specialisation
		ideology_anti_environment 			#Environmental Laws
		ideology_nationalist 				#Foreign Economic Policy
		
		#Political Ideologies
		ideology_communitarian				#General Political Beliefs
		ideology_progressive				#Specific Political Beliefs
		ideology_communist					#Government Types 
		
		#Dynamic Ideologies
		#ideology_neutral_spending
		#ideology_neutral_population
		
	}
	
	character_ideologies = {
		ideology_autocratic
		ideology_power_hungry
		ideology_elitist
		ideology_democratic
		ideology_liberal
	}
	
	enable = {
		always = yes
	}
	
	# Deprecated; use on_enable effect to assign traits instead
	traits = {
		ig_trait_solidarity
		ig_trait_industrial_organizers
		ig_trait_work_to_rule
	}	
	
	on_enable = {
		ig:ig_trade_unions = {
			set_ig_trait = ig_trait:ig_trait_work_to_rule
			set_ig_trait = ig_trait:ig_trait_industrial_organizers
			set_ig_trait = ig_trait:ig_trait_solidarity
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
	
		#Only Accepted Pops can organise into IGs
		#pop_acceptance >= acceptance_status_3
		
		#Correct pop type
		OR = { 
			is_pop_type = machinists		#Skilled Labourers
			AND = { 						#Unskilled Labourers
				is_pop_type = laborers
				NOR = { 
					pop_employment_building_group = bg_subsistence_agriculture
					pop_employment_building_group = bg_subsistence_ranching
				}
			}
			is_pop_type = academics
		}
		
		NOT = { is_pop_type = capitalists }
		
		#Private Sector Only
		NOT = { pop_employment_building_group = bg_government }
		
	}
	
	pop_weight = {
		value = 0

		add = {
			desc = "URBAN_POP"
			if = {
				limit = {
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 50
				if = {
					limit = {
						strata = middle
					}
					subtract = {
						value = this.standard_of_living
						add = 10 # 15 SoL [Middling] will halve middle-strata pops' attraction to the unions
					}
				}
			}
		}

		add = {
			desc = "POP_MACHINISTS"
			if = {
				limit = {
					is_pop_type = machinists
				}
				value = 150
			}
		}

		add = {
			desc = "POP_LABORERS"
			if = {
				limit = {
					is_pop_type = laborers
				}
				if = {
					limit = {
						NOR = {
							pop_employment_building_group = bg_agriculture
							pop_employment_building_group = bg_ranching
							pop_employment_building_group = bg_plantations
						}
					}
					value = 100
				}
				else = {
					value = 50
				}
			}
		}

		add = {
			desc = "POP_CLERKS"
			if = {
				limit = {
					is_pop_type = clerks
				}
				value = 35
				subtract = standard_of_living
			}
		}

		add = {
			desc = "CONSCRIPTS"
			if = {
				limit = {
					is_pop_type = soldiers
					pop_employment_building_group = bg_conscription
				}
				value = 25
			}
		}

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader ?= {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_LOYALISTS"
				value = pop_loyalist_fraction
				divide = 4
				add = 1
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 4
				add = 1
			}
		}
	}
	
	monarch_weight = {
		value = 0.1 # this is just inherently unlikely
		# Monarch is not likely to adopt a marginal IG
		if = {
			limit = {
				ig_counts_as_marginal = yes
			}
			multiply = {
				value = 0.1
			}
		}
		# Monarch is more likely to adopt a Powerful IG
		if = {
			limit = {
				is_powerful = yes
			}
			multiply = {
				value = 2
			}
		}
		# More likely with Worker's Protections
		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_worker_protections
				}
			}
			multiply = {
				value = 2
			}
		}
		# Impossible with No Workers Rights
		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_no_workers_rights
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	agitator_weight = {
		# Agitators more likely to come from populist IG's rather than elitist
		value = 1.0
		# Agitators will never be part of a marginalized IG
		if = {
			limit = {
				owner.ig:ig_trade_unions ?= {
					ig_counts_as_marginal = yes
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	commander_weight = {
		value = 0.5
	}
	
	magnate_weight = {
		value = 0.1
	}		

	executive_weight = {
		value = 0.2
	}

	noble_chance = {
		value = 0.05
	}

	female_commander_chance = {
		value = 0.0
	}

	female_executive_chance = {
		value = 0.0
	}

	female_monarch_chance = {
		value = 0.025
	}

	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_full_womens_rights
				}
				law_stance = {
					law = law_type:law_full_womens_rights
					value >= approve
				}
			}
			add = {
				value = 0.1
			}
		}
	}

	female_agitator_chance = {
		value = 0.05

		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_some_womens_rights
				}

			}
			add = {
				value = 0.05
			}
		}

		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_full_womens_rights
				}

			}
			add = {
				value = 0.10
			}
		}
	}

	# Multiplier applied to the weight for an appropriate commander will be selected from the character pool to lead the IG
	# root = interest group
	commander_leader_weight = {
		value = 0.75
	}
	
	# Multiplier applied to the weight for an appropriate magnate will be selected from the character pool to lead the IG
	# root = interest group
	magnate_leader_weight = {
		value = 0.5
	}

	# Multiplier applied to the weight for an appropriate executive will be selected from the character pool to lead the IG
	# root = interest group
	executive_leader_weight = {
		value = 0.1
	}
	
}
