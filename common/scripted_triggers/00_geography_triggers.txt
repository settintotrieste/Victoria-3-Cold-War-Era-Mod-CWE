is_country_on_same_continent = {
	custom_description = {
		text = "is_country_on_same_continent"
		subject = this
		object = $TARGET$

		trigger_if = {
			limit = {
				this.capital.state_region = {
					is_in_geographic_region = geographic_region_europe
				}
			}
			$TARGET$.capital.state_region = {
				is_in_geographic_region = geographic_region_europe
			}
		}
		trigger_else_if = {
			limit = {
				this.capital.state_region = {
					is_in_geographic_region = geographic_region_americas
				}
			}
			$TARGET$.capital.state_region = {
				is_in_geographic_region = geographic_region_americas
			}
		}
		trigger_else_if = {
			limit = {
				this.capital.state_region = {
					is_in_geographic_region = geographic_region_africa
				}
			}
			$TARGET$.capital.state_region = {
				is_in_geographic_region = geographic_region_africa
			}
		}
		trigger_else_if = {
			limit = {
				this.capital.state_region = {
					is_in_geographic_region = geographic_region_asia
				}
			}
			$TARGET$.capital.state_region = {
				is_in_geographic_region = geographic_region_asia
			}
		}
	}
}
