superstate_je_trigger_conditions = {

	#Possible superstate country
	OR = { 
	
		#North American Union
		AND = { 
			country_is_in_north_america = yes 
			any_primary_culture = { 
				OR = { 
					has_discrimination_trait_group = heritage_group_european 
					has_discrimination_trait = indigenous_american_heritage
				}
			} 
		}
		
		#Central American Union
		AND = { 
			country_is_in_central_america = yes 
			any_primary_culture = { 
				OR = { 
					has_discrimination_trait_group = heritage_group_european
					has_discrimination_trait = african_heritage
					has_discrimination_trait = indigenous_american_heritage
				}
			} 
		}
	
		#South American Union
		AND = { 
			country_is_in_south_america = yes
			any_primary_culture = {
				OR = { 
					has_discrimination_trait_group = heritage_group_european
					has_discrimination_trait = indigenous_american_heritage
				}
			}
		}
	
		#Eurasian Union
		AND = { 
			OR = { 
				country_is_in_central_asia = yes 
				capital = { state_region = { is_homeland = cu:soviet } }
			}
			any_primary_culture = { 
				OR = { 
					has_discrimination_trait_group = heritage_group_european
					has_discrimination_trait_group = heritage_group_central_asian
				}
			}
		}
	
		#European Union
		AND = { 
			country_is_in_europe = yes 
			any_primary_culture = { 
				has_discrimination_trait_group = heritage_group_european 
			} 
		}
		
		#Arab Union
		AND = { 
			OR = { 
				country_is_in_middle_east = yes 
				country_is_in_africa = yes 
			}
			any_primary_culture = { 
				has_discrimination_trait = middle_eastern_heritage
				NOT = { has_discrimination_trait = jewish_group }
			} 
		}
		
		#African Union
		AND = { 
			country_is_in_africa = yes 
			any_primary_culture = { 
				has_discrimination_trait = african_heritage
			}
		}
		
		#East Asian Union
		AND = { 
			country_is_in_east_asia = yes 
			any_primary_culture = { 
				OR = { 
					has_discrimination_trait_group = heritage_group_east_asian
					has_discrimination_trait = north_asian_heritage
					has_discrimination_trait_group = heritage_group_south_asian
				}
			}
		}
		
		#Pacific Union
		AND = { 
			capital = { region = sr:region_oceania }
			any_primary_culture = { 
				OR = {
					has_discrimination_trait_group = heritage_group_european
					has_discrimination_trait = indigenous_oceanic_heritage
				}
			}
		}
		
		#South East Asian Union
		AND = { 
			country_is_in_southeast_asia = yes
			any_primary_culture = { 
				has_discrimination_trait_group = heritage_group_southeast_asian
			}
		}
		
	}
	
	#Tech researched
	has_technology_researched = tech_geopolitics_2
	
	#THIS is a sovereign democratic country
	is_a_democracy = yes
	is_sufficiently_independent = yes
	is_at_war = no
	
	#Regionalist laws
	NOT = { has_law_or_variant = law_type:law_isolationism }

}
