diplo_play_war_check_cwe = {

	#THIS does not have restricted war rights
	NOR = { 
		has_law = law_type:law_no_war
		has_law = law_type:law_defensive_war
	}
	
	#THIS has some army
	NOT = { has_law = law_type:law_peasant_levies }
	
	#Ruler is jingoistic OR is a dictatorship and has a war mongering IG
	OR = { 
	
		AND = { 
			exists = ruler
			ruler ?= {
				OR = {
					has_ideology = ideology:ideology_autocratic
					has_ideology = ideology:ideology_power_hungry
				}
			}
		}
		
		AND = {
			is_a_dictatorship = yes
			hidden_trigger = { is_player = yes }
			any_interest_group = {
				is_in_government = yes
				is_powerful = yes
				leader ?= { 
					OR = {
						has_ideology = ideology:ideology_autocratic
						has_ideology = ideology:ideology_power_hungry
					}
				}
			}
		}
	
	}
	
	#No non-aggression_pact with TARGET
	has_no_non_aggression_pact_with_target = yes
	
	#Parliament and Cabinet approves
	government_has_legislative_simple_majority = yes
	
	#Bad relations with TARGET
	custom_tooltip = {
		text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS
		scope:target_country.relations:root < relations_threshold:poor
	}

}

government_has_legislative_simple_majority = {
	
	#government_legitimacy > 50
	
	OR = {
	
		#If bicameral
		trigger_if = {
			limit = { has_journal_entry = je_upper_house }
			AND = { 
				has_upper_house_support_simple_majority = yes
				has_lower_house_support_simple_majority = yes
			}
		}
		
		#If unicameral
		trigger_if = {
			limit = { NOT = { has_journal_entry = je_upper_house } }
			has_lower_house_support_simple_majority = yes
		}
		
		has_modifier = executive_order_issued
		
		has_modifier = state_of_emergency_issued
		
	}
	
	custom_tooltip = {
		text = has_no_low_cabinet_unity.tt
		NOT = { has_variable = low_cabinet_unity }
	}
	
}

	NOR = { 
		has_law = law_type:law_no_war
		has_law = law_type:law_defensive_war
	}
	
	#THIS has some army
	NOT = { has_law = law_type:law_peasant_levies }
	
	#Ruler is jingoistic OR is a dictatorship and has a war mongering IG
	OR = { 
	
		AND = { 
			exists = ruler
			ruler ?= {
				OR = {
					has_ideology = ideology:ideology_autocratic
					has_ideology = ideology:ideology_power_hungry
				}
			}
		}
		
		AND = {
			is_a_dictatorship = yes
			hidden_trigger = { is_player = yes }
			any_interest_group = {
				is_in_government = yes
				is_powerful = yes
				leader ?= { 
					OR = {
						has_ideology = ideology:ideology_autocratic
						has_ideology = ideology:ideology_power_hungry
					}
				}
			}
		}
	
	}
	
	#Parliament and Cabinet approves
	government_has_legislative_simple_majority = yes
	
	#No non-aggression_pact with TARGET
	#has_no_non_aggression_pact_with_target = yes
	
	#Bad relations with TARGET or INITIATOR
	custom_tooltip = {
		text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS
		OR = { 
			scope:target_country.relations:root < relations_threshold:poor
