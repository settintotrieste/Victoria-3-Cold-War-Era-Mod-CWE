	background = "gfx/interface/icons/company_icons/company_backgrounds/comp_illu_manufacturing_heavy.dds" icon = "gfx/interface/icons/company_icons/historical_company_icons/skoda.dds"
	
	flavored_company = yes
	
	preferred_headquarters = { STATE_BOHEMIA }
	
	building_types = {
		building_steel_factory
		building_automobiles_factory
		building_artillery_factory
	}
	
	extension_building_types = {
		building_engines_factory
	}
	
	prestige_goods_trigger = {  } possible_prestige_goods = {
		prestige_good_generic_steel
		prestige_good_generic_automobiles
	}
	
	potential = {
		has_interest_marker_in_region = region_south_germany
	}
	
	attainable = {
		hidden_trigger = { # Hide attainable if it's always fulfilled by possible
			any_scope_state = {
				state_region = s:STATE_BOHEMIA
			}
		}
	}
	
	possible = { 
		any_scope_state = {
			state_region = s:STATE_BOHEMIA
			is_incorporated = yes
			any_scope_building = {
				OR = {
					is_building_type = building_steel_factory
					is_building_type = building_small_arms_factory
					is_building_type = building_artillery_factory
				}
				level >= 10
			}
		}
	}
	
	prosperity_modifier = {
		character_advancement_speed_add = 1
		building_tanks_factory_throughput_add = 0.1
	}	
	
	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
