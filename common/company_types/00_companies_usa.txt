	texture = "gfx/interface/icons/company_icons/historical_company_icons/american_standard_oil.dds"
	
	flavored_company = yes
	
	building_types = {  
		building_oil_rig
		building_railway
	}
	
	potential = {
		has_interest_marker_in_region = region_the_midwest
	}

		any_scope_state = {
			region = sr:region_the_midwest

			}
		}

	}
	
	prosperity_modifier = {
		building_railway_throughput_add = 0.2
	}

		value = 3 # Higher base value for flavored companies
	}
}

	}
	
	potential = {

			country_has_primary_culture = cu:yankee
			country_has_primary_culture = cu:dixie
		}
		OR = {
			has_interest_marker_in_region = region_central_america
			has_interest_marker_in_region = region_caribbean
			has_interest_marker_in_region = region_mexico
			has_interest_marker_in_region = region_gran_colombia
		}
	}
	
	possible = {
		OR = {
			any_scope_state = {
				OR = {
					state_is_in_south_america = yes
					state_is_in_central_america = yes
				}
				any_scope_building = {
					is_building_type = building_banana_plantation
					level > 5
				}
			}
		}
		any_subject_or_below = {
			OR = {
				country_is_in_south_america = yes
				country_is_in_central_america = yes
			}
			any_scope_state = {
				any_scope_building = {
					is_building_type = building_banana_plantation
					level > 5
				}
			}
		}
	}
	
	prosperity_modifier = {
		building_group_bg_plantations_throughput_add = 0.2
		country_infamy_decay_mult = -0.25
	}	
	
	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
