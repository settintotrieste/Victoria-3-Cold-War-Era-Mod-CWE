start_nuclear_weapons_program_decision = {

	is_shown = {
	
		NOT = { has_modifier = nuclear_capability }
		
		NOT = { has_variable = nuclear_program_started }
		
		#THIS must not be a revolutionary
		is_revolutionary = no is_secessionist = no
		
	}

	possible = {
	
		#THIS has a jingoistic government
		has_jingoistic_ruler = yes
		
		#Must be a sovereign state
		is_sufficiently_independent = yes
		
		#THIS is not legally barred from starting a nuclear program
		has_law = law_type:law_wmd_allowed
		
		#THIS is not demilitarised
		NOT = { has_law = law_type:law_peasant_levies }
		
		#THIS is not broke
		in_default = no
		
		#THIS is not in the midst of a revolution
		has_revolution = no
		
		#Has nuclear weapons tech
		has_technology_researched = tech_military_nuclear_2
		
	}
	
	when_taken = {
		
		#Mark country as having started a nuclear program
		set_variable = nuclear_program_started
		
		#Add JE
		add_journal_entry = { type = je_develop_nuclear_weapons }
		
		#For AI
		set_variable = {
			name = nuclear_program_started_AI_timer
			months = 60
		}
		
		#Hostile countries become more hostile
		every_country = {
			limit = {
				OR = {
				
					AND = {
						country_rank > root.country_rank
						can_go_to_war = yes
						OR = { has_modifier = nuclear_capability has_variable = nuclear_program_started }
			
						OR = { 
							has_attitude = {
								who = ROOT
								attitude <= antagonistic
							}
							relations:root < relations_threshold:poor
						}
					}
					
					AND = {
						is_sufficiently_independent = yes
						has_diplomatic_pact = { who = ROOT type = rivalry }
						OR = { has_modifier = nuclear_capability has_variable = nuclear_program_started }
					}
					
				}
			}
			change_relations = {
				country = ROOT
				value = -100
			}
		}
		
		#International community does not like this
		change_infamy = 10
		
	}

	ai_chance = {
		value = 0
		
		if = {
		
			limit = {
			
				OR = { 
				
					#There is a hostile country with rank greater than THIS that has nukes OR is developing nukes
					any_country = {
						country_rank > root.country_rank
						can_go_to_war = yes
						OR = { has_modifier = nuclear_capability has_variable = nuclear_program_started }
			
						OR = { 
							has_attitude = {
								who = ROOT
								attitude <= antagonistic
							}
							relations:root < relations_threshold:poor
						}
					}
				
					#There is a neigbouring country that is hostile against THIS that has nukes OR is developing nukes
					any_neighbouring_state = {
						owner = {
							can_go_to_war = yes
							OR = { has_modifier = nuclear_capability has_variable = nuclear_program_started }
			
							OR = { 
								has_attitude = {
									who = ROOT
									attitude <= antagonistic
								}
								relations:root < relations_threshold:poor
							}
						}
					}
					
					#Rival has nukes OR is developing nukes
					any_rival_country = { 
						OR = { has_modifier = nuclear_capability has_variable = nuclear_program_started }
					}
					
					#THIS is a superpower or great power
					country_rank >= rank_value:great_power
				
				}
				
				#THIS is a superpower or great power
				country_rank >= rank_value:major_power
				
			}
			
			add = 100
			
		}

	}
	
}

join_nuclear_non_proliferation_treaty_decision = {

	is_shown = {
		
		#THIS has not already signed the NPT
		NOT = { has_variable = signed_nuclear_non_proliferation_treaty }
		
		#Date of establishment
		game_date >= 1968.7.1
		
	}

	possible = {
		
		#Not currently developing WMD
		NOT = { has_journal_entry = je_develop_nuclear_weapons }
		
		#THIS is sufficiently independent
		is_sufficiently_independent = yes
		
		#THIS is not currently at war
		is_at_war = no
		
		#THIS is a recognised country
		has_modifier = unga_member
		
	}
	
	when_taken = {
	
		custom_tooltip = {
			text = signed_NPT_1_tt
			
			#Mark THIS as having signed the NPT
			set_variable = signed_nuclear_non_proliferation_treaty
			
			#No cheesing
			set_variable = {
				name = nuclear_non_proliferation_treaty_timer
				years = 10
			}
			
		}
		
		#Disable nuclear program & weapons
		if = {
			limit = { has_law = law_type:law_wmd_allowed }
			activate_law = law_type:law_wmd_banned
			change_infamy = -50
			every_country = {
				limit = { OR = { has_strategic_adjacency = ROOT country_rank >= rank_value:great_power } }
				change_relations = {
					country = ROOT
					value = 50
				}
			}
		}
		else = {
			change_infamy = -10
			every_country = {
				limit = { OR = { has_strategic_adjacency = ROOT country_rank >= rank_value:great_power } }
				change_relations = {
					country = ROOT
					value = 10
				}
			}
		}
		
	}

	ai_chance = {
		value = 0
		
		if = {
		
			limit = {
			
				#Country is not faced with a nuclear threat from somewhere
				NOR = {
					country_rank >= rank_value:great_power
					any_rival_country = { 
						OR = { 
							has_modifier = nuclear_capability
							has_variable = nuclear_program_started
						}
					}
					any_country = {
						relations:root < relations_threshold:poor
						OR = { 
							has_modifier = nuclear_capability
							has_variable = nuclear_program_started
						}
					}
				}
				
				#Country is democratic
				is_some_kind_of_democracy = yes
				
				#Ruler is not jingoistic
				NOT = { has_jingoistic_ruler = yes }
				
			}
			
			add = 100
			
		}
		
		#Special
		if = {
			limit = {
				c:JAP ?= this
				is_a_democracy = yes
			}
			add = 100
		}
		
	}
	
}

leave_nuclear_non_proliferation_treaty_decision = {

	is_shown = {
		
		#THIS has already signed the NPT
		has_variable = signed_nuclear_non_proliferation_treaty
		
	}

	possible = {
		
		#THIS must have upheld the treaty for at least 10 years
		custom_tooltip = {
			text = signed_NPT_2_tt
			NOT = { has_variable = nuclear_non_proliferation_treaty_timer }
		}
		
		#THIS is sufficiently independent
		is_sufficiently_independent = yes
		
		#THIS is not currently at war
		is_at_war = no
		
	}
	
	when_taken = {
	
		#Mark THIS as having signed the NPT
		remove_variable = signed_nuclear_non_proliferation_treaty
		
		#Everyone is not happy with THIS
		change_infamy = 20
		every_country = {
			limit = { OR = { has_strategic_adjacency = ROOT country_rank >= rank_value:great_power } }
			change_relations = {
				country = ROOT
				value = -25
			}
		}
		
	}

	ai_chance = {
		value = 0
		
		if = {
		
			limit = {
			
				#Country is faced with a nuclear threat from somewhere
				OR = {
					any_rival_country = { 
						OR = { 
							has_modifier = nuclear_capability
							has_variable = nuclear_program_started
						}
					}
					any_country = {
						relations:root < relations_threshold:poor
						OR = { 
							has_modifier = nuclear_capability
							has_variable = nuclear_program_started
						}
					}
				}
				
				#Ruler is jingoistic
				has_jingoistic_ruler = yes
				
			}
			
			add = 100
			
		}
		
	}
	
}
