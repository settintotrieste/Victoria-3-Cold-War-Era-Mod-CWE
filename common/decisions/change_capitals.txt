change_capital_BRZ_decision = {

	is_shown = {
		
		#Correct country
		c:BRZ ?= this
		
		#Correct date
		game_date >= 1960.4.21
		
		#Owns new capital
		s:STATE_GOIAS = {
			any_scope_state = {
				owner = ROOT
				is_capital = no
			}
		}
		
	}

	possible = {
	
		#Country is not at war
		is_at_war = no
		
		#Country can afford new capital
		gold_reserves > 1000000
		
	}
	
	when_taken = {
	
		#Shift capital
		set_capital = STATE_GOIAS
	
		#Cost of building new capital
		add_treasury = -1000000
		
	}

	ai_chance = {
		base = 100		
	}
	
}
