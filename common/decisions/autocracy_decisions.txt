enact_cult_of_personality_decision = {

	is_shown = {
		
		#No cult
		NOT = { 
			active_law:lawgroup_legislature ?= {
				has_amendment = amendment_type:amendment_cult_of_personality
			}
		}
		
	}

	possible = {
		
		#Autocratic Regime
		OR = {
			has_law_or_variant = law_type:law_appointed_legislature
			has_law_or_variant = law_type:law_technocratic_legislature
		}
		OR = {
			has_law_or_variant = law_type:law_outlawed_dissent
			has_law_or_variant = law_type:law_censorship
		}
		
		#Autocratic Ruler
		any_scope_character = {
			is_ruler_of_own_country = yes
			character_is_autocratic = yes 
		}
		
		#Sufficient Government support
		government_has_legislative_super_majority = yes
		
	}
	
	when_taken = {
		
		#Amendments
		ruler.interest_group ?= { save_scope_as = ruler_ig }
		active_law:lawgroup_legislature ?= {
			add_amendment = {
				type = amendment_cult_of_personality
				sponsor = scope:ruler_ig
			}
		}
		
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				ruler ?= {
					popularity <= -50
				}
			}
			add = 10
		}
	}
	
}

enact_corporate_state_decision = {

	is_shown = {
		
		#No corporate state
		NOT = { 
			active_law:lawgroup_governance_principles ?= {
				has_amendment = amendment_type:amendment_corporate_state
			}
		}
		
		#Appropriate Government Type
		OR = {
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOS_monarch
			has_law_or_variant = law_type:law_HOS_nationalist_leader
		}
		
	}

	possible = {
		
		#Autocratic Regime
		OR = {
			has_law_or_variant = law_type:law_appointed_legislature
			has_law_or_variant = law_type:law_technocratic_legislature
		}
		OR = {
			has_law_or_variant = law_type:law_outlawed_dissent
			has_law_or_variant = law_type:law_censorship
		}
		
		#Autocratic Ruler
		any_scope_character = {
			is_ruler_of_own_country = yes
			character_is_autocratic = yes
			OR = { 
				is_interest_group_type = ig_industrialists
				is_interest_group_type = ig_armed_forces
				is_interest_group_type = ig_intelligentsia
			}
		}
		
		#Sufficient Government support
		government_has_legislative_super_majority = yes
		
	}
	
	when_taken = {
		
		#Amendments
		ruler.interest_group ?= { save_scope_as = ruler_ig }
		active_law:lawgroup_governance_principles ?= {
			add_amendment = {
				type = amendment_corporate_state
				sponsor = scope:ruler_ig
			}
		}
		
	}

	ai_chance = {
		value = 1
	}
	
}
