ban_party_hopping_decision = {

	is_shown = {
	
		#Has parties
		OR = { 
			has_law_or_variant = law_type:law_parties_allowed
			has_law_or_variant = law_type:law_dominant_party_state 
		}
		
		#Not already has amendment
		NOT = {
			active_law:lawgroup_political_parties ?= {
				has_amendment = amendment_type:amendment_party_hopping_banned
			}
		}
		
	}

	possible = {
		
		#Government is sufficiently strong
		government_has_legislative_super_majority = yes
		
	}
	
	when_taken = {
	
		#Add amendment
		ruler ?= { interest_group = { save_scope_as = ruler_ig } }
		active_law:lawgroup_political_parties ?= {
			add_amendment = {
				type = amendment_party_hopping_banned
				sponsor = scope:ruler_ig
			}
		}
		
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				ruler ?= { character_is_autocratic = yes }
			}
			add = 10
		}
	}
	
}
