grant_washington_dc_statehood_decision = {

	is_shown = {
		
		#Owns new capital
		owns_entire_state_region = STATE_DISTRICT_OF_COLUMBIA
		s:STATE_DISTRICT_OF_COLUMBIA ?= {
			any_scope_state = {
				has_variable = election_excluded_state
				is_incorporated = yes
			}
		}
		
	}

	possible = {
	
		#Standard checks
		is_at_war = no
		in_election_campaign = no
		
		#Government is sufficiently strong
		government_has_legislative_super_majority = yes
		
	}
	
	when_taken = {
	
		#Grant Statehood
		random_scope_state = {
			limit = { 
				state_region = s:STATE_DISTRICT_OF_COLUMBIA
				has_variable = election_excluded_state
			}
			remove_variable = election_excluded_state
			add_modifier = { 
				name = national_reconciliation 
				years = 5 
				is_decaying = yes 
			}
			add_loyalists_in_state = {
				value = very_large_radicals
			}
		}
		
		#Please all opposition IGs
		every_interest_group = {
			limit = { is_in_government = no }
			save_scope_as = ig_scope
			add_loyalists = {
				value = small_radicals
				interest_group = scope:ig_scope
			}
		}
		
	}

	ai_chance = {
		value = 10
		if = {
			limit = {
				c:USA ?= this
			}
			add = -100
		}
	}
	
}
