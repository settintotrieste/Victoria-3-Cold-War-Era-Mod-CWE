government_ceremonial_body_custom_title = {
	type = country
	first_valid = yes

	#Monarchy
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_monarch
			NOT = { has_law_or_variant = law_type:law_HOG_prime_minister }
		}
		localization_key = ceremonial_monarchy_name
	}
	
	#Privy Council
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_monarch
			has_law_or_variant = law_type:law_HOG_prime_minister
		}
		localization_key = ceremonial_privy_council_name
	}
	
	#President
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOG_prime_minister
		}
		localization_key = ceremonial_presidency_name
	}
	
	#Presidential Council
	text = {
		trigger = {
			country_has_primary_culture = cu:german
			has_law_or_variant = law_type:law_HOS_president
		}
		localization_key = ceremonial_council_of_elders_name
	}
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_president
		}
		localization_key = ceremonial_presidential_council_name
	}
	
	#Guardian Council
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_religious_leader
		}
		localization_key = ceremonial_guardian_council_name
	}
	
	#Military Council
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_nationalist_leader
		}
		localization_key = ceremonial_military_council_name
	}
	
	#State Council
	text = {
		trigger = {
			country_has_primary_culture = cu:chinese
			capital = { state_region = { is_homeland = cu:chinese } }
			has_law_or_variant = law_type:law_HOS_communist_leader
		}
		localization_key = ceremonial_State_Council_name
	}
	
	#Presidium
	text = {
		trigger = {
			has_law_or_variant = law_type:law_HOS_communist_leader
		}
		localization_key = ceremonial_presidium_name
	}
	
}
