religious_leader_title_custom_title = {
	type = country
	first_valid = yes
	
	#Archbishop
	text = {
		trigger = {
			OR = { 
				country_has_state_religion = rel:catholic
				country_has_state_religion = rel:protestant
			}
		}
		localization_key = custom_archbishop_title_name
	}
	
	#Patriarch
	text = {
		trigger = {
			OR = { 
				country_has_state_religion = rel:orthodox
				country_has_state_religion = rel:oriental_orthodox
			}
		}
		localization_key = custom_patriarch_title_name
	}
	
	#Grand Mufti
	text = {
		trigger = {
			OR = { 
				country_has_state_religion = rel:sunni
				country_has_state_religion = rel:ibadi
			}
		}
		localization_key = custom_grand_mufti_title_name
	}
	
	#Grand Ayatollah
	text = {
		trigger = {
			country_has_state_religion = rel:shiite
		}
		localization_key = custom_grand_ayatollah_title_name
	}
	
	#Chief Rabbi
	text = {
		trigger = {
			country_has_state_religion = rel:jewish
		}
		localization_key = custom_chief_rabbi_title_name
	}
	
	#Abbot
	text = {
		trigger = {
			OR = { 
				country_has_state_religion = rel:mahayana
				country_has_state_religion = rel:gelugpa
				country_has_state_religion = rel:theravada
			}
		}
		localization_key = custom_abbot_title_name
	}
	
	#High Priest
	text = {
		trigger = {
			NOT = { country_has_state_religion = rel:atheist }
		}
		localization_key = custom_high_priest_title_name
	}
	
}
