					this = c:GBR
				}
				AND = {
					has_law_or_variant = law_type:law_HOS_monarch # "Hans/Hennes Majestäts Skepp" only makes sense for monarchies

				}
			}
		}

			has_law_or_variant = law_type:law_HOS_monarch # "Seiner Majestät Schiff" only makes sense for monarchies

					this = c:PRU

			has_law_or_variant = law_type:law_HOS_monarch # "Nava Majestăţii Sale" only makes sense for monarchies

			exists = c:HAW

		}
		localization_key = flotilla_prefix_hhms

			exists = c:TUR
			this = c:TUR
		}
		localization_key = flotilla_prefix_tcg 
	}
	text = {
		trigger = {
			OR = { # "Armada de la República Mexicana" only makes sense if Mexico is a republic
				has_law_or_variant = law_type:law_HOS_president
				has_law_or_variant = law_type:law_HOS_nationalist_leader
				has_law_or_variant = law_type:law_HOS_communist_leader

	}

			exists = c:CSA

		}
		localization_key = flotilla_prefix_css
	}

    text = {
		trigger = {
			always = yes
		}
		localization_key = flotilla_prefix_blank # Many countries used no prefix at all
	}
