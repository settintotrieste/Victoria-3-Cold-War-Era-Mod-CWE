country_power = {
	type = country
	random_valid = no

	text = {
		trigger = {
			country_rank >= rank_value:great_power
		}
		localization_key = custom_loc_great_power
	}

	text = {
		trigger = {
			country_rank = rank_value:major_power
		}
		localization_key = custom_loc_major_power
	}

	text = {
		trigger = {
			country_rank = rank_value:minor_power
		}
		localization_key = custom_loc_minor_power
	}
	
	text = {
		trigger = {
			country_rank = rank_value:insignificant_power
		}
		localization_key = custom_loc_insignificant_power
	}
}

enactment_phase_0 = {
	type = country

	text = {
		localization_key = enactment_phase_0_courep
		trigger = {
			has_law_or_variant = law_type:law_HOS_communist_leader
			has_law_or_variant = law_type:law_appointed_legislature
		}
	}
	
	text = {
		localization_key = enactment_phase_0_autocracy
		trigger = {
			OR = { 
				has_law_or_variant = law_type:law_HOS_nationalist_leader 
				has_law_or_variant = law_type:law_HOS_president 
			}
			has_law_or_variant = law_type:law_appointed_legislature
		}
	}

	text = {
		localization_key = enactment_phase_0_demrep
		trigger = {
			has_law_or_variant = law_type:law_HOS_president
			NOT = { has_law_or_variant = law_type:law_appointed_legislature }
		}
	}
	
	text = {
		localization_key = enactment_phase_0_conmon
		trigger = {
			has_law_or_variant = law_type:law_HOS_monarch
			NOT = { has_law_or_variant = law_type:law_appointed_legislature }
		}
	}
	
	text = {
		localization_key = enactment_phase_0_mon
		trigger = {
			has_law_or_variant = law_type:law_HOS_monarch
			has_law_or_variant = law_type:law_appointed_legislature
		}
	}

	text = {
		localization_key = enactment_phase_0_generic
		trigger = {
			always = yes
		}
	}
}
