viceroy_custom_title = {
	type = country
	first_valid = yes

	#Governor-General
	text = {
		trigger = {
			ROOT ?= {
				OR = {
					country_tier = empire
					country_tier = hegemony
					country_has_primary_culture = cu:indian
				}
				NAND = {
					c:HND ?= this
					aggressive_diplomatic_plays_permitted = yes
				}
			}
		}
		localization_key = custom_viceroy_title
	}
	text = {
		trigger = {
			ROOT ?= {
				country_has_primary_culture = cu:malay
			}
		}
		localization_key = custom_high_commissioner_title
	}
	text = {
		trigger = {
			ROOT ?= {
				country_has_primary_culture = cu:singaporean
				is_subject_type = subject_type_special_administrative_region
			}
		}
		localization_key = custom_Yang_di_Pertuan_Negara_title
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = RULER_TITLE_GOVERNOR_GENERAL
	}
	
}
