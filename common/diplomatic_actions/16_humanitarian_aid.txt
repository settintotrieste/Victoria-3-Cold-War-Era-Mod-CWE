		weekly_effect = {

				}

						limit = {
							NOT = {

									owes_obligation_to = root			
								}

							998 = {}

		}

			trigger = {

						}

		requirement_to_maintain = {

		accept_score = {

			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"
	
				if = {
					limit = { scope:actor = { country_rank < scope:target_country.country_rank } }
					value = -5
				}
				
				if = {
					limit = { scope:actor = { country_rank > scope:target_country.country_rank } }
					value = 10
				}				
			}				
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_GDP"
				value = "scope:actor.ai_gdp_comparison(root)"
				multiply = -1
				max = 20
				min = -20
			}
			
			if = {
				limit = { scope:target_country.scaled_debt >= 0.05 }
				
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_DEBT"

					multiply = 30	
					min = 1
				}					
			}
			else_if = {
				limit = { 
					scope:target_country.scaled_gold_reserves >= 0.05 
				}

					desc = "DIPLOMATIC_ACCEPTANCE_GOLD_RESERVES"				
					value = scope:target_country.scaled_gold_reserves
					multiply = -20
				}
			}
			
			#TARGET has similar ideology as THIS
			if = {
				limit = {
					scope:actor = {
						NOR = {
							AND = { 
								is_some_kind_of_democracy = yes
								scope:target_country = { is_some_kind_of_democracy = yes }
							}
							AND = { 
								is_a_monarchy = yes
								scope:target_country = { is_a_monarchy = yes }
							}
							AND = { 
								is_a_communist_government = yes
								scope:target_country = { is_a_communist_government = yes }
							}
							AND = { 
								is_a_nationalist_government = yes
								scope:target_country = { is_a_nationalist_government = yes }
							}
							AND = { 
								is_a_theocratic_government = yes
								scope:target_country = { is_a_theocratic_government = yes }
							}
						}
					}
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION_NEGATIVE"
					value = -25
				}
			}
			else_if = {
				limit = {
					scope:actor = {
						OR = {
							AND = { 
								is_some_kind_of_democracy = yes
								scope:target_country = { is_some_kind_of_democracy = yes }
							}
							AND = { 
								is_a_monarchy = yes
								scope:target_country = { is_a_monarchy = yes }
							}
							AND = { 
								is_a_communist_government = yes
								scope:target_country = { is_a_communist_government = yes }
							}
							AND = { 
								is_a_nationalist_government = yes
								scope:target_country = { is_a_nationalist_government = yes }
							}
							AND = { 
								is_a_theocratic_government = yes
								scope:target_country = { is_a_theocratic_government = yes }
							}
						}
						OR = {
							AND = { 
								left_wing_ruling_coalition = yes
								scope:target_country = { left_wing_ruling_coalition = yes }
							}
							AND = { 
								right_wing_ruling_coalition = yes
								scope:target_country = { right_wing_ruling_coalition = yes }
							}
						}
					}
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION"
					value = 25
				}
			}
			
		}

		# The format here is as if scope:actor was asking for a humanitarian_aid from root
		junior_accept_score = {
			value = 0

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = -10
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"

				if = {
					limit = { scope:actor = { country_rank > scope:target_country.country_rank } }
					value = 10
				}

				if = {
					limit = { scope:actor = { country_rank < scope:target_country.country_rank } }
					value = -20
				}				
			}				

			subtract = {
				value = 10
				multiply = scope:actor.num_income_transfer_pacts
			}

			subtract = {
				value = 10
				multiply = num_income_transfer_pacts
			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_GDP"
				value = "scope:actor.ai_gdp_comparison(root)"
				add = 2
				multiply = -10
				min = -100
				max = 100
			}
			
			if = {
				limit = { ROOT.scaled_debt >= 0.05 }
				
				subtract = {
					desc = "DIPLOMATIC_ACCEPTANCE_DEBT"
					value = ROOT.scaled_debt
					multiply = 100	
					min = 1
				}					
			}
			else_if = {
				limit = { ROOT.scaled_gold_reserves >= 0.05 }
				
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_GOLD_RESERVES"				
					value = ROOT.scaled_gold_reserves
					multiply = -25
				}						
			}
			
			#TARGET has similar ideology as THIS
			if = {
				limit = {
					scope:actor = {
						NOR = {
							AND = { 
								is_some_kind_of_democracy = yes
								scope:target_country = { is_some_kind_of_democracy = yes }
							}
							AND = { 
								is_a_monarchy = yes
								scope:target_country = { is_a_monarchy = yes }
							}
							AND = { 
								is_a_communist_government = yes
								scope:target_country = { is_a_communist_government = yes }
							}
							AND = { 
								is_a_nationalist_government = yes
								scope:target_country = { is_a_nationalist_government = yes }
							}
							AND = { 
								is_a_theocratic_government = yes
								scope:target_country = { is_a_theocratic_government = yes }
							}
						}
					}
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION_NEGATIVE"
					value = -25
				}
			}
			else_if = {
				limit = {
					scope:actor = {
						OR = {
							AND = { 
								is_some_kind_of_democracy = yes
								scope:target_country = { is_some_kind_of_democracy = yes }
							}
							AND = { 
								is_a_monarchy = yes
								scope:target_country = { is_a_monarchy = yes }
							}
							AND = { 
								is_a_communist_government = yes
								scope:target_country = { is_a_communist_government = yes }
							}
							AND = { 
								is_a_nationalist_government = yes
								scope:target_country = { is_a_nationalist_government = yes }
							}
							AND = { 
								is_a_theocratic_government = yes
								scope:target_country = { is_a_theocratic_government = yes }
							}
						}
						OR = {
							AND = { 
								left_wing_ruling_coalition = yes
								scope:target_country = { left_wing_ruling_coalition = yes }
							}
							AND = { 
								right_wing_ruling_coalition = yes
								scope:target_country = { right_wing_ruling_coalition = yes }
							}
						}
					}
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION"
					value = 25
				}
			}
			
		}
	
		propose_score = {
			value = 10
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			net_fixed_income > 0	
			has_attitude = {
				who = scope:target_country
				attitude = protective
			}
			OR = {
				num_income_transfer_pacts < 1
				AND = {
					num_income_transfer_pacts < 2
					income >= 200000
				}
				AND = {
					num_income_transfer_pacts < 3
					income >= 400000
				}				
			}
			
			OR = {
				root = {
					owes_obligation_to = scope:target_country			
				}	
				NOT = { 
					scope:target_country = {
						owes_obligation_to = root			
					}						
				}
				AND = {
					is_at_war = no
					scope:target_country = { is_at_war = yes }
				}			
			}		
			# We want to have at least five times their income, and at least 50k income in general
			OR = {
				AND = {
					income >= 50000
					net_fixed_income >= 4000
					scope:target_country.income <= 10000
				}
				AND = {
					income >= 75000
					net_fixed_income >= 6000
					scope:target_country.income <= 15000
				}				
				AND = {
					income >= 100000
					net_fixed_income >= 8000
					scope:target_country.income <= 20000
				}
				AND = {
					income >= 125000
					net_fixed_income >= 10000
					scope:target_country.income <= 25000
				}
				AND = {
					income >= 150000
					net_fixed_income >= 12000
					scope:target_country.income <= 30000
				}	
				AND = {
					income >= 175000
					net_fixed_income >= 14000
					scope:target_country.income <= 35000
				}
				AND = {
					income >= 200000
					net_fixed_income >= 16000
					scope:target_country.income <= 40000
				}
				AND = {
					income >= 225000
					net_fixed_income >= 18000
					scope:target_country.income <= 45000
				}	
				AND = {
					income >= 250000
					net_fixed_income >= 20000
					scope:target_country.income <= 50000
				}					
			}			
		}
		
		will_break = { 
			NOT = {
				AND = {
					has_variable = forced_humanitarian_aid_time
					var:forced_humanitarian_aid_time ?= scope:target_country
				}
			}
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					OR = {
						net_fixed_income < 0	
						is_at_war = yes
					}				
				}				
				NOT = {
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}				
				}
				num_income_transfer_pacts > 3
				AND = {
					num_income_transfer_pacts > 1
					OR = {
						net_fixed_income < -10000	
						income < 300000
					}			
				}
				AND = {
					num_income_transfer_pacts > 2
					OR = {
						net_fixed_income < -5000					
						income < 500000
					}
				}				
				NOR = {
					root = {
						owes_obligation_to = scope:target_country			
					}	
					NOT = { 
						scope:target_country = {
							owes_obligation_to = root			
						}						
					}
					AND = {
						is_at_war = no
						scope:target_country = { is_at_war = yes }
					}			
				}							
				# Threshold for breaking off a humanitarian_aid is higher than for starting it
				# We also don't look at net_fixed_income here since starting a humanitarian_aid changes it
				NOR = {
					AND = {
						income >= 40000
						scope:target_country.income <= 12500
					}				
					AND = {
						income >= 50000
						scope:target_country.income <= 15000
					}
					AND = {
						income >= 75000
						scope:target_country.income <= 20000
					}				
					AND = {
						income >= 100000
						scope:target_country.income <= 25000
					}
					AND = {
						income >= 125000
						scope:target_country.income <= 30000
					}
					AND = {
						income >= 150000
						scope:target_country.income <= 35000
					}	
					AND = {
						income >= 175000
						scope:target_country.income <= 40000
					}
					AND = {
						income >= 200000
						scope:target_country.income <= 45000
					}
					AND = {
						income >= 225000
						scope:target_country.income <= 50000
					}
					AND = {
						income >= 250000
						scope:target_country.income <= 55000
					}					
				}						
			}
		}			
	}

request_humanitarian_aid = {
	groups= { general } requires_approval = yes
	should_notify_third_parties = no

	selectable = {
		
	}

	potential = {
		
		is_revolutionary = no is_secessionist = no
		scope:target_country = {
			is_revolutionary = no is_secessionist = no
		}
		
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		
		OR = { 
			average_sol < 20
			any_scope_state = {
				has_famine = yes
			}
		}
		
		scope:target_country = {  
			
			scope:target_country.gdp > scope:actor.gdp
			
			#Country is independent
			OR = { 
				is_sufficiently_independent = yes
				is_direct_subject_of = scope:actor
			}
			
		}
		
	}
	
	possible = {
		
		scope:target_country = {  
	
			has_diplomatic_relevance = scope:actor
		
			NOT = { is_diplomatic_play_enemy_of = scope:actor }
			
			in_default = no
			
		}
		
	}
	
	reverse_pact = humanitarian_aid
	
	ai = {
		check_acceptance_for_will_propose = yes

		propose_score = {
			value = 10
		}

		propose_break_score = {
			value = 10
		}

		use_obligation_chance = {
			value = 0
		}

		owe_obligation_chance = {
			value = 0
		}
	}
	
}
