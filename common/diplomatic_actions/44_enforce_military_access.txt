enforce_military_access = {
	groups = {
		overlord
	}

	texture = "gfx/interface/icons/diplomatic_action_icons/enforce_military_access.dds"

	requires_approval = no
	should_notify_third_parties = yes
	show_effect_in_tooltip = yes
	can_use_obligations = no

	possible = {
	
		#THIS is at war
		is_at_war = yes
		
		#TARGET is a non-sovereign country
		scope:target_country = {
			is_sufficiently_independent = no
		}
		
		#Not at war with TARGET
		NOR = {
			is_diplomatic_play_enemy_of = scope:target_country 
			has_war_with = scope:target_country
		}
		
		#TARGET is not involved in war & is land adjacent to enemy
		any_scope_war = {
			NOT = { is_war_participant = scope:target_country }
			any_war_participant = {
				NOT = { this = root }
				has_war_with = root
				has_strategic_land_adjacency = scope:target_country
				NOT = { has_strategic_land_adjacency = root }
			}
		}
		
	}

	accept_effect = {
		every_scope_war = {
			join_war = {
				target = scope:target_country
				side = root
			}
		}
		if = {
			limit = {
				scope:target_country = { 
					OR = { 
						is_a_colonial_subject_gov = yes
						is_in_our_personal_union = yes
					}
				}
			}
			scope:target_country = {
				add_liberty_desire = 10
			}			
		}
		else = {
			scope:target_country = {
				add_liberty_desire = 20
			}		
		}
	}

	ai = {
		evaluation_chance = {
			value = 0.0
			if = {
				limit = { is_at_war = yes }
				add = 1.0				
			}
		}
		
		propose_score = {
			value = 1000
		}

		will_propose = {
			any_scope_war = {
				NOT = { is_war_participant = scope:target_country }
				any_war_participant = {
					NOT = { this = root }
					has_war_with = root
					has_strategic_land_adjacency = scope:target_country
					OR = {
						NOT = { has_strategic_land_adjacency = root }
						has_port_country = no
					}
				}
			}
		}
		
	}
}
