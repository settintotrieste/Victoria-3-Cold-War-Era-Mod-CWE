support_separatism = {
    groups = {
        general
    }

	requires_approval = no
	show_effect_in_tooltip = yes
	can_use_obligations = no
    should_notify_third_parties = no
	show_confirmation_box = no

	unlocking_technologies = {
		
	}

    selectable = {
        NOT = { is_country_type = decentralized }
		is_revolutionary = no
    }

    potential = {
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		is_revolutionary = no
    }

    possible = {
		NOT = {
			any_primary_culture = {
				is_primary_culture_of = scope:target_country
			}
		}
		custom_tooltip = {
			text = SUPPORT_SEPARATISM_SUFFICIENT_POPS
			any_primary_culture = {
				save_temporary_scope_as = actor_primary_culture
				scope:target_country = {
					any_scope_state = {
						is_homeland_of_country_cultures = root
						"culture_percent_state(scope:actor_primary_culture)" >= 0.1
					}
				}
			}
		}
		OR = {
			is_subject = no
			aggressive_diplomatic_plays_permitted = yes
		}
    }

    accept_effect = {
        custom_tooltip = support_separatism_action_effect_breakdown
    }

    is_hostile = yes

    pact = {
        cost = 100
        is_hostile = yes
        is_two_sided_pact = no
        show_in_outliner = yes

		relations_progress_per_day = -1 # At -100, -1 relations
		relations_improvement_min = -25

		requirement_to_maintain = {
			trigger = {
				has_diplomatic_relevance = scope:target_country
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_AT_PEACE"
					NOT = { has_war_with = scope:target_country }
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_NOT_DECENTRALIZED"
					NOT = { is_country_type = decentralized }
					scope:target_country = {
						NOT = { is_country_type = decentralized }
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = DIPLO_ACTION_RELATIONS_AMICABLE_OR_GREATER_NOT
					NOT = {
						scope:target_country.relations:root >= relations_threshold:amicable
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
					scope:target_country = {
						any_primary_culture = {
							NOT = { is_primary_culture_of = root }
						}
					}
			}
		}

		requirement_to_maintain = {
			trigger = {
				NOT = { scope:target_country = { has_diplomats_expelled = root } }
			}
		}
    }

    ai = {
		max_influence_spending_fraction = 0.25

        evaluation_chance = {
            value = 0.5
        }

		propose_score = {
			value = 10

			if = {
				limit = {
					OR = {
						has_attitude = {
							who = scope:target_country
							attitude = belligerent
						}
						has_attitude = {
							who = scope:target_country
							attitude = domineering
						}
						has_attitude = {
							who = scope:target_country
							attitude = antagonistic
						}
						has_attitude = {
							who = scope:target_country
							attitude = defiant
						}
						has_attitude = {
							who = scope:target_country
							attitude = rebellious
						}
					}
				}

				add = 10
			}

			if = {
				limit = {
					is_subject_of = scope:target_country
				}
				multiply = 2.5
			}

			if = {
				limit = {
					AND = {
						has_radical_leftist_government = no
						scope:target_country = {
							has_radical_leftist_government = yes
						}
					}

				}
				multiply = 2.5
			}

			if = {
				limit = {
					AND = {
						has_radical_leftist_economy = no
						scope:target_country = {
							has_radical_leftist_economy = yes
						}
					}

				}
				multiply = 2.5
			}
		}

		will_propose = {
			OR = {
				AND = {
					has_attitude = {
						who = scope:target_country
						attitude = antagonistic
					}
					scope:target_country.relations:root <= relations_threshold:poor
				}
				has_attitude = {
					who = scope:target_country
					attitude = belligerent
				}
				has_attitude = {
					who = scope:target_country
					attitude = domineering
				}
				has_attitude = {
					who = scope:target_country
					attitude = defiant
				}
				has_attitude = {
					who = scope:target_country
					attitude = rebellious
				}
			}
		}

		will_break = {
			NOR = {
				has_attitude = {
					who = scope:target_country
					attitude = belligerent
				}
				has_attitude = {
					who = scope:target_country
					attitude = domineering
				}
				has_attitude = {
					who = scope:target_country
					attitude = defiant
				}
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
				has_attitude = {
					who = scope:target_country
					attitude = rebellious
				}
			}
		}
    }
}
