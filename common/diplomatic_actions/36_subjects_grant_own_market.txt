grant_own_market = {
	requires_approval = no
	groups= { 
		overlord 
		power_bloc_leader
	}
	
	texture = "gfx/interface/icons/diplomatic_action_icons/grant_market_independence.dds"
	confirmation_sound = "event:/SFX/UI/MapInteraction/SubjectInteractions/diplomatic_action_subjects_05_grant_own_market_benign"
	potential = {
		scope:target_country = {
			market = prev.market
			NOT = { is_diplomatic_play_enemy_of = root }
		}
	}
	
	possible = {
		NOT = { market.owner = scope:target_country }
		OR = {
			AND = { # the target controls the market and the requester is independent
				scope:target_country = {
					is_subject = no
				}
				market.owner = root
			}
			scope:target_country = {
				is_direct_subject_of = root
			}
		}
	}
	
	pact = {	
		cost = 0
		forced_duration = 12
		
		market_owner = second_country
		
		requirement_to_maintain = {
			trigger = {
				scope:target_country = {
					OR = {
						is_direct_subject_of = root
						power_bloc.power_bloc_leader ?= root
					}
				}
			}
		}
		
		actor_can_break = {
		
			#Superstate Power Blocs can only invite same Superstate members / aspirants
			trigger_if = {
				limit = { 
					scope:actor = { 
						power_bloc ?= {
							has_identity = identity:identity_superstate
						}
					}
				}
				OR = { 
					custom_tooltip = { 
						text = does_not_shares_same_superstate.tt
						scope:target_country = { 
							shares_same_superstate = no
						}
					}
					scope:target_country = {
						save_temporary_scope_as = THIS_scope
						custom_tooltip = {
							text = NOT_superstate_customs_union_TAG.tt
							shares_same_superstate = yes
							NAND = {
								has_variable = superstate_customs_union
								power_bloc ?= {
									has_identity = identity:identity_superstate
								}
							}
						}
					}
				}
			}
			
		}
		
		manual_break_effect = {
			scope:target_country = {
				if = {
					limit = {
						is_subject_of = root 
						is_non_customs_union_subject = yes
					}
					add_liberty_desire = 50
					scope:actor = { change_infamy = 10 }
					change_relations = { country = scope:actor value = -50 }
				}
				else = {
					add_liberty_desire = 25
				}
			}
		}
	}
	
	accept_effect = {
		if = {
			limit = {
				"scope:target_country.economic_dependence(root)" > 1
				scope:target_country.capital = {
					is_isolated_from_market = no
				}
				scope:target_country = {
					is_subject_of = root
				}
			}
			
			custom_tooltip = { text = grant_own_market_increase_liberty_desire_effect }	
			scope:target_country = {
				add_liberty_desire = {
					add = "scope:target_country.economic_dependence(root)"
					subtract = 1
					multiply = 40
					add = 10
				}
			}				
		}
		else_if = {
			limit = {
				"scope:target_country.economic_dependence(root)" < 1
				scope:target_country = {
					is_subject_of = root
				}
			}
			
			custom_tooltip = { text = grant_own_market_decrease_liberty_desire_effect }			
			scope:target_country = {
				add_liberty_desire = {
					add = 1
					subtract = "scope:target_country.economic_dependence(root)"
					multiply = -15
					add = -5
				}
			}				
		}
	}
	
	ai = {
	
		evaluation_chance = {
			value = 0.25		
		}
	
		propose_score = { 
			value = 100
		}
	
		will_propose = {
		
			#General conditions
			OR = { 
				AND = { 
					NOT = { is_adjacent_to_country = scope:target_country }
					scope:target_country.capital = { is_isolated_from_market = yes }
				}
				AND = {
					scope:target_country = { is_non_customs_union_subject = yes }
					scope:actor = { 
						ruler ?= {
							interest_group = {
								law_stance = {
									law = law_type:law_free_trade
									value >= approve
								}
							}
						}
						NOT = { has_law_or_variant = law_type:law_free_trade }
						NOT = { is_a_democracy = yes }
					}
				}
			}
			
			#Special cases
			trigger_if = {
				limit = {
					OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this c:BRU ?= this }
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				NAND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this } }
					scope:target_country = { c:SGP ?= this }
				}
				NAND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this } }
					scope:target_country = { c:BRU ?= this }
				}
			}
			
			#Special cases
			trigger_if = {
				limit = {
					OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this c:BRU ?= this }
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				NAND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this } }
					scope:target_country = { c:SGP ?= this }
				}
				NAND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this } }
					scope:target_country = { c:BRU ?= this }
				}
			}
			trigger_if = {
				limit = {
					OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this }
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				NAND = {
					scope:actor = { OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this } }
					scope:target_country = { OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this } }
				}
			}
			trigger_if = {
				limit = {
					c:NEJ ?= this
					has_global_flag = GCC_Union_formed
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				NAND = {
					scope:actor = { c:NEJ ?= this }
					scope:target_country = { 
						OR = { c:KUW ?= this c:BHR ?= this c:QAT ?= this c:UAE ?= this c:YEM ?= this c:NYE ?= this c:SYE ?= this } 
						has_variable = GCC_Union_offered
					}
				}
			}
			
		}
		
		will_break = {
		
			#General conditions
			OR = {
				is_adjacent_to_country = scope:target_country
				AND = {
					supply_network_strength >= 0.75
					has_port = yes
					scope:target_country = { has_port = yes }
				}
			}
			NAND = {
				scope:target_country = { 
					is_non_customs_union_subject = yes
				}
				scope:actor = { 
					OR = { 
						is_a_democracy = yes
						ruler ?= {
							OR = {
								character_is_democratic = yes
								interest_group = {
									law_stance = {
										law = law_type:law_free_trade
										value < approve
									}
								}
							}
						}
						infamy >= infamy_threshold:infamous
						NOT = { has_law_or_variant = law_type:law_free_trade }
					}
				}
			}
			
			#Special cases
			trigger_if = {
				limit = {
					OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this c:BRU ?= this }
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no }
					}
				}
				AND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this } }
					scope:target_country = { c:SGP ?= this }
				}
				AND = {
					scope:actor = { OR = { c:MSY ?= this c:MLY ?= this c:SGP ?= this } }
					scope:target_country = { c:BRU ?= this }
				}
			}
			trigger_if = {
				limit = {
					OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this }
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				AND = {
					scope:actor = { OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this } }
					scope:target_country = { OR = { c:BEL ?= this c:NET ?= this c:LUX ?= this } }
				}
			}
			trigger_if = {
				limit = {
					c:NEJ ?= this
					has_global_flag = GCC_Union_formed
					OR = {
						scope:actor = { is_adjacent_to_country = scope:target_country }
						scope:target_country.capital = { is_isolated_from_market = no } 
					}
				}
				AND = {
					scope:actor = { c:NEJ ?= this }
					scope:target_country = { 
						OR = { c:KUW ?= this c:BHR ?= this c:QAT ?= this c:UAE ?= this c:YEM ?= this c:NYE ?= this c:SYE ?= this } 
						has_variable = GCC_Union_offered
					}
				}
			}
			
		}
	}
	
}
