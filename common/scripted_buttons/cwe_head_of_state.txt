je_fire_head_of_state_button = {
	name = "je_fire_head_of_state_button"
	desc = "je_fire_head_of_state_button_desc"

	visible = { 
		NOT = { has_variable = elected_head_of_state_system }
	}

	possible = {
	
		government_has_legislative_super_majority = yes
		in_election_campaign = no
		
		#Timer
		custom_tooltip = {
			text = head_of_state_timer.tt
			NOT = { has_variable = fire_head_of_state_timer }
		}
		
	}

	effect = {
		
		#Democrats are angry
		if = {
			limit = { has_variable = elected_head_of_state_system }
			antagonise_democratic_and_opposition_IGs_large = yes
		}
		
		#Change HOS
		custom_tooltip = { 
			text = head_of_state_dimissed.tt
			trigger_event = { id = reserve_powers_system.20 } 
		}
		
		#Only 1 time per year
		set_variable = { name = fire_head_of_state_timer years = 1 }
		
	}
	
	ai_chance = {
		base = 0
	}
	
}

je_elected_head_of_state_button = {
	name = "je_elected_head_of_state_button"
	desc = "je_elected_head_of_state_button_desc"

	visible = { 
		NOT = { has_variable = elected_head_of_state_system }
	}

	possible = {
	
		government_has_legislative_super_majority = yes
		in_election_campaign = no
		
		#Timer
		custom_tooltip = {
			text = head_of_state_timer.tt
			NOT = { has_variable = fire_head_of_state_timer }
		}
		
	}

	effect = {
	
		set_variable = elected_head_of_state_system
	
		#PM IG is angry
		every_scope_character = {
			limit = {
				has_role = character_role_ig_leader
				character_is_democratic = yes
			}
			interest_group ?= {
				save_scope_as = THIS_IG_scope
			}
			owner ?= {
				add_loyalists = {
					value = medium_radicals
					interest_group = scope:THIS_IG_scope
				}
			}
		}
		every_scope_character = {
			limit = { 
				has_role = character_role_ig_leader
				interest_group ?= { is_in_government = yes }
				character_is_autocratic = yes
			}
			interest_group ?= {
				save_scope_as = THIS_IG_scope
			}
			owner ?= {
				add_radicals = {
					value = medium_radicals
					interest_group = scope:THIS_IG_scope
				}
			}
		}
		
		#Only 1 time per year
		set_variable = { name = fire_head_of_state_timer years = 1 }
		
	}
	
	ai_chance = {
		base = 0
	}
	
}
