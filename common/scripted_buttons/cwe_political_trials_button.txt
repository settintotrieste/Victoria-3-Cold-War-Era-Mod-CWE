je_political_trials_drop_charges_button = {
	name = "je_political_trials_drop_charges_button"
	desc = "je_political_trials_drop_charges_button_desc"

	visible = { 
		
		
		
	}

	possible = {
		
		#A character is on trial
		any_scope_character = {
			has_modifier = indicted_character
			NOT = { has_role = character_role_prisoner }
		}
		
		#Cabinet agrees with this
		custom_tooltip = {
			text = has_no_low_cabinet_unity.tt
			NOT = { has_variable = low_cabinet_unity }
		}
		
	}

	effect = {
		
		#Automatic acquittal
		random_scope_character = {
			limit = { 
				has_modifier = indicted_character
				NOT = { has_role = character_role_prisoner }
			}
			remove_modifier = indicted_character
		}
		
		#Government reaction
		if = {
			limit = { 
				is_a_dictatorship = yes 
				scope:convicted_character_scope ?= { interest_group = { NOT = { is_same_interest_group_type = root.ruler.interest_group } } }
			}
			add_radicals = {
				value = medium_radicals
				interest_group = ruler.interest_group
			}
		}
		
	}
	
	ai_chance = {
		base = 0
		
		if = {
			limit = { 
				
				exists = ruler
				ruler ?= {
					NOR = {
						has_ideology = ideology:ideology_autocratic
						has_ideology = ideology:ideology_power_hungry
						has_ideology = ideology:ideology_elitist
					}
				}
				
			}
			add = 10 
		}
		
	}
	
}

je_political_trials_bribe_judge_button = {
	name = "je_political_trials_bribe_judge_button"
	desc = "je_political_trials_bribe_judge_button_desc"

	visible = { 
		
		#Not redundant
		NAND = {
			has_law_or_variant = law_type:law_politicised_judiciary 
			has_law_or_variant = law_type:law_kangaroo_law
		}
		
	}

	possible = {
	
		#Cabinet agrees with this
		custom_tooltip = {
			text = has_no_low_cabinet_unity.tt
			NOT = { has_variable = low_cabinet_unity }
		}
		
		#Has not already bribed judge
		custom_tooltip = {
			text = political_trials_bribe_judge.tt
			NOT = { has_variable = political_trials_bribed_judge }
		}
		
	}

	effect = {
		
		#Only 1 time
		set_variable = political_trials_bribed_judge
		
		#Domestic reaction
		random_scope_character = {
			limit = { 
				has_modifier = indicted_character
				NOT = { has_role = character_role_prisoner }
			}
			interest_group ?= {
				save_scope_as = indicted_character_ig_scope
			}
		}
		add_radicals = {
			value = large_radicals
			interest_group = scope:indicted_character_ig_scope
		}
		
		#Increase corruption in country
		if = {
			limit = { has_law_or_variant = law_type:law_very_high_anti_corruption }
			activate_law = law_type:law_high_anti_corruption
		}
		else_if = {
			limit = { has_law_or_variant = law_type:law_high_anti_corruption }
			activate_law = law_type:law_medium_anti_corruption
		}
		else_if = {
			limit = { has_law_or_variant = law_type:law_medium_anti_corruption }
			activate_law = law_type:law_low_anti_corruption
		}
		else_if = {
			limit = { has_law_or_variant = law_type:law_low_anti_corruption }
			activate_law = law_type:law_very_low_anti_corruption
		}
		
		#Ruler can get tainted with corruption
		if = {
			limit = { 
				ruler ?= { NOT = { has_modifier = corruption_scandal_character_modifier } }
			}
			random_list = {
				3 = { }
				1 = { 
					add_modifier = { 
						name = corruption_scandal_character_modifier 
						years = 2 
						is_decaying = yes 
					} 
				}
			}
		}
		
	}
	
	ai_chance = {
		base = 0
		
		if = {
			limit = {
				ruler ?= {
					OR = {
						has_ideology = ideology:ideology_autocratic
						has_ideology = ideology:ideology_power_hungry
					}
				}
			}
			add = 10 
		}
		
	}
	
}
