je_occupation_of_austria_end_occupation_button = {
	name = "je_occupation_of_austria_end_occupation_button"
	desc = "je_occupation_of_austria_end_occupation_button_desc"

	visible = {
	
		#Occupies Austria in a condominium
		exists = c:AUS
		c:AUS ?= { has_modifier = condominium_modifier }
		has_diplomatic_pact = { who = c:AUS type = condominium is_initiator = yes }
		
	}

	possible = {
	
		is_sufficiently_independent = yes
		is_at_war = no
		
		c:AUS ?= { is_at_war = no }
		
	}

	effect = {
		
		#International reaction
		change_infamy = -10
		
		#Option for USSR et al to accept
		every_country = { 
			limit = { 
				NOT = { THIS = ROOT }
				has_diplomatic_pact = { who = c:AUS type = condominium is_initiator = yes }
			}
			trigger_event = { id = austria_formation.2 } 
		}
		
		#End condominium
		remove_diplomatic_pact = { country = c:AUS type = condominium }
		if = {
			limit = { c:AUS = { is_subject_of = ROOT } }
			c:AUS = { make_independent = yes }
		}
		c:AUS ?= {
			effect_starting_politics_war_policy_pacifism = yes
		}
		
		#Austrian State Treaty
		every_scope_treaty = {
			limit = {
				binds = c:AUS
				any_scope_article = {
					OR = { 
						has_type = foreign_investment_rights
						has_type = trade_privilege
						has_type = guarantee_independence
					}
				}
			}
			withdraw = {
				country = ROOT
			}
		}
		c:AUS ?= {
			create_treaty = {
				name = treaty_name_AUS_state_treaty
				first_country = ROOT
				second_country = c:AUS

				is_draft = no
				binding_period = { years = 10 }

				articles_to_create = {
					{
						article = foreign_investment_rights
						source_country = c:AUS
						target_country = ROOT
					}
					{
						article = trade_privilege
						source_country = c:AUS
						target_country = ROOT
					}
					{
						article = law_commitment
						source_country = c:AUS
						target_country = ROOT
						inputs = {
							{ law_type = law_type:law_no_war }
						}
					}
					{
						article = guarantee_independence
						source_country = ROOT
						target_country = c:AUS
					}
				}
			}
		}
		
	}
	
	ai_chance = {
		base = 0
		
		#AI will only do this if ruler is hated
		if = { 
			limit = {
				OR = {
					AND = {
						infamy >= infamy_threshold:pariah
						ruler ?= { character_is_autocratic = no }
					}
					in_default = yes
				}
			}
			add = 10
		}
		
	}
	
}

je_occupation_of_berlin_end_occupation_button = {
	name = "je_occupation_of_berlin_end_occupation_button"
	desc = "je_occupation_of_berlin_end_occupation_button_desc"

	visible = {
	
		#Occupies Berlin in a condominium
		exists = c:FCB
		c:FCB ?= { has_modifier = condominium_modifier }

	possible = {
	
		is_sufficiently_independent = yes
		is_at_war = no
		
		c:FCB ?= { is_at_war = no }
		
	}

	effect = {
		
		#International reaction
		change_infamy = -10
		
		#Option for et al to accept
		every_country = { 
			limit = { 
				NOT = { THIS = ROOT }
				has_diplomatic_pact = { who = c:FCB type = condominium is_initiator = yes }
			}
			trigger_event = { id = historical_territorial_transfers.47 } 
		}
		
		#End condominium
		remove_diplomatic_pact = { country = c:FCB type = condominium }
		if = {
			limit = { 
				exists = c:EGR
				c:FCB = { 
					is_subject_of = ROOT 
					is_adjacent_to_country = c:EGR
				}
			}
			c:EGR ?= { annex_with_incorporation = c:FCB }
		}
		else_if = {
			limit = { 
				exists = c:WGR
				c:FCB = { 
					is_subject_of = ROOT 
					is_adjacent_to_country = c:WGR
				}
			}
			c:WGR ?= { annex_with_incorporation = c:FCB }
		}
		else_if = {
			limit = { 
				exists = c:GER
				c:FCB = { 
					is_subject_of = ROOT
					is_adjacent_to_country = c:GER
				}
			}
			c:GER ?= { annex_with_incorporation = c:FCB }
		}
		else_if = {
			limit = {
				OR = {
					exists = c:GER
					exists = c:WGR
					exists = c:EGR
				}
				c:FCB = { 
					is_subject_of = ROOT
				}
			}
			random_country = {
				limit = {
					OR = { 
						c:GER ?= this
						c:WGR ?= this
						c:EGR ?= this
					}
				}
				annex_with_incorporation = c:FCB
			}
		}
		else = {
			if = {
				limit = { c:FCB = { is_subject_of = ROOT } }
				c:FCB = { make_independent = yes }
			}
		}
		
	}
	
	ai_chance = {
		base = 0
		
		#AI will only do this if ruler is hated
		if = { 
			limit = {
				OR = {
					AND = {
						infamy >= infamy_threshold:pariah
						ruler ?= { character_is_autocratic = no }
					}
					in_default = yes
				}
			}
			add = 10
		}
		
	}
	
}

je_german_unification_button = {
	name = "je_german_unification_button"
	desc = "je_german_unification_button_desc"

	visible = {
		
		NOT = { c:GER ?= this }
		
	}

	possible = {
	
		#German reunification factors
		OR = {
			
			#THIS has not become Germany yet
			AND = {
			
				#Independent or are subjects of the same overlord
				OR = {
					AND = {
						is_sufficiently_independent = yes
						OR = { c:WGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:WGR } c:WGR ?= this }
						OR = { c:EGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:EGR } c:EGR ?= this }
						OR = { c:SGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:SGR } c:SGR ?= this }
					}
					AND = {
						is_subject = yes
						overlord ?= {
							is_sufficiently_independent = yes
							is_at_war = no
						
							OR = { c:WGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:WGR } }
							OR = { c:EGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:EGR } }
							OR = { c:SGR ?= { OR = { is_direct_subject_of = ROOT is_sufficiently_independent = yes } } NOT = { exists = c:SGR } }
						}
					}
				}
				
				#Other German States share the same gov type
				custom_tooltip = {
					text = GER_has_same_gov_type_tt
					OR = {
						AND = { OR = { c:WGR ?= { is_a_democracy = yes } NOT = { exists = c:WGR } } OR = { c:EGR ?= { is_a_democracy = yes } NOT = { exists = c:EGR } } OR = { c:SGR ?= { is_a_democracy = yes } NOT = { exists = c:SGR } } }
						AND = { OR = { c:WGR ?= { is_a_monarchy = yes } NOT = { exists = c:WGR } } OR = { c:EGR ?= { is_a_monarchy = yes } NOT = { exists = c:EGR } } OR = { c:SGR ?= { is_a_monarchy = yes } NOT = { exists = c:SGR } } }
						AND = { OR = { c:WGR ?= { is_a_communist_government = yes } NOT = { exists = c:WGR } } OR = { c:EGR ?= { is_a_communist_government = yes } NOT = { exists = c:EGR } } OR = { c:SGR ?= { is_a_communist_government = yes } NOT = { exists = c:SGR } } }
						AND = { OR = { c:WGR ?= { is_a_nationalist_government = yes } NOT = { exists = c:WGR } } OR = { c:EGR ?= { is_a_nationalist_government = yes } NOT = { exists = c:EGR } } OR = { c:SGR ?= { is_a_nationalist_government = yes } NOT = { exists = c:SGR } } }
						AND = { OR = { c:WGR ?= { is_a_theocratic_government = yes } NOT = { exists = c:WGR } } OR = { c:EGR ?= { is_a_theocratic_government = yes } NOT = { exists = c:EGR } } OR = { c:SGR ?= { is_a_theocratic_government = yes } NOT = { exists = c:SGR } } }
					}
				}
				
			}
			
		}
		
		#THIS is not at war
		is_at_war = no
		
	}

	effect = {
		
		#Become Germany
		if = {
			limit = {
				NOT = { exists = c:GER }
				NOT = { c:GER ?= this }
			}
			change_tag = GER
		}
		if = {
			limit = {
				exists = c:GER
				NOT = { c:GER ?= this }
			}
			c:GER ?= { annex_with_incorporation = ROOT }
		}
		
		#Unify Germany
		every_country = {
			limit = { 
				is_player = no
				OR = { 
					c:WGR ?= this
					c:EGR ?= this
					c:SGR ?= this
				}
			}
			save_scope_as = GER_state
			c:GER ?= { annex_with_incorporation = scope:GER_state }
		}
		
		#Set new capital
		if = {
			limit = {
				c:GER ?= this
				exists = c:FCB
				c:FCB ?= { 
					OR = { 
						is_sufficiently_independent = yes
						has_modifier = condominium_modifier 
						is_subject_of = ROOT
					}
				}
			}
			annex_with_incorporation = c:FCB
		}
		if = {
			limit = { 
				capital = { NOT = { state_region = s:STATE_BRANDENBURG } }
				has_state_in_state_region = STATE_BRANDENBURG
			}
			set_capital = STATE_BRANDENBURG
		}
		
		#Prestige from re-unification
		add_modifier = { name = national_pride_modifier months = 60 is_decaying = yes }
		
	}
	
	ai_chance = {
		base = 0
		
		#AI will give priority to player
		if = {
			limit = {
				OR = {
					c:WGR ?= this
					c:SGR ?= this
				}
				NOR = {
					c:WGR ?= { is_player = yes }
					c:EGR ?= { is_player = yes }
					c:SGR ?= { is_player = yes }
				}
			}
			add = 100
		}
		
	}
	
}
