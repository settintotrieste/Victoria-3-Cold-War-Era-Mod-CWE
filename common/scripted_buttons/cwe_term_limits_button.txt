je_create_collective_presidency_button = {
	name = "je_create_collective_presidency_button"
	desc = "je_create_collective_presidency_button_desc"

	visible = { 
		
		#Does not have Collective HOS
		NOT = { 
			active_law:lawgroup_governance_principles ?= {
				has_amendment = amendment_type:amendment_rotating_presidency
			}
		}
		
	}

	possible = {
	
		#Presidential system
		OR = { 
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOS_nationalist_leader
			has_law_or_variant = law_type:law_HOS_religious_leader
			has_law_or_variant = law_type:law_HOS_communist_leader
		}
		
		#Not for Prime Ministers
		NOT = { has_law_or_variant = law_type:law_HOG_prime_minister }
		
		#Has Term Limit
		has_law_or_variant = law_type:law_yes_term_limit
		
		#Legal requirements
		government_has_legislative_super_majority = yes
		
	}

	effect = {
	
		#Add Amendment
		active_law:lawgroup_governance_principles ?= {
			add_amendment = {
				type = amendment_rotating_presidency
				sponsor = PREV.ig:ig_petty_bourgeoisie 
			}
		}
		
	}
	
	ai_chance = {
		base = 0
	}
	
}
