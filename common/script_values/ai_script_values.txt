base_impose_law_weight= {
	value = 0.01

	# Impose law is a fairly hostile act, AIs should use it mostly when angry at a subject
	if = {
		limit = {
			has_attitude = {
				who = scope:target_country
				attitude = domineering
			}
		}
		multiply = 0.5
	}

	# Non ideological unions make less use of this
	if = {
		limit = {
			NOT = { power_bloc ?= { has_identity = identity:identity_ideological_union } }
		}
		multiply = 0.5
	}

	# Less likely to do this to subjects unless LD is low
	if = {
		limit = {
			scope:target_country = {
				is_subject = yes
				liberty_desire > 25
			}
		}
		multiply = 0.5
	}

	# Shouldn't care much about meddling in really tiny countries
	if = {
		limit = {
			scope:target_country = {
				gdp < 1000000
			}
		}
		multiply = 0.25
	}
	else_if = {
		limit = {
			scope:target_country = {
				gdp < 2500000
			}
		}
		multiply = 0.5
	}
	else_if = {
		limit = {
			scope:target_country = {
				gdp < 5000000
			}
		}
		multiply = 0.75
	}

	# Impact of AI strategies
	if = {
		limit = {
			has_strategy = ai_strategy_world_revolution
		}
		multiply = 5
	}
	else_if = {
		limit = {
			OR = {
				has_strategy = ai_strategy_anti_imperialism
				has_strategy = ai_strategy_maintain_power_balance
			}
		}
		multiply = 2
	}
	else_if = {
		limit = {
			OR = {
				has_strategy = ai_strategy_armed_isolationism
				has_strategy = ai_strategy_socialism_in_one_country
			}
		}
		multiply = 0.5
	}

}

	}

		value = cached_ai_incorporated_coastal_population
		divide = 10000000
	}

		value = cached_ai_unincorporated_coastal_population
		divide = 40000000

		value = cached_ai_overseas_subject_population

			value = scope:pre_overseas_empire_value

		}

	if = {
		limit = { has_technology_researched = tech_military_theory_2 }

		limit = { has_technology_researched = tech_military_theory_6 }

	}

		limit = { has_technology_researched = tech_military_theory_8 }
		multiply = 0.8
	}

		limit = { has_technology_researched = tech_military_theory_10 }
		multiply = 0.8
	}

	if = {
		limit = {
			OR = {
				has_strategy = ai_strategy_territorial_expansion 
				has_strategy = ai_strategy_unify_germany
				has_strategy = ai_strategy_unify_italy
				has_strategy = ai_strategy_unify_ethiopia
				has_strategy = ai_strategy_unify_china
			}
		}
		multiply = 1.25
	}
	
	# Scale army size to rank
	#if = {
	#	limit = { country_rank = rank_value:super_power }
	#	multiply = 2
	#}
	if = {
		limit = { country_rank >= rank_value:great_power }
		multiply = 0.80
	}
	if = {
		limit = { country_rank = rank_value:major_power }
		multiply = 0.60
	}
	if = {
		limit = { country_rank = rank_value:minor_power }
		multiply = 0.40
	}
	if = {
		limit = { country_rank < rank_value:minor_power }
		multiply = 0.20
	}
	
	#Rivals
	if = {
		limit = { 
			num_rivals >= 3
		}
		multiply = 1.75
	}
	else_if = {
		limit = { 
			num_rivals >= 2
		}
		multiply = 1.50
	}
	else_if = {
		limit = { 
			num_rivals >= 1
		}
		multiply = 1.25
	}
	
	#Governments
	if = {
		limit = { 
			has_law_or_variant = law_type:law_professional_army
		}
		multiply = 0.75
	}
	else_if = {
		limit = { 
			has_law_or_variant = law_type:law_national_militia
		}
		multiply = 0.50
	}
	if = {
		limit = { 
			has_law_or_variant = law_type:law_defensive_war
		}
		multiply = 0.75
	}
	else_if = {

			has_law_or_variant = law_type:law_no_war

		multiply = 0.50
	}

			is_a_dictatorship = yes
		}
		multiply = 2
	}
	
	#Neutrality
	if = {
		limit = { is_neutral = yes }
		multiply = 0.1
	}
	
	# Multiply final value by diplomatic pact expenses ratio - countries paying a lot to pacts have lower spending ambitions
	multiply = {
		value = 1
		subtract = diplomatic_pact_expense_ratio
	}

	min = {
		value = 20

		if = { #If accepted Negotiation quest, build barracks until JE ends
			limit = {
				has_journal_entry = je_negotiate_army_quest
			}
			add = {
				every_interest_group = { #first get the desired app from the IG
					limit = {
						has_variable = desired_army_size
					}
					add = var:desired_army_size
				}
				divide = average_barracks_power_projection #divide by the amount of barracks this will need based on average barracks power projection
				multiply = 1.10 #safety margin
			}
		}
		
	}
}

ship_mod_ai_price_ratio_reject     = { value = 1.25 }  # Weighted buy/sell ratio at/above which a mod is rejected (multiplier = 0)
ship_mod_ai_price_ratio_bonus      = { value = 0.75 }  # At/below which the full bonus multiplier applies
ship_mod_ai_price_bonus_multiplier = { value = 2.0 }   # Multiplier at the bonus ratio (linear interpolation in between)

# Scales each mod's base ai_weight by a function of the goods-weighted buy/sell
# ratio for its construction goods, against the country's market.
# Curve:
#   weighted_ratio >= ship_mod_ai_price_ratio_reject   -> multiplier = 0
#   weighted_ratio <= ship_mod_ai_price_ratio_bonus    -> multiplier = ship_mod_ai_price_bonus_multiplier
#   between                                            -> linear interpolation
# Linear form: multiplier = BONUS_MULT * (REJECT - ratio) / (REJECT - BONUS)
# At ratio = 1 (neutral market) this is BONUS_MULT * 0.5 / 0.5 = 1 with default
# thresholds (1.25 / 0.75 / 2.0), so a neutral-market mod keeps its base weight.

ship_mod_market_multiplier = {
	# Evaluate the trigger once and save the ratio as a temporary; we refer to
	# it three times below (two limit branches + the else arithmetic).
	value = "ship_modification_market_demand_ratio(scope:ship_modification_type)"
	save_temporary_value_as = temp_ship_mod_demand_ratio

	# Base multiplier = 1; each branch overrides via multiply.
	value = 1
	if = {
		limit = { scope:temp_ship_mod_demand_ratio >= ship_mod_ai_price_ratio_reject }
		multiply = 0
	}
	else_if = {
		limit = { scope:temp_ship_mod_demand_ratio <= ship_mod_ai_price_ratio_bonus }
		multiply = ship_mod_ai_price_bonus_multiplier
	}
	else = {
		multiply = {
			value = ship_mod_ai_price_bonus_multiplier
			multiply = {
				value = ship_mod_ai_price_ratio_reject
				subtract = scope:temp_ship_mod_demand_ratio
			}
			divide = {
				value = ship_mod_ai_price_ratio_reject
				subtract = ship_mod_ai_price_ratio_bonus
			}
		}
	}
}


ship_mod_doctrine_boost = { value = 1.5 }  # Multiplier applied when a bundle's doctrinal condition fires

# Professional-navy / great-power balance: invest in the heaviest mods on
# capital hulls. Also picks up Fire Control and Range Finder utility mods —
# tools of disciplined capital-ship gunnery.
ship_mod_capital_doctrine = {
	value = 1
	if = {
		limit = {
			OR = {
				has_law = law_type:law_professional_navy
				has_strategy = ai_strategy_maintain_power_balance
			}
		}
		multiply = ship_mod_doctrine_boost
	}
}
