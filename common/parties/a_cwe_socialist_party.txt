#Political Spectrum -> Right

socialist_party = {
	color = { 255 0 0 }
	
	name = {
		first_valid = {
			triggered_desc = {
				desc = party_socialist_faction
				trigger = {
					OR = { 
						has_law_or_variant = law_type:law_parties_not_allowed
						has_law_or_variant = law_type:law_single_party_state
					}
				}
			}
			triggered_desc = {
				desc = party_socialist
				trigger = {
					always = yes
				}
			}
		}
	}
	
	icon = {
		default = "/gfx/interface/icons/political_parties/social_democrats.dds"
	}
	
	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		always = yes
	}
	
	available_for_interest_group = {
		NOT = { has_modifier = IG_expelled_socialist_party }
		IG_not_excluded_from_party_membership = yes
		OR = {
			owner = { NOR = { has_law_or_variant = law_type:law_parties_not_allowed has_law_or_variant = law_type:law_dominant_party_state } }
			is_in_government = yes ig_clout >= 0.10
		}
		IG_armed_forces_party_eligibility_check = yes IG_civil_service_party_eligibility_check = yes
	}
	
	join_weight = {
		value = {
			value = -8
			desc = "base_value"
		}
		
		#IG has correct leader political leaning
		if = {
			limit = {
				exists = leader leader = {
					OR = { 
						has_ideology = ideology:ideology_elitist
						has_ideology = ideology:ideology_democratic
					}
				}
			}
			add = {
				value = 1
				desc = "from_correct_leader_political_leaning"
			}
		}
		
		#IG has correct IG general political leaning
		if = {
			limit = {
				OR = { 
					has_ideology = ideology:ideology_authoritarian
					has_ideology = ideology:ideology_communitarian
				}
			}
			add = {
				value = 2
				desc = "from_correct_ig_general_political_leaning"
			}
		}
		
		#IG has correct IG specific political leaning
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_progressive
					
					has_ideology = ideology:ideology_communist
					has_ideology = ideology:ideology_republican
					has_ideology = ideology:ideology_social_militarist
				}
			}
			add = {
				value = 3
				desc = "from_correct_ig_specific_political_leaning"
			}
		}
		
		#IG has correct IG general economic leaning
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_left_economy
					has_ideology = ideology:ideology_far_left_economy
				}
			}
			add = {
				value = 4
				desc = "from_correct_general_economic_leaning"
			}
		}
		
		#IG has correct IG specific economic leaning
		if = {
			limit = {
				OR = { 
					has_ideology = ideology:ideology_nationalist
					has_ideology = ideology:ideology_regionalist
				}
			}
			add = {
				value = 1
				desc = "from_correct_specific_economic_leaning"
			}
		}
		
	}
	
}
