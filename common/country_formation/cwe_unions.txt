		exists = c:RUS
		exists = c:BYE
		NOT = { c:SOV ?= this }
		
		#Sufficient level of integration
		trigger_if = {
			limit = { c:RUS ?= this }
			c:BYE ?= { is_subject_of = c:RUS }
			is_in_customs_union_with = c:BYE
			has_diplomatic_pact = { who = c:BYE type = legal_union is_initiator = yes }
			has_diplomatic_pact = { who = c:BYE type = currency_union is_initiator = yes }
		}
		trigger_if = {
			limit = { c:BYE ?= this }
			c:RUS ?= { is_subject_of = c:BYE }
			is_in_customs_union_with = c:RUS
			has_diplomatic_pact = { who = c:RUS type = legal_union is_initiator = yes }
			has_diplomatic_pact = { who = c:RUS type = currency_union is_initiator = yes }
		}
		
	}
