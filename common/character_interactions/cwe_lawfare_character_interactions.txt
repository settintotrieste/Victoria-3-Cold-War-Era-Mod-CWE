indict_character = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/exile_character"

	potential = {
	
		#Character is owned by THIS
		owner ?= scope:actor
		
		#Not already a prisoner
		NOT = { has_role = character_role_prisoner }
		
		#Not already on trial
		NOR = {
			has_modifier = indicted_character
			has_modifier = convicted_character
			has_modifier = unjustly_convicted_character
			has_modifier = acquitted_character
		}
		
		#Not an excluded character
		character_has_diplomatic_immunity = no
		is_in_exile_pool = no
		is_in_void = no
		is_ruler_of_own_country = no
		is_heir_of_own_country = no
		
	}

	possible = {
	
		#Character is not pre-occupied
		is_busy = no
		is_in_battle = no
		
		#Not already currently putting someone on trial
		owner ?= {
			hidden_trigger = { NOT = { has_journal_entry = je_political_trials_system } }
		}
		
		#Coup
		custom_tooltip = {
			text = pending_coup_event_tt
			NOT = {
				owner = {
					has_variable = coup_event_active
				}
			}
		}
		custom_tooltip = {
			text = doing_coup_for_player_tt
			NOT = {
				has_variable = commander_doing_coup_on_player_behalf
			}
		}
		
		#Has cause
		OR = {
		
			#Country can arrest dissidents at will
			owner ?= {
				OR = {
					has_law_or_variant = law_type:law_outlawed_dissent
					AND = { 
						is_a_dictatorship = yes
						NOT = { has_law_or_variant = law_type:law_full_free_speech }
					}
					has_modifier = state_of_emergency_issued
				}
			}
			
			## Character has committed a crime ##
			
			#Corruption scandal
			has_modifier = ignored_corruption
			has_modifier = corrupt_connections_exposed
			has_modifier = corrupt_connections_exposed_minor
			has_modifier = corruption_scandal_character_modifier
			has_modifier = dealing_with_corruption
			has_trait = grifter
			has_trait = framed_for_crimes
			
			#Traitor
			has_modifier = treasonous_traitor
			
			#Abuse of Power
			has_modifier = seizing_power
			has_trait = war_criminal
			
			# Personal Misconduct
			AND = {
				OR = {
					has_trait = cocaine_addiction
					has_trait = opium_addiction
				}
				owner ?= {
					OR = {
						has_law_or_variant = law_type:law_no_minority_issues
						has_law_or_variant = law_type:law_some_minority_issues
					}
				}
			}
			AND = {
				OR = { 
					has_trait = alcoholic
					has_trait = hedonist
				}
				owner ?= { 
					has_law_or_variant = law_type:law_no_minority_issues
				}
			}
			AND = {
				has_trait = homosexual_character
				owner ?= { has_law_or_variant = law_type:law_no_gender_issues }
			}
			
		}
		
	}

	effect = {
		
		#Put character on trial
		add_modifier = { name = indicted_character }
		owner ?= { hidden_effect = { add_journal_entry = { type = je_political_trials_system } } }
		
		#Domestic reaction
		interest_group ?= { save_scope_as = THIS_IG_scope }
		owner ?= {
			add_radicals = {
				value = small_radicals
				interest_group = scope:THIS_IG_scope
			}
		}
		
	}

	cooldown = { months = 6 }
	show_confirmation_box = yes
	
	should_ai_evaluate = {
		country_rank >= rank_value:minor_power # Don't burn performance on tiny AI countries messing around with character interactions
	}

	ai_chance = {
		value = -50
		if = {
			limit = {
				is_historical = yes
			}
			add = -1000
		}
		if = {
			limit = {
				interest_group = {
					ig_approval <= -5
				}		
			}
			add = 25
		}
		if = {
			limit = {
				interest_group = {
					ig_approval <= -10
				}		
			}
			add = 30
		}
		if = {
			limit = {
				scope:actor ?= {
					is_at_war = yes
				}
				OR = {
					has_role_of_type = general
					has_role_of_type = admiral
				}
			}
			add = -1000
		}
		if = {
			limit = {
				scope:actor ?= {
					has_law_or_variant = law_type:law_secret_police
				}			
			}
			add = 30
		}
		if = {
			limit = {
				scope:actor ?= {
					has_law_or_variant = law_type:law_outlawed_dissent
				}			
			}
			add = 30
		}
		if = {
			limit = {
				scope:actor ?= {
					has_law_or_variant = law_type:law_censorship
				}			
			}
			add = 10
		}
		if = {
			limit = {
				owner ?= { country_turmoil < 0.25 }
			}
			add = -1000
		}
	}
	
}

grant_amnesty = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	potential = {
		
		#Character is owned by THIS
		owner ?= scope:actor
		
		#Is in prison
		has_role = character_role_prisoner
		has_modifier = convicted_character
		
	}

	show_requirements = yes

	possible = {
		
		#Appropriate Laws
		owner ?= {
		
			#Cabinet agrees with this
			custom_tooltip = {
				text = has_no_low_cabinet_unity.tt
				NOT = { has_variable = low_cabinet_unity }
			}
			
		}
		
	}

	effect = {
		
		#Release Character
		replace_character_roles = { add = { character_role_politician } remove_all = yes }
		remove_modifier = convicted_character
		save_scope_as = released_prisoner
		post_notification = political_prisoner_released
		
		#Domestic reaction
		interest_group ?= { save_scope_as = THIS_IG_scope }
		owner ?= {
			add_loyalists = {
				value = small_radicals
				interest_group = scope:THIS_IG_scope
			}
		}
		
	}

	cooldown = { months = 6 }
	show_confirmation_box = yes

	should_ai_evaluate = {
		country_rank >= rank_value:minor_power # Don't burn performance on tiny AI countries messing around with character interactions
	}

	ai_chance = {
		value = -50
		if = {
			limit = {
				scope:actor ?= {
					ruler ?= {
						character_is_democratic = yes
					}
				}
			}
			add = 100
		}
		if = {
			limit = {
				scope:actor ?= {
					any_political_movement = {
						is_revolutionary = yes
					}
					ruler ?= {
						character_is_autocratic = no
					}
				}
			}
			add = 51
		}
	}
	
}
