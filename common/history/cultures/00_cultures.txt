CULTURES = {

	### Initialising baseline 10 fervour for every culture in the world
	every_country = { # "every_culture" is not a thing
		limit = {
			NOT = {
				any_scope_culture = {
					culture_current_fervor > 0
					count = all
				}
			}
		}
		#Fervor for Non-Colonies
		every_scope_state = {
			limit = { is_incorporated = yes }
			every_scope_culture = {
				limit = {
					culture_current_fervor < 10
				}
				set_fervor = 10
			}
		}
		every_scope_state = {
			limit = { is_incorporated = no }
			every_scope_culture = {
				limit = {
					culture_current_fervor < 20
				}
				set_fervor = 20
			}
		}
	}

}
