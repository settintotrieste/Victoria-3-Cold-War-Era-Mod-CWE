REPLACE_OR_CREATE:naval_mission_type_piracy = {
    icon = "gfx/interface/icons/commander_order_icons/piracy_privateering.dds"
    clicksound = "event:/SFX/UI/NavalBattle/Missions/naval_mission_piracy"

    piracy = yes
    piracy_targets = { non_allies }
    ai_will_cancel_for_transportation = yes

    command_modifier = {
        character_piracy_goods_capacity_mult = 1.5
        ship_vulnerability_mult = 0.25
        ship_visibility_mult = -0.5
    }

    potential = {
        owner ?= {
		
			#Proto-state
            OR = {
				is_revolutionary = yes 
				is_secessionist = yes
				is_country_type = decentralized
			}
            NOT = {
                any_scope_treaty = {
                    any_scope_article = {
                        source_country = root.owner
                        has_type = abandon_piracy
                    }
                }
            }
			
			#Seize shipping permitted
			trigger_if = {
				limit = {
					exists = top_overlord
					is_sufficiently_independent = no
				}
				top_overlord ?= {
					issued_letters_of_marque = yes
				}
			}
			trigger_else = {
				issued_letters_of_marque = yes
				is_sufficiently_independent = yes
			}
			
			#Correct ruler ideology
			has_jingoistic_ruler = yes
			
        }
    }
}
