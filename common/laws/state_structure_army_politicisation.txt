law_politicised_armed_forces = {
	group = lawgroup_armed_forces_politicisation
	
	icon = "gfx/interface/icons/law_icons/organic_regulation.dds"
	
	can_impose = { can_impose_law_default = yes } #progressiveness = -50
	
	can_enact = {
	
		#Constitutional Laws system (Requires a simple majority to amend constitutional laws)
		government_has_legislative_simple_majority = yes
		
		#Cabinet agrees with this law
		custom_tooltip = {
			text = no_cabinet_objection.tt
			NOT = {
				any_interest_group = {
					OR = { 
						has_modifier = cabinet_minister_ig
						leader ?= { has_modifier = cabinet_minister }
					}
					law_stance = {
						law = law_type:law_politicised_armed_forces
						value < neutral
					}
				}
			}
		}
		
	}
	
	on_activate = {
		
	}
	
	modifier = {
	
		#Loyalty over competence
		country_military_goods_cost_mult = 0.10
		
		#More influential armed_forces
		interest_group_ig_armed_forces_pol_str_mult = 0.50
		
	}
	
	possible_political_movements = {
		#law_politicised_armed_forces
		law_semi_politicised_armed_forces
		law_no_politicised_armed_forces
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}
