	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}
