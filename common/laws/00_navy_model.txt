law_merchant_navy = {
	group = lawgroup_navy_model
	icon = "gfx/interface/icons/law_icons/merchant_fleet.dds"

	progressiveness = -50

	is_visible = {
		any_scope_state = {
			is_coastal = yes
		}
	}

	modifier = {
		country_ship_group_supply_ships_construction_efficiency_add = 0.3
		country_ship_type_troop_ship_construction_efficiency_add = 0.3
		country_ship_construction_efficiency_add = -0.1
		ship_interest_gain_mult = -0.1
	}

	on_enact = {
	}

	# AI will never return to merchant navy
	ai_will_do = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

law_professional_navy = {
	group = lawgroup_navy_model
	icon = "gfx/interface/icons/law_icons/ocean_domination.dds"

	progressiveness = 0

	unlocking_technologies = { military_drill }

	is_visible = {
		any_scope_state = {
			is_coastal = yes
		}
	}

	modifier = {
		building_naval_administration_hiring_rate_mult = 0.25
		country_officers_pol_str_mult = 0.2
		country_ship_group_capital_ships_construction_efficiency_add = 0.25
		country_prestige_from_navy_power_projection_mult = 0.2
	}

	on_enact = {
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}

		# Once the AI gets to a certain army tech level they should try to ditch merchant navy
		if = {
			limit = {
				OR = {
					#has_technology_researched = logistics
					#has_technology_researched = general_staff
					# Naval Techs need adding
				}
				has_law_or_variant = law_type:law_merchant_navy
			}

			add = 1000
		}
	}
}
