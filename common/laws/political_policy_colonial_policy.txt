	can_enact = {
	
		custom_tooltip = {
			modifier:country_disallow_law_no_colonial_affairs_bool = no
			text = disallow_law_no_colonial_affairs_tt
		}
		
	}
	
	modifier = {
		country_acceptance_homeland_add = 10
	}
	
	possible_political_movements = {
		#law_no_colonial_affairs
		law_colonial_resettlement
		law_colonial_exploitation
	}
	
	pop_support = {
		value = 0

		add = {
			desc = "POP_LITERACY"		
			if = {
				limit = { 
					NOT = { strata = upper }
				}
				add = {
					value = literacy_rate
					divide = 2
				}
			}
		}
		
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
		
		if = {
			limit = { 
				NOT = { has_jingoistic_ruler = yes }
				game_date >= 1961.11.27
			}
			add = 750
		}
		
	}
	
}
