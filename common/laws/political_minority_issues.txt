# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_no_minority_issues = {
	group = lawgroup_minority_issues
	icon = "gfx/interface/icons/law_icons/imperial_subjecthood.dds"
	
	can_impose = { can_impose_law_default = yes } #progressiveness = -100
	
	on_enact = {
		
	}
	
	can_enact = {
	
		#Constitutional Laws system (Requires simple majority to amend constitutional laws)
		government_has_legislative_simple_majority = yes
		
		#No restriction by legal union
		THIS_is_not_restricted_by_confederal_law = yes
		
		#Cabinet agrees with this law
		custom_tooltip = {
			text = no_cabinet_objection.tt
			NOT = {
				any_interest_group = {
					OR = { 
						has_modifier = cabinet_minister_ig
						leader ?= { has_modifier = cabinet_minister }
					}
					law_stance = {
						law = law_type:law_no_minority_issues
						value < neutral
					}
				}
			}
		}
		
		#Not blocked by constitutional values
		custom_tooltip = {
			text = blocked_by_constitutional_value.tt
			NAND = {
				has_law_or_variant = law_type:law_constitutional_supremacy
				has_variable = equality_nv_constitutionalised
			}
		}
		
	}

	modifier = {
		
		# Political Movements are happier about enactments they approve of, and less angry about enactments they disapprove of
		political_movement_radicalism_from_enactment_disapproval_mult = 0.10
		political_movement_radicalism_from_enactment_approval_mult = -0.10
		
	}
	
	possible_political_movements = {
		#law_no_minority_issues
		law_some_minority_issues
		law_full_minority_issues
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}
