#No Concentration Camp
law_slavery_banned = {
	group = lawgroup_slavery
	
	can_impose = { can_impose_law_default = yes } #progressiveness = 0
	
	icon = "gfx/interface/icons/law_icons/slavery_banned.dds"
	
	on_enact = {
		
	}

	on_activate = {
		custom_tooltip = {
			text = liberate_slaves_tt
			liberate_slaves = yes
		}
	}
	
	can_enact = {
		
		#No restriction by legal union
		THIS_is_not_restricted_by_confederal_law = yes
		
	}
	
	modifier = {
		
	}
	
	possible_political_movements = {
		#law_slavery_banned
		law_slavery_political
		law_slavery_cultural
	}
	
	pop_support = {
		value = 0
		
		add = {
			desc = "POP_SLAVES"
			if = {
				limit = { 
					is_pop_type = slaves
				}
				value = 1
			}
		}		
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}

	}

		law_slavery_banned
		#law_slavery_political
		law_slavery_cultural
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
		
		if = {
			limit = {
				has_law_or_variant = law_type:law_slavery_cultural
				NOR = {
				
					#Examples: Myanmar, Serbia, Rwanda, Darfur, Cambodia
					is_revolutionary = yes
					has_revolution = yes
					capital = { turmoil > 0.5 }
					
					#Examples: Iraq
					AND = { 
						has_variable = recently_lost_war
						capital = { turmoil > 0.5 }
					}
					
					#Examples: Nazi Germany
					AND = { 
						is_at_war = yes
						any_scope_war = {
							has_war_support = {
								target = ROOT
								value < 0
							}
						}
					}
					
				}
			}
			add = 1000
		}
		
		
	}
	
}
