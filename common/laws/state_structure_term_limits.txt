	icon = "gfx/interface/icons/law_icons/autocracy.dds"
	
	can_impose = { can_impose_law_default = yes } #progressiveness = -50
	
	on_activate = {
		
	}

	}
	
	modifier = {
		
		
		
	}
	
	possible_political_movements = {
		#law_no_term_limit
		law_yes_term_limit
	}
	
	pop_support = { value = 0 }
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

law_yes_term_limit = {
	group = lawgroup_term_limits
	
	icon = "gfx/interface/icons/law_icons/parliamentary_republic.dds"
	
	can_impose = { can_impose_law_default = yes } #progressiveness = -50
	
	on_activate = {
		
		#Reset Dynastic Republic
		if = {
			limit = { 
				is_a_dynastic_government = yes
			}
			custom_tooltip = {
				text = abolish_dynastic_republic.tt
				ruler ?= { remove_variable = dynastic_ruler_enabled }
				heir ?= { disinherit_character = yes }
			}
		}
		
	}
	
	can_enact = {
		
	}
	
	modifier = {
		
		#More agitation
		political_movement_radicalism_add = -0.10
		
	}
	
	possible_political_movements = {
		law_no_term_limit
		#law_yes_term_limit
	}
	
	pop_support = { value = 0 }
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				is_involved_in_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
	
}
