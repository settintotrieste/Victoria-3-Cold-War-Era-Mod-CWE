liberate_country_as_subject = {
	icon = "gfx/interface/icons/lens_toolbar_icons/puppet.dds"

	kind = liberate_country

	settings = {
		require_target_be_part_of_war
		country_creation
		requires_interest

		# Validation
		conflicts_with_country_creation
		conflicts_with_annex_state

		conflicts_with_existing_subject

	execution_priority = 90

	contestion_type = control_any_releasable_state

	target_type = country

	possible = {
		# trigger to determine if a goal with its target data is listed when selecting a war goal in the diplo play panel
		# scopes: root = holder, creator_country, diplomatic_play, target_country, target_state, stakeholder, target_region, article_options
	}

	}

	maneuvers = {
		value = 0

		add = {
			desc = "MANEUVERS_BASE_VALUE"
			value = 30
		}
	}

	infamy = {
		value = 5
	}

	on_enforced = {
		# script effect on top of the predefined code effect
		scope:creator_country = {
			set_variable = { name = country_liberator days = 1 }

		}
		scope:target_country = {
			set_variable = { name = country_liberated_source days = 1 }

		}
	}

		is_significant_demand = no
