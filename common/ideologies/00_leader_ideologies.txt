		value = 100	
		
		# Effect of Party Lean
		if = {
			limit = {
				is_in_government = yes
				#has_party = yes
				owner = {
					has_variable = very_radical_party
				}
			}
			add = 1000
		}
		# Effect of existing IG leader
		else_if = {
			limit = {
				is_in_government = yes
				#has_party = no
				owner = {
					government_legitimacy >= 50
				}
				leader ?= { has_ideology = ideology:ideology_autocratic }
			}
			add = 500
		}
		
		# angry IGs are more likely to have autocratic opinions
		if = {
			limit = {
				ig_approval < -5
			}
			add = 25
		}
		if = {
			limit = {
				ig_approval < -10
			}
			add = 50
		}
		
		# IGs in power are more likely to have autocratic opinions
		if = {
			limit = {
				is_in_government = yes
			}
			add = 50
		}
		
		# more likely if you border a rival
		if = {
			limit = {
				owner = {
					any_rival_country = {
						any_neighbouring_state = {
							owner = root.owner
						}
					}
				}
			}
			add = 25
		}
		# more likely if you're the defender in a war
		if = {
			limit = {
				owner = {
					is_at_war = yes
					any_scope_war = {
						defender_warleader = { this = ROOT.owner }
					}
				}
			}
			add = 25
		}
		
		# Cycle of life
		if = {
			limit = {
				is_in_government = no
				owner = { is_a_democracy = yes }
			}
			add = 50
		}
		if = {
			limit = {
				is_in_government = no
				owner = { is_a_dictatorship = yes }
			}
			add = -50
		}
		
		# National Values
		if = {
			limit = {
				owner = { has_modifier = order_national_value }
			}
			add = 25
		}
		
		# World Tension
		if = {
			limit = {
				owner = { has_global_variable = very_high_world_tension }
			}
			add = 25
		}
		
	}

		law_HOS_communist_leader = disapprove
	}
	
	interest_group_leader_weight = {
		value = 100	
		
		# Effect of Party Lean
		if = {
			limit = {
				is_in_government = yes
				#has_party = yes
				owner = {
					has_variable = very_reformist_party
				}
			}
			add = 1000
		}
		# Effect of existing IG leader
		else_if = {
			limit = {
				is_in_government = yes
				#has_party = no
				owner = {
					government_legitimacy >= 50
				}
				leader ?= { has_ideology = ideology:ideology_liberal }
			}
			add = 500
		}
		
		# Happy IG's more likely to have democratic opinions
		if = {
			limit = {
				ig_approval > 5
			}
			add = 25
		}
		if = {
			limit = {
				ig_approval > 10
			}
			add = 50
		}
		
		# IGs in opposition are more likely to have autocratic opinions
		if = {
			limit = {
				is_in_government = no
			}
			add = 100
		}
		
		# emerges to counter powerful armed forces
		if = {
			limit = {
				is_in_government = no
				owner = {
					exists = ig:ig_armed_forces
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
				}
			}
			add = 10
		}
		# more likely if a war you started isn't going well
		if = {
			limit = {
				is_in_government = no
				owner = {
					is_at_war = yes
					any_scope_war = {
						attacker_warleader = { this = ROOT.owner }
						has_war_support = {
							target = root.owner
							value < 0
						}
					}
				}
			}
			add = 100
		}
		# more likely with devastation in incorporated states
		if = {
			limit = {
				owner = {
					any_scope_state = {
						is_incorporated = yes
						devastation > 20
					}
				}
			}
			add = 50
		}
		
		# Cycle of life
		if = {
			limit = {
				is_in_government = no
				owner = { is_a_democracy = yes }
			}
			add = -50
		}
		if = {
			limit = {
				is_in_government = no
				owner = { is_a_dictatorship = yes }
			}
			add = 50
		}
		
		# National Values
		if = {
			limit = {
				owner = { has_modifier = liberty_national_value }
			}
			add = 25
		}
		
		# World Tension
		if = {
			limit = {
				owner = { has_global_variable = very_low_world_tension }
			}
			add = 25
		}
		
	}
