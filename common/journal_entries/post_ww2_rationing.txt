je_post_ww2_rationing = {
	icon = "gfx/interface/icons/invention_icons/military_statistics.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		
	}

	possible = {
		
	}

	complete = {
		
		#Sufficient SOL
		average_sol > 20
		
		#THIS is not at war
		is_at_war = no
		
	}

	on_complete = {
		
		#Please citizens
		every_primary_culture = {
			save_scope_as = primary_culture_scope
			add_loyalists = {
				value = medium_radicals
				culture = scope:primary_culture_scope
			}
		}
		
	}
	
	timeout = 3650
	
	on_timeout = {
		
		#Anger citizens
		every_primary_culture = {
			save_scope_as = primary_culture_scope
			add_radicals = {
				value = medium_radicals
				culture = scope:primary_culture_scope
			}
		}
		add_modifier = { 
			name = domestic_unrest
			months = 60
			is_decaying = yes 
		}
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		post_ww2_rationing_modifier
	}

	weight = 1000
	should_be_pinned_by_default = yes
}
