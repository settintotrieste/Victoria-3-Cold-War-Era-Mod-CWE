je_private_media_system = {
	icon = "gfx/interface/icons/invention_icons/mass_communication.dds"

	group = je_group_internal_affairs
	
	scripted_button = je_private_media_system_bribe_media_button
	scripted_button = je_private_media_system_poke_media_button
	
	is_shown_when_inactive = {
		
		#Has some free media
		OR = { 
			has_law_or_variant = law_type:law_limited_free_speech
			has_law_or_variant = law_type:law_full_free_speech
		}
		NOT = { is_a_socialist_economy = yes }
		
	}
	
	possible = {
		
		exists = ig:ig_industrialists
		
	}

	immediate = {
		
		#Set Media Leader
		if = {
			limit = { 
				exists = ig:ig_industrialists
			}
			ig:ig_industrialists = {
				save_scope_as = private_media_ig
			}
		}
		
		#Set Private Media Capacity
		set_variable = { 
			name = private_media_loyalty
			value = 50
		}
		
		#Set Private Media Action
		set_variable = { 
			name = private_media_action
			value = 0 
		}
		
	}
	
	on_monthly_pulse = {
		
		effect = {
		
			#Gain Private Media Capacity
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval >= 0
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval >= 5
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval >= 10
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval >= 15
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval >= 20
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval >= 0
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval >= 5
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval >= 10
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval >= 15
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval >= 20
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = 0.5
				}
			}
			
			#Lose Private Media Capacity
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						ig_approval < 0
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = -2.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						ig_approval < 0
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = -2.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_industrialists
					ig:ig_industrialists ?= {
						is_in_government = no
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = -2.5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_loyalty 
					exists = ig:ig_petty_bourgeoisie
					ig:ig_petty_bourgeoisie ?= {
						is_in_government = no
					}
				}
				change_variable = {
					name = private_media_loyalty
					add = -2.5
				}
			}
			
			#Apply min max values
			if = {
				limit = { has_variable = private_media_loyalty }
				clamp_variable = {
					name = private_media_loyalty
					max = 100
					min = 0
				}
			}
			
			### Private Media Effects ###
			if = {
				limit = { 
					has_variable = private_media_action
					has_law_or_variant = law_type:law_limited_free_speech
				}
				change_variable = {
					name = private_media_action
					add = 5
				}
			}
			if = {
				limit = { 
					has_variable = private_media_action
					has_law_or_variant = law_type:law_full_free_speech
				}
				change_variable = {
					name = private_media_action
					add = 10
				}
			}
			if = {
				limit = { has_variable = private_media_action }
				clamp_variable = {
					name = private_media_action
					max = 120
					min = 0
				}
			}
			if = {
				limit = {
					scope:journal_entry = {
						is_goal_complete = yes
					}
					has_variable = private_media_loyalty
					root.var:private_media_loyalty >= 50
				}
				random_list = {
					1 = {
						ruler ?= {
							add_modifier = { 
								name = positive_news_story
								years = 1
								is_decaying = yes
							}
						}
						post_notification = private_media_boost_leader_notification
					}
					1 = {
						random_scope_character = {
							limit = { 
								has_role = politician
								interest_group = {
									is_in_government = no
								}
								is_ruler = no
							}
							add_modifier = { 
								name = smear_story
								years = 1
								is_decaying = yes
							}
							interest_group = {
								save_scope_as = smeared_IG
							}
						}
						post_notification = private_media_smear_opposition_notification
					}
					1 = {
						add_modifier = { 
							name = distracting_story
							years = 1
						}
						post_notification = private_media_distract_public_notification
					}
				}
				set_variable = {
					name = private_media_action
					value = 0
				}
			}
			if = {
				limit = {
					scope:journal_entry = {
						is_goal_complete = yes
					}
					has_variable = private_media_loyalty
					root.var:private_media_loyalty < 50
				}
				random_list = {
					1 = {
						ruler ?= {
							add_modifier = { 
								name = smear_story
								years = 1
								is_decaying = yes
							}
						}
						post_notification = private_media_smear_leader_notification
					}
					1 = {
						random_scope_character = {
							limit = { 
								has_role = politician
								interest_group = {
									is_in_government = no
								}
								is_ruler = no
							}
							add_modifier = { 
								name = positive_news_story
								years = 1
								is_decaying = yes
							}
							interest_group = {
								save_scope_as = boosted_IG
							}
						}
						post_notification = private_media_boost_opposition_notification
					}
					1 = {
						add_modifier = { 
							name = depressing_story
							years = 1
						}
						post_notification = private_media_agitate_public_notification
					}
				}
				set_variable = {
					name = private_media_action
					value = 0
				}
			}
			
		}
		
	}

	fail = {
		
		OR = { 
		
			#No longer has free state media
			NOR = { 
				has_law_or_variant = law_type:law_limited_free_speech
				has_law_or_variant = law_type:law_full_free_speech
			}
			
			is_a_socialist_economy = yes
			
		}
		
	}
	
	on_fail = {
		
		#Clear variables
		remove_variable = private_media_loyalty
		remove_variable = private_media_action
		
	}
	
	current_value = {
		value =	root.var:private_media_action
	}
	
	goal_add_value = {
		add = 120
	}
	
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = private_media_advancing
				trigger = {
					has_variable = private_media_loyalty 
					OR = { 
						ig:ig_industrialists ?= {
							ig_approval >= 0
							is_in_government = yes
						}
						ig:ig_petty_bourgeoisie ?= {
							ig_approval >= 0
							is_in_government = yes
						}
					}
				}
			}
			triggered_desc = {
				desc = private_media_declining
				trigger = {
					has_variable = private_media_loyalty
					OR = {
						ig:ig_industrialists ?= {
							OR = { 
								ig_approval < 0
								is_in_government = no
							}
						}
						ig:ig_petty_bourgeoisie ?= {
							OR = { 
								ig_approval < 0
								is_in_government = no
							}
						}
					}
				}
			}
			triggered_desc = {
				desc = private_media_stable
				trigger = {
					
				}
			}
		}
	}

	weight = 100

	should_be_pinned_by_default = no

	transferable = no

	progressbar = yes
	
}
