je_head_of_state_system = {
	icon = "gfx/interface/icons/law_icons/racial_segregation.dds"

	group = je_group_political_organs
	
	scripted_button = je_fire_head_of_state_button
	
	scripted_button = je_elected_head_of_state_button
	scripted_button = je_appointed_head_of_state_button

	is_shown_when_inactive = {
		
	}
	
	possible = {
		
		#THIS has an executive PM and Ceremonial President
		OR = { 
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOS_nationalist_leader
			has_law_or_variant = law_type:law_HOS_communist_leader
			AND = { 
				has_law_or_variant = law_type:law_HOS_religious_leader
				has_journal_entry = je_national_religious_order
			}
		}
		has_law_or_variant = law_type:law_HOG_prime_minister
		
	}

	immediate = {
		
		#Find President 
		if = {
			limit = {
				AND = {
					has_law_or_variant = law_type:law_HOS_religious_leader
					has_journal_entry = je_national_religious_order
				}
				any_scope_character = {
					has_role = character_role_chief_religious_officer
					is_ruler_of_own_country = no
				}
				NOT = { has_variable = elected_head_of_state_system }
			}
			random_scope_character = {
				limit = { 
					has_role = character_role_chief_religious_officer
					is_ruler_of_own_country = no
				}
				save_scope_as = head_of_state_pm_scope
				add_character_role = character_role_head_of_state
			}
		}
		else_if = {
			limit = {
				AND = {
					has_law_or_variant = law_type:law_HOS_religious_leader
					has_journal_entry = je_national_religious_order
				}
				any_scope_character = {
					has_role_of_type = politician
					is_ruler_of_own_country = no
					interest_group ?= { is_interest_group_type = ig_devout }
				}
			}
			ordered_scope_character = {
				limit = {
					has_role_of_type = politician
					is_ruler_of_own_country = no
					interest_group ?= { is_interest_group_type = ig_devout }
				}
				order_by = popularity
				position = 0
				check_range_bounds = no
				save_scope_as = head_of_state_pm_scope
				add_character_role = character_role_head_of_state
			}
		}
		else_if = {
			limit = {
				any_scope_character = {
					has_role_of_type = politician
					is_ruler_of_own_country = no
					interest_group ?= {
						is_in_government = yes 
						is_strongest_ig_in_government = yes
					}
				}
				NOT = { has_variable = elected_head_of_state_system }
			}
			random_scope_character = {
				limit = { 
					has_role_of_type = politician
					is_ruler_of_own_country = no
					interest_group ?= { 
						is_in_government = yes 
						is_strongest_ig_in_government = yes
					}
				}
				save_scope_as = head_of_state_pm_scope
				add_character_role = character_role_head_of_state
			}
		}
		else_if = {
			limit = {
				any_scope_character = {
					has_role_of_type = politician
					is_ruler_of_own_country = no
				}
			}
			ordered_interest_group = {
				order_by = ig_clout
				position = 0
				check_range_bounds = no
				save_scope_as = elected_head_of_state_candidate_IG
			}
			random_scope_character = {
				limit = { 
					has_role_of_type = politician
					is_ruler_of_own_country = no
					interest_group ?= { is_interest_group_type = scope:elected_head_of_state_candidate_IG }
				}
				save_scope_as = head_of_state_pm_scope
				add_character_role = character_role_head_of_state
			}
		}
		else = {
			random_primary_culture = { save_scope_as = dominant_country_culture }
			ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			create_character = {
				role = character_role_head_of_state
				culture = scope:dominant_country_culture
				interest_group = scope:dominant_country_IG
				on_created = {
					add_character_role = character_role_politician
				}
			}
			random_scope_character = {
				limit = { 
					has_role = character_role_head_of_state
				}
				save_scope_as = head_of_state_pm_scope
			}
		}
		
		#Set term duration
		set_variable = { 
			name = character_role_head_of_state_term 
			value = 0
		}
		
	}
	
	current_value = {
		value =	root.var:character_role_head_of_state_term
	}

	goal_add_value = {
		add = 5
	}

	on_yearly_pulse = {
		
		#Add term ticker
		effect = {
		
			change_variable = {
				name = character_role_head_of_state_term
				add = 1
			}
			
		}
		
	}
	
	complete = {
	
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOR = { 
			has_modifier = state_of_emergency_issued 
			active_law:lawgroup_distribution_of_power ?= { has_amendment = amendment_type:amendment_wartime_suspended_elections } 
			has_modifier = wartime_measures_martial_law 
		}
		
		#Only step down during election campaign
		trigger_if = {
			limit = { 
				country_has_voting_franchise = yes 
			}
			in_election_campaign = yes
		}
		
	}
	
	on_complete = {
	
		every_scope_character = {
			limit = { has_role = character_role_head_of_state }
			remove_character_role = character_role_head_of_state
		}
		
	}
	
	fail = {
		
		OR = {
		
			#Inappropriate laws
			NAND = {
				OR = { 
					has_law_or_variant = law_type:law_HOS_president
					has_law_or_variant = law_type:law_HOS_nationalist_leader
					has_law_or_variant = law_type:law_HOS_communist_leader
					AND = { 
						has_law_or_variant = law_type:law_HOS_religious_leader
						has_journal_entry = je_national_religious_order
					}
				}
				has_law_or_variant = law_type:law_HOG_prime_minister
			}
			
			hidden_trigger = {
				NOT = { 
					scope:head_of_state_pm_scope ?= {
						has_role = character_role_head_of_state
					}
				}
			}
			
			hidden_trigger = { ruler ?= { has_role = character_role_head_of_state } }
		
		}
		
	}
	on_fail = {
		
		every_scope_character = {
			limit = { has_role = character_role_head_of_state }
			remove_character_role = character_role_head_of_state
		}
		
	}
	
	on_monthly_pulse = {
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		
	}
	
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = HOS_ig_industrialists
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_industrialists } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_petty_bourgeoisie
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_petty_bourgeoisie } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_trade_unions
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_trade_unions } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_landowners
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_landowners } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_devout
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_devout } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_rural_folk
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_rural_folk } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_intelligentsia
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_intelligentsia } }
				}
			}
			triggered_desc = {
				desc = HOS_ig_armed_forces
				trigger = {
					scope:head_of_state_pm_scope ?= { interest_group ?= { is_interest_group_type = ig_armed_forces } }
				}
			}
			triggered_desc = {
				desc = HOS_vacant
				trigger = {
					
				}
			}
		}
	}

	weight = 200

	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no

	progressbar = yes
}
