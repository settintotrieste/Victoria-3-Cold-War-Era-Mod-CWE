je_resident_general = {
	icon = "gfx/interface/icons/lens_toolbar_icons/puppet.dds"

	group = je_group_foreign_affairs

	scripted_button = je_end_protectorate_button

	is_shown_when_inactive = {
		
		OR = {
			is_a_protectorate_type_subject = yes
			is_a_republican_colonial_subject_gov = yes
			custom_tooltip = {
				text = has_dominate_politics_junior.tt
				has_diplomatic_pact = { who = ROOT type = dominate_politics is_initiator = no }
			}
		}
		
	}

	possible = {
		
	}

	immediate = {
		
		#Generate Resident Generals
		generate_resident_general_for_country = yes
		
	}

	fail = {
		
		NOR = {
			is_a_protectorate_type_subject = yes
			is_a_republican_colonial_subject_gov = yes
			custom_tooltip = {
				text = has_no_dominate_politics_junior.tt
				has_diplomatic_pact = { who = ROOT type = dominate_politics is_initiator = no }
			}
		}
		
	}
	
	on_fail = {
		
		#Clear Resident Generals
		every_scope_character = {
			limit = { has_role = character_role_resident_general }
			save_scope_as = resident_general_character_label
			custom_tooltip = {
				text = resident_general_character_label.tt
			}
			hidden_effect = { retire_character = yes }
		}
		
	}
	
	on_yearly_pulse = {
	
		effect = {
			
			#If there is no Viceroy
			generate_resident_general_for_country = yes
			
		}
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		resident_general_system
	}

	weight = 50

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
