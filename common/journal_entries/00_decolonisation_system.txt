je_generic_decolonisation_subject_system = {
	icon = "gfx/interface/icons/invention_icons/political_agitation.dds"

	group = je_group_internal_affairs
	
	scripted_button = je_generic_decolonisation_civil_disobedience_button

	is_shown_when_inactive = {
		
	}

	possible = {
	
		#Is a colonial country
		is_sufficiently_independent = no
		is_a_colonial_subject_gov = yes
		
	}
	
	complete = {
		
		#Country is independent
		is_sufficiently_independent = yes
		
	}
	on_complete = {
		
		#Independence Fervor
		add_loyalists = {
			value = large_radicals
		}
		
		#Clean up
		hidden_effect = {
			remove_modifier = colonial_citizenship_modifier_0
			remove_modifier = colonial_citizenship_modifier_1
			remove_modifier = colonial_citizenship_modifier_2
			remove_modifier = colonial_citizenship_modifier_3
			remove_modifier = colonial_citizenship_modifier_4
		}
		
	}
	
	on_monthly_pulse = {
		effect = {
			
			#Apply citizenship
			if = {
				limit = { 
					NOT = { has_modifier = colonial_citizenship_modifier_0 }
					root.overlord ?= { has_law_or_variant = law_type:law_ethnostate } 
					trigger_if = { 
						limit = { is_subject = yes }
						NOT = { any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord } }
					}
				}
				#remove_modifier = colonial_citizenship_modifier_0
				remove_modifier = colonial_citizenship_modifier_1
				remove_modifier = colonial_citizenship_modifier_2
				remove_modifier = colonial_citizenship_modifier_3
				remove_modifier = colonial_citizenship_modifier_4
				add_modifier = { name = colonial_citizenship_modifier_0 }
			}
			else_if = {
				limit = { 
					NOT = { has_modifier = colonial_citizenship_modifier_1 }
					root.overlord ?= { has_law_or_variant = law_type:law_national_supremacy } 
					trigger_if = { 
						limit = { is_subject = yes }
						NOT = { any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord } }
					}
				}
				remove_modifier = colonial_citizenship_modifier_0
				#remove_modifier = colonial_citizenship_modifier_1
				remove_modifier = colonial_citizenship_modifier_2
				remove_modifier = colonial_citizenship_modifier_3
				remove_modifier = colonial_citizenship_modifier_4
				add_modifier = { name = colonial_citizenship_modifier_1 }
			}
			else_if = {
				limit = { 
					NOT = { has_modifier = colonial_citizenship_modifier_2 }
					root.overlord ?= { has_law_or_variant = law_type:law_racial_segregation } 
					trigger_if = { 
						limit = { is_subject = yes }
						NOT = { any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord } }
					}
				}
				remove_modifier = colonial_citizenship_modifier_0
				remove_modifier = colonial_citizenship_modifier_1
				#remove_modifier = colonial_citizenship_modifier_2
				remove_modifier = colonial_citizenship_modifier_3
				remove_modifier = colonial_citizenship_modifier_4
				add_modifier = { name = colonial_citizenship_modifier_2 }
			}
			else_if = {
				limit = { 
					NOT = { has_modifier = colonial_citizenship_modifier_3 }
					root.overlord ?= { has_law_or_variant = law_type:law_cultural_exclusion } 
					trigger_if = { 
						limit = { is_subject = yes }
						NOT = { any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord } }
					}
				}
				remove_modifier = colonial_citizenship_modifier_0
				remove_modifier = colonial_citizenship_modifier_1
				remove_modifier = colonial_citizenship_modifier_2
				#remove_modifier = colonial_citizenship_modifier_3
				remove_modifier = colonial_citizenship_modifier_4
				add_modifier = { name = colonial_citizenship_modifier_3 }
			}
			else_if = {
				limit = { 
					NOT = { has_modifier = colonial_citizenship_modifier_4 }
					root.overlord ?= { has_law_or_variant = law_type:law_multicultural } 
					trigger_if = { 
						limit = { is_subject = yes }
						NOT = { any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord } }
					}
				}
				remove_modifier = colonial_citizenship_modifier_0
				remove_modifier = colonial_citizenship_modifier_1
				remove_modifier = colonial_citizenship_modifier_2
				remove_modifier = colonial_citizenship_modifier_3
				#remove_modifier = colonial_citizenship_modifier_4
				add_modifier = { name = colonial_citizenship_modifier_4 }
			}
			else_if = {
				limit = { 
					is_subject = yes
					OR = { 
						has_modifier = colonial_citizenship_modifier_0
						has_modifier = colonial_citizenship_modifier_1
						has_modifier = colonial_citizenship_modifier_2
						has_modifier = colonial_citizenship_modifier_3
						has_modifier = colonial_citizenship_modifier_4
					}
					any_primary_culture = { shares_heritage_trait_group_with_any_primary_culture = root.overlord }
				}
				remove_modifier = colonial_citizenship_modifier_0
				remove_modifier = colonial_citizenship_modifier_1
				remove_modifier = colonial_citizenship_modifier_2
				remove_modifier = colonial_citizenship_modifier_3
				remove_modifier = colonial_citizenship_modifier_4
			}
			
		}
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		colonial_country_modifier
	}
	
	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no

	progressbar = no
}
