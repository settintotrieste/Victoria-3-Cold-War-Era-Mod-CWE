je_cohabitation_system = {
	icon = "gfx/interface/icons/law_icons/racial_segregation.dds"

	group = je_group_political_organs
	
	scripted_button = je_fire_prime_minister_button

	is_shown_when_inactive = {
		
	}
	
	possible = {
		
		#THIS has an executive PM and President
		has_law_or_variant = law_type:law_HOG_president_prime_minister
		
		#Timer
		NOT = { has_variable = head_of_government_system_replace_timer }
		
	}

	immediate = {
		
		#Find PM
		if = {
			limit = {
				any_scope_character = {
					has_role = character_role_head_of_government
				}
			}
			random_scope_character = {
				limit = { 
					has_role = character_role_head_of_government
				}
				save_scope_as = cohabitation_pm_scope
			}
		}
		else_if = {
			limit = {
				any_scope_character = {
					has_role = character_role_reserved_powers_holder
					character_has_legal_restrictions = no
				}
			}
			random_scope_character = {
				limit = { 
					has_role = character_role_reserved_powers_holder
					character_has_legal_restrictions = no
				}
				save_scope_as = cohabitation_pm_scope
				add_character_role = character_role_head_of_government
			}
		}
		else_if = {
			limit = {
				any_scope_character = {
					is_ruler_of_own_country = no
					is_an_IG_leader = yes
					interest_group ?= { 
						is_in_government = yes 
						is_strongest_ig_in_government = yes
					}
					NOT = { has_variable = excluded_head_of_government_candidate }
					character_has_legal_restrictions = no
				}
			}
			random_scope_character = {
				limit = { 
					is_ruler_of_own_country = no
					is_an_IG_leader = yes
					interest_group ?= { 
						is_in_government = yes 
						is_strongest_ig_in_government = yes
					}
					NOT = { has_variable = excluded_head_of_government_candidate }
					character_has_legal_restrictions = no
				}
				save_scope_as = cohabitation_pm_scope
				add_character_role = character_role_head_of_government
			}
		}
		else = {
			random_primary_culture = { save_scope_as = dominant_country_culture }
			ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			create_character = {
				role = character_role_head_of_government
				culture = scope:dominant_country_culture
				interest_group = scope:dominant_country_IG
				on_created = {
					add_character_role = character_role_politician
				}
			}
			random_scope_character = {
				limit = { 
					has_role = character_role_head_of_government
				}
				save_scope_as = cohabitation_pm_scope
			}
		}
		
	}
	
	invalid = {
		
		OR = {
		
			#THIS has an executive PM and President
			NOT = { has_law_or_variant = law_type:law_HOG_president_prime_minister }
			
			hidden_trigger = {
				OR = {
					scope:cohabitation_pm_scope ?= {
						NOT = { has_role = character_role_head_of_government }
					}
					NOT = { exists = scope:cohabitation_pm_scope }
				}
			}
			
			#hidden_trigger = { ruler ?= { has_role = character_role_head_of_government } }
		
		}
		
	}
	on_invalid = {
		
		scope:cohabitation_pm_scope ?= {
			if = {
				limit = { has_role = character_role_head_of_government }
				remove_character_role = character_role_head_of_government
			}
		}
		
	}
	
	on_monthly_pulse = {
		
		#PM will veto laws he does not approve of
		events = {
			reserve_powers_system.8
		}
		
		effect = {
			if = {
				limit = {
					scope:cohabitation_pm_scope ?= {
						has_role = character_role_head_of_government
						is_ruler_of_own_country = yes
					}
				}
				trigger_event = { id = reserve_powers_system.19 }
			}
			if = {
				limit = {
					any_scope_character = {
						has_role = character_role_head_of_government
						count > 1
					}
				}
				every_scope_character = {
					limit = { 
						has_role = character_role_head_of_government 
						NOT = { this = scope:cohabitation_pm_scope }
					}
					remove_character_role = character_role_head_of_government
				}
			}
		}
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		cohabitation_modifier
	}
	
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = pm_ig_support
				trigger = {
					scope:cohabitation_pm_scope ?= { has_role = character_role_head_of_government }
				}
			}
			triggered_desc = {
				desc = pm_vacant
				trigger = {
					
				}
			}
		}
	}

	weight = 200

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
