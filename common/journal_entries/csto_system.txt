je_csto_system = {
	icon = "gfx/interface/icons/diplo_icons/csto_member.dds"

	group = je_group_global_international_organisations
	
	scripted_button = je_request_csto_intervention_button
	scripted_button = je_fund_csto_button
	scripted_button = je_defund_csto_button
	scripted_button = je_hold_csto_exercise_button
	
	should_be_involved = {
		OR = { 
			has_modifier = csto_leader
			has_modifier = csto_member
		}
	}
	
	should_show_when_not_involved = {
		OR = { 
			has_modifier = csto_leader
			has_modifier = csto_member
		}
	}
	
	immediate = {
		
		#Count post_colonial_org members
		random_country = {
			limit = {
				OR = {
					has_modifier = csto_leader
					has_modifier = csto_member
				}
			}
			set_global_variable = { 
				name = csto_org_member_count
				value = 0 
			}
			every_country = {
				limit = {
					OR = { 
						has_modifier = csto_leader
						has_modifier = csto_member
					}
				}
				change_global_variable = {
					name = csto_org_member_count
					add = 1
				}
			}
		}
		
		#Mark leader
		random_country = {
			limit = { has_modifier = csto_leader }
			save_scope_as = csto_leader
		}
		
		#Default is leader funds Rio Pact
		if = {
			limit = { has_modifier = csto_leader }
			set_global_variable = fund_csto
		}
		
	}
	
	on_monthly_pulse = {
		
		effect = {
		
			#Count post_colonial_org members
			random_country = {
				limit = {
					OR = {
						has_modifier = csto_leader
						has_modifier = csto_member
					}
				}
				set_global_variable = { 
					name = csto_org_member_count
					value = 0 
				}
				every_country = {
					limit = {
						OR = { 
							has_modifier = csto_leader
							has_modifier = csto_member
						}
					}
					change_global_variable = {
						name = csto_org_member_count
						add = 1
					}
				}
			}
		
		}
		
	}
	
	modifiers_while_active = {
		
	}

	invalid = {
	
		OR = {
		
			#No more members of org
			any_country = {
				count < 1
				OR = { 
					has_modifier = csto_leader
					has_modifier = csto_member
				}
			}
			
			#Different leader
			scope:csto_leader ?= {
				NOT = { has_modifier = csto_leader }
			}
			
			NOT = { exists = scope:csto_leader }
			
		}
		
	}
	
	on_invalid_all_involved = {
	
		remove_global_variable = csto_org_member_count
		
	}
	
	on_no_longer_involved = {
		
		#Clear modifier
		if = {
			limit = { has_modifier = csto_exercise }
			hidden_effect = { remove_modifier = csto_exercise }
		}
		
	}

	weight = 100

	should_be_pinned_by_default = no

	progressbar = no
}
