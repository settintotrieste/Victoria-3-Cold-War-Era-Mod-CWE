je_backbenchers_plot_system = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	group = je_group_crises

	possible = {
		
		#Cause
		any_scope_character = {
			OR = {
				AND = {
					is_ruler_of_own_country = yes
					owner ?= { has_law_or_variant = law_type:law_HOG_prime_minister }
				}
				AND = {
					has_role = character_role_head_of_government
					owner ?= { NOT = { has_law_or_variant = law_type:law_HOG_prime_minister } }
				}
			}
			OR = {
			
				#Deep unpopularity
				AND = {
					popularity < -50
					owner ?= {
						government_legitimacy < 20
						is_a_parliamentary_democracy = yes
					}
				}
				
				#Committeed Crimes
				has_variable = investigation_found_crimes
			
			}
		}
		
		#Timer
		NOT = { has_variable = backbenchers_plot_timer }
		
	}

	immediate = {
		
		#Mark ruler as backbenchers_plot so as to prevent JE from firing too frequently
		random_scope_character = {
			limit = {
				OR = {
					AND = {
						is_ruler_of_own_country = yes
						owner ?= { has_law_or_variant = law_type:law_HOG_prime_minister }
					}
					AND = {
						has_role = character_role_head_of_government
						owner ?= { NOT = { has_law_or_variant = law_type:law_HOG_prime_minister } }
					}
				}
				OR = {
				
					#Deep unpopularity
					AND = {
						popularity < -50
						owner ?= {
							government_legitimacy < 20
							is_a_parliamentary_democracy = yes
						}
					}
					
					#Committeed Crimes
					has_variable = investigation_found_crimes
				
				}
			}
			add_modifier = { name = backbenchers_plot_politician }
			save_scope_as = backbenchers_plot_prime_minister_scope
		}
		
	}
	
	# Ruler has 90 days to avoid backbenchers plot
	timeout = 90
	
	# Parliament/Party to vote after 90 days
	on_timeout = {
	
		#Timer
		set_variable = { name = backbenchers_plot_timer years = 2 }
	
		# backbencher's plot is successful
		if = {
			limit = { 
			
				OR = {
				
					#The most powerful IG within the ruling coalition still hates the PM
					any_interest_group = {
						is_in_government = yes
						leader = { is_ruler_of_own_country = no }
						OR = { 
							is_strongest_ig_in_government = yes
							ig_government_power_share >= 0.35
						}
						ig_approval < neutral
					}
					
					#Low government support to protect criminality
					AND = {
						scope:backbenchers_plot_prime_minister_scope ?= {
							has_variable = investigation_found_crimes
						}
						government_legitimacy < 66
					}
					
				}
			
			}
			
			#PM's IG is tarnished for 1 electoral cycle
			scope:backbenchers_plot_prime_minister_scope ?= {
				interest_group = {
					hidden_effect = { remove_modifier = party_in_chaos_modifier }
					add_modifier = { 
						name = party_in_chaos_modifier
						years = 4
						is_decaying = yes
					}
				}
			}
	
			#PM is removed from office
			custom_tooltip = {
				text = je_backbenchers_plot_system_1.tt
				if = {
					limit = {
						scope:backbenchers_plot_prime_minister_scope ?= { is_ruler_of_own_country = yes }
					}
					if = {
						limit = {
							any_scope_character = {
								has_role_of_type = politician
								is_ruler_of_own_country = no
								interest_group ?= { is_interest_group_type = scope:backbenchers_plot_prime_minister_scope.interest_group }
							}
						}
						ordered_scope_character = {
							limit = {
								has_role_of_type = politician
								is_ruler_of_own_country = no
								interest_group ?= { OR = { is_interest_group_type = scope:backbenchers_plot_prime_minister_scope.interest_group is_in_government = yes } }
							}
							order_by = popularity
							position = 0
							check_range_bounds = no
							set_character_as_ruler = yes
						}
					}
					else = {
						create_character = {
							ruler = yes
							interest_group = scope:backbenchers_plot_prime_minister_scope.interest_group
							on_created = {
								add_character_role = politician
							}
						}
					}
				}
				else_if = {
					limit = {
						scope:backbenchers_plot_prime_minister_scope ?= { has_role = character_role_head_of_government }
					}
					if = {
						limit = {
							any_scope_character = {
								has_role_of_type = politician
								is_ruler_of_own_country = no
								NOT = { has_role = character_role_head_of_government }
								interest_group ?= { OR = { is_interest_group_type = scope:backbenchers_plot_prime_minister_scope.interest_group is_in_government = yes } }
							}
						}
						ordered_scope_character = {
							limit = {
								has_role_of_type = politician
								is_ruler_of_own_country = no
								NOT = { has_role = character_role_head_of_government }
								interest_group ?= { OR = { is_interest_group_type = scope:backbenchers_plot_prime_minister_scope.interest_group is_in_government = yes } }
							}
							order_by = popularity
							position = 0
							check_range_bounds = no
							add_character_role = character_role_head_of_government
						}
					}
					else = {
						create_character = {
							has_role = character_role_head_of_government
							interest_group = scope:backbenchers_plot_prime_minister_scope.interest_group
							on_created = {
								add_character_role = politician
							}
						}
					}
				}
			}
			
			#PM is put on trial
			if = {
				limit = { scope:backbenchers_plot_prime_minister_scope ?= { has_variable = investigation_found_crimes } }
				scope:backbenchers_plot_prime_minister_scope ?= {
					remove_variable = investigation_found_crimes
					replace_character_roles = { add = { character_role_politician } remove_all = yes }
					add_modifier = { name = indicted_character }
					owner ?= { hidden_effect = { add_journal_entry = { type = je_political_trials_system } } }
				}
			}
			
		}
		
		# backbencher's plot is unsuccessful
		else_if = {
			limit = { 
			
				NOT = { 
					any_interest_group = {
						is_in_government = yes
						leader = { is_ruler_of_own_country = no }
						OR = { 
							is_strongest_ig_in_government = yes
							ig_government_power_share >= 0.35
						}
						ig_approval < neutral
					}
				}
			
			}
	
			#PM's popularity is raised for 1 electoral cycle
			custom_tooltip = {
				text = je_backbenchers_plot_system_2.tt
			}
			scope:backbenchers_plot_prime_minister_scope ?= {
				add_modifier = { name = survived_backbenchers_plot years = 4 is_decaying = yes } 
			}
			
			#All dissenting factions lose clout
			every_interest_group = {
				limit = {
					is_in_government = yes
					leader = { is_ruler_of_own_country = no }
					ig_approval < neutral
				}
				hidden_effect = { remove_modifier = party_in_chaos_modifier }
				add_modifier = {
					name = party_in_chaos_modifier
					years = 2
					is_decaying = yes
				}
			}
			
		}
		
		#Clear modifiers
		scope:backbenchers_plot_prime_minister_scope ?= {
			if = {
				limit = { has_modifier = backbenchers_plot_politician }
				remove_modifier = backbenchers_plot_politician
			}
		}
		
	}
	
	invalid = {
		
		OR = {
			
			#Ineligible character
			NOT = { exists = scope:backbenchers_plot_prime_minister_scope }
			scope:backbenchers_plot_prime_minister_scope ?= { OR = { has_modifier = indicted_character has_role = character_role_prisoner } }
			scope:backbenchers_plot_prime_minister_scope ?= { 
				NOR = {
					AND = {
						is_ruler_of_own_country = yes
						owner ?= { has_law_or_variant = law_type:law_HOG_prime_minister }
					}
					AND = {
						has_role = character_role_head_of_government
						owner ?= { NOT = { has_law_or_variant = law_type:law_HOG_prime_minister } }
					}
				}
			}
			scope:backbenchers_plot_prime_minister_scope ?= { NOT = { owner ?= ROOT } }
			
			#Prison Breakout
			capital ?= {
				market_access <= 0 # Occupied
			}
		
		}
		
	}
	
	on_invalid = {
	
		#Clear backbenchers_plot markers
		scope:backbenchers_plot_prime_minister_scope ?= {
			if = {
				limit = { has_modifier = backbenchers_plot_politician }
				remove_modifier = backbenchers_plot_politician
			}
		}
		
	}
	
	weight = 200

	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no

	progressbar = no
	
}
