je_tekhpromfinplans = {
	icon = "gfx/interface/icons/invention_icons/philosophical_pragmatism.dds"

	group = je_group_historical_content

	scripted_button = je_start_inject_tekhpromfinplans_button
	scripted_button = je_stop_inject_tekhpromfinplans_button

	is_shown_when_inactive = {
		
		has_law_or_variant = law_type:law_semi_command_economy
		
	}

	possible = {
		
		#Semi Planned Economy
		has_law_or_variant = law_type:law_semi_command_economy
		
	}

	immediate = {
		
		calculate_tekhpromfinplans = yes
		
	}

	fail = {
		
		#No longer a Semi Planned Economy
		NOT = { has_law_or_variant = law_type:law_semi_command_economy }
		
	}
	
	on_fail = {
		
		remove_variable = tekhpromfinplans_fund_size
		
		remove_variable = injecting_tekhpromfinplans
		
	}
	
	on_monthly_pulse = {
	
		effect = {
			
			#Auto injection
			if = {
				limit = { has_variable = injecting_tekhpromfinplans }
				inject_tekhpromfinplans = yes
			}
			else = {
				calculate_tekhpromfinplans = yes
			}
			
		}
		
	}

	weight = 50

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}

je_state_run_companies = {
	icon = "gfx/interface/icons/invention_icons/central_planning.dds"

	group = je_group_historical_content

	scripted_button = je_start_inject_tekhpromfinplans_button
	scripted_button = je_stop_inject_tekhpromfinplans_button

	is_shown_when_inactive = {
		
		has_law_or_variant = law_type:law_command_economy
		
	}

	possible = {
		
		#Has companies
		any_company = { 
			count > 0 
		}
		
	}

	immediate = {
		
		calculate_tekhpromfinplans = yes
		
	}

	fail = {
		
		#No longer a Semi Planned Economy
		OR = { 
			NOT = { has_law_or_variant = law_type:law_command_economy }
			any_company = {
				count = 0
			}
		}
		
	}
	
	on_fail = {
		
		remove_variable = tekhpromfinplans_fund_size
		
		remove_variable = injecting_tekhpromfinplans
		
	}
	
	on_monthly_pulse = {
	
		effect = {
			
			#Auto injection
			if = {
				limit = { has_variable = injecting_tekhpromfinplans }
				inject_tekhpromfinplans = yes
			}
			else = {
				calculate_tekhpromfinplans = yes
			}
			
		}
		
	}

	weight = 50

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
