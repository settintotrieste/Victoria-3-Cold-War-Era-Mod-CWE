je_rio_pact_system = {
	icon = "gfx/interface/icons/diplo_icons/rio_pact_member.dds"

	group = je_group_global_international_organisations
	
	scripted_button = je_request_rio_pact_intervention_button
	scripted_button = je_fund_rio_pact_button
	scripted_button = je_defund_rio_pact_button
	scripted_button = je_hold_rio_pact_exercise_button
	
	should_be_involved = {
		OR = { 
			has_modifier = rio_pact_leader
			has_modifier = rio_pact_member
		}
	}
	
	should_show_when_not_involved = {
		OR = { 
			has_modifier = rio_pact_leader
			has_modifier = rio_pact_member
		}
	}
	
	immediate = {
		
		#Count post_colonial_org members
		random_country = {
			limit = {
				OR = {
					has_modifier = rio_pact_leader
					has_modifier = rio_pact_member
				}
			}
			set_global_variable = { 
				name = rio_pact_org_member_count
				value = 0 
			}
			every_country = {
				limit = {
					OR = { 
						has_modifier = rio_pact_leader
						has_modifier = rio_pact_member
					}
				}
				change_global_variable = {
					name = rio_pact_org_member_count
					add = 1
				}
			}
		}
		
		#Mark leader
		random_country = {
			limit = { has_modifier = rio_pact_leader }
			save_scope_as = rio_pact_leader
		}
		
		#Default is leader funds Rio Pact
		if = {
			limit = { has_modifier = rio_pact_leader }
			set_global_variable = fund_rio_pact
		}
		
	}
	
	on_monthly_pulse = {
		
		effect = {
		
			#Count post_colonial_org members
			random_country = {
				limit = {
					OR = {
						has_modifier = rio_pact_leader
						has_modifier = rio_pact_member
					}
				}
				set_global_variable = { 
					name = rio_pact_org_member_count
					value = 0 
				}
				every_country = {
					limit = {
						OR = { 
							has_modifier = rio_pact_leader
							has_modifier = rio_pact_member
						}
					}
					change_global_variable = {
						name = rio_pact_org_member_count
						add = 1
					}
				}
			}
		
		}
		
	}
	
	modifiers_while_active = {
		
	}

	invalid = {
	
		OR = {
		
			#No more members of org
			any_country = {
				count < 1
				OR = { 
					has_modifier = rio_pact_leader
					has_modifier = rio_pact_member
				}
			}
			
			#Different leader
			scope:rio_pact_leader ?= {
				NOT = { has_modifier = rio_pact_leader }
			}
			
			NOT = { exists = scope:rio_pact_leader }
			
		}
		
	}
	
	on_invalid_all_involved = {
	
		remove_global_variable = rio_pact_org_member_count
		
	}
	
	on_no_longer_involved = {
		
		#Clear modifier
		if = {
			limit = { has_modifier = rio_pact_exercise }
			hidden_effect = { remove_modifier = rio_pact_exercise }
		}
		
	}

	weight = 100

	should_be_pinned_by_default = no

	progressbar = no
}
