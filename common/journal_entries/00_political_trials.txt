je_political_trials_system = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	scripted_button = je_political_trials_drop_charges_button
	scripted_button = je_political_trials_bribe_judge_button

	possible = {
		
	}

	immediate = {
		
		#Mark character to be tried
		random_scope_character = {
			limit = {
				has_modifier = indicted_character
				NOT = { has_role = character_role_prisoner }
			}
			save_scope_as = indicted_character_scope
			interest_group ?= {
				save_scope_as = indicted_character_ig_scope
			}
		}
		
		#Trial duration
		set_variable = {
			name = trial_progress_var
			value = 0
		}
		
	}
	
	current_value = {
		value =	root.var:trial_progress_var
	}

	goal_add_value = {
		add = 12
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		scope:indicted_character_scope ?= {
			has_modifier = indicted_character
			NOT = { has_role = character_role_prisoner }
			is_busy = no
			is_in_battle = no
		}
	}
	
	on_complete = {
	
		#Guaranteed conviction if indicted character committed crimes OR there was a kangaroo court
		if = {
			limit = {
				scope:indicted_character_scope ?= {
					
					OR = {
						
						## Character has committed a crime ##
						
						#Corruption scandal
						has_modifier = ignored_corruption
						has_modifier = corrupt_connections_exposed
						has_modifier = corrupt_connections_exposed_minor
						has_modifier = corruption_scandal_character_modifier
						has_modifier = dealing_with_corruption
						has_trait = grifter
						has_trait = framed_for_crimes
						
						#Traitor
						has_modifier = treasonous_traitor
						
						#Abuse of Power
						has_modifier = seizing_power
						has_trait = war_criminal
						
						# Personal Misconduct
						AND = {
							OR = {
								has_trait = cocaine_addiction
								has_trait = opium_addiction
							}
							owner ?= {
								OR = {
									has_law_or_variant = law_type:law_no_minority_issues
									has_law_or_variant = law_type:law_some_minority_issues
								}
							}
						}
						AND = {
							OR = { 
								has_trait = alcoholic
								has_trait = hedonist
							}
							owner ?= { 
								has_law_or_variant = law_type:law_no_minority_issues
							}
						}
						AND = {
							has_trait = homosexual_character
							owner ?= { has_law_or_variant = law_type:law_no_gender_issues }
						}
						
					}
					
				}
			}
			scope:indicted_character_scope ?= {
				
				#Imprison Character
				hidden_effect = {
					remove_modifier = indicted_character
					replace_character_roles = { add = { character_role_prisoner } remove_all = yes }
					add_modifier = { name = convicted_character years = 10 }
					set_variable = { name = convicted_character_flag months = 1 }
				}
				
			}
			custom_tooltip = {
				text = government_wins_political_trial.tt
				trigger_event = { id = political_trials_system.1 } #Government won trial
			}
		}
		else_if = {
			limit = {
				OR = { 
					AND = { 
						has_law_or_variant = law_type:law_politicised_judiciary 
						has_law_or_variant = law_type:law_kangaroo_law 
					}
					has_variable = political_trials_bribed_judge
				}
			}
			scope:indicted_character_scope ?= {
			
				#Imprison Character
				hidden_effect = {
					remove_modifier = indicted_character
					replace_character_roles = { add = { character_role_prisoner } remove_all = yes }
					add_modifier = { name = convicted_character years = 10 }
					set_variable = { name = convicted_character_flag months = 1 }
				}
				
			}
			custom_tooltip = {
				text = government_wins_political_trial.tt
				trigger_event = { id = political_trials_system.1 } #Government won trial
			}
		}
		
		#50% chance of conviction if no crimes committed IF court is somewhat crooked
		else_if = {
			limit = {
				OR = { 
					NOT = { has_law_or_variant = law_type:law_independent_judiciary }
					NOT = { has_law_or_variant = law_type:law_common_law }
				}
			}
			
			random_list = { 
			
				#Character is acquitted
				1 = {
					scope:indicted_character_scope ?= {
						hidden_effect = {
							remove_modifier = indicted_character
							add_modifier = { name = acquitted_character years = 10 is_decaying = yes }
							set_variable = { name = acquitted_character_flag months = 1 }
						}
					}
					custom_tooltip = {
						text = government_loses_political_trial.tt
						trigger_event = { id = political_trials_system.2 } #Government lost trial
					}
				}
				
				#Character is convicted
				1 = {
					scope:indicted_character_scope ?= {
					
						#Imprison Character
						hidden_effect = {
							remove_modifier = indicted_character
							replace_character_roles = { add = { character_role_prisoner } remove_all = yes }
							add_modifier = { name = convicted_character years = 10 }
							set_variable = { name = convicted_character_flag months = 1 }
						}
						
					}
					custom_tooltip = {
						text = government_wins_political_trial.tt
						trigger_event = { id = political_trials_system.1 } #Government won trial
					}
				}
				
			}
			
		}
		
		#Guaranteed acquittal if courts are independent
		else = {
			
			scope:indicted_character_scope ?= {
				hidden_effect = {
					remove_modifier = indicted_character
					add_modifier = { name = acquitted_character years = 10 is_decaying = yes }
					set_variable = { name = acquitted_character_flag months = 1 }
				}
			}
			custom_tooltip = {
				text = government_loses_political_trial.tt
				trigger_event = { id = political_trials_system.2 } #Government lost trial
			}
			
		}
		
		
	}

	on_monthly_pulse = {
		
		effect = {
		
			#Add trial ticker
			if = {
				limit = { 
					has_law_or_variant = law_type:law_politicised_judiciary
					has_law_or_variant = law_type:law_kangaroo_law
				}
				change_variable = {
					name = trial_progress_var
					add = 3
				}
			}
			else_if = {
				limit = { 
					OR = { 
						NOT = { has_law_or_variant = law_type:law_independent_judiciary }
						NOT = { has_law_or_variant = law_type:law_common_law }
					}
				}
				change_variable = {
					name = trial_progress_var
					add = 2
				}
			}
			else = {
				change_variable = {
					name = trial_progress_var
					add = 1
				}
			}
			
			#Add domestic reaction
			if = {
				limit = { has_law_or_variant = law_type:law_full_free_speech }
				add_radicals = {
					value = 0.04
					interest_group = scope:indicted_character_ig_scope
				}
			}
			else_if = {
				limit = { has_law_or_variant = law_type:law_limited_free_speech }
				add_radicals = {
					value = 0.02
					interest_group = scope:indicted_character_ig_scope
				}
			}
			else_if = {
				limit = { has_law_or_variant = law_type:law_censorship }
				add_radicals = {
					value = 0.01
					interest_group = scope:indicted_character_ig_scope
				}
			}
			
		}
		
	}
	
	fail = {
		OR = {
			scope:indicted_character_scope ?= {
				OR = {
					is_ruler_of_own_country = yes
					is_heir_of_own_country = yes
					character_has_diplomatic_immunity = yes
				}
			}
			custom_tooltip = {
				text = indicted_character_is_acquitted.tt
				scope:indicted_character_scope ?= {
					OR = {
						NOT = { has_modifier = indicted_character }
						has_modifier = acquitted_character
					}
				}
			}
		}
	}
	
	on_fail = {
		scope:indicted_character_scope ?= {
			hidden_effect = {
				remove_modifier = indicted_character
				add_modifier = { name = acquitted_character years = 2 is_decaying = yes }
				set_variable = { name = acquitted_character_flag months = 1 }
			}
		}
		custom_tooltip = {
			text = government_loses_political_trial.tt
			trigger_event = { id = political_trials_system.2 } #Government lost trial
		}
	}
	
	invalid = {
		OR = {
			
			#Ineligible character
			NOT = { exists = scope:indicted_character_scope }
			scope:indicted_character_scope ?= { has_role = character_role_prisoner }
			scope:indicted_character_scope ?= { NOT = { owner ?= ROOT } }
			
			#Prison Breakout
			capital ?= {
				market_access <= 0 # Occupied
			}
			
		}
	}
	
	on_invalid = {
		scope:indicted_character_scope ?= {
			if = {
				limit = { has_modifier = indicted_character }
				remove_modifier = indicted_character
			}
		}
	}
	
	weight = 200

	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no

	progressbar = yes
	
}
