je_reserved_powers = {
	icon = "gfx/interface/icons/law_icons/presidential_republic.dds"

	group = je_group_political_organs

	scripted_button = je_fire_council_chairperson_button

	possible = {
	
		#Has reserve powers system
		has_law_or_variant = law_type:law_hybrid_chamber
	
		#Not a revolutionary country
		is_revolutionary = no is_secessionist = no
		
		#Timer
		NOT = { has_variable = ceremonial_head_of_state_replace_timer }
		
	}

	immediate = {
		
		#Create second executive
		if = {
			limit = {
				NOT = {
					any_scope_character = {
						has_role = character_role_reserved_powers_holder
					}
				}
			}
			
			#law_HOS_monarch
			if = {
				limit = {
					has_law_or_variant = law_type:law_HOS_monarch
					has_law_or_variant = law_type:law_HOG_prime_minister
					any_scope_character = {
						has_role = character_role_monarch
					}
				}
				random_scope_character = {
					limit = {
						has_role = character_role_monarch
					}
					add_character_role = character_role_reserved_powers_holder
				}
			}
			
			#law_HOS_president
			if = {
				limit = {
					has_law_or_variant = law_type:law_HOS_president
					NOT = { has_law_or_variant = law_type:law_HOG_prime_minister }
					any_scope_character = {
						has_role = character_role_head_of_state
					}
				}
				random_scope_character = {
					limit = {
						has_role = character_role_head_of_state
					}
					add_character_role = character_role_reserved_powers_holder
				}
			}
			
			#law_HOS_religious_leader
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_HOS_religious_leader
					has_law_or_variant = law_type:law_HOG_prime_minister
					any_scope_character = {
						has_role = character_role_chief_religious_officer
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
					}
				}
				random_scope_character = {
					limit = { 
						has_role = character_role_chief_religious_officer
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
					}
					add_character_role = character_role_reserved_powers_holder
				}
			}
			
			#law_HOS_nationalist_leader
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_HOS_nationalist_leader
					has_law_or_variant = law_type:law_HOG_prime_minister
					any_scope_character = {
						has_modifier = head_of_armed_forces
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
					}
				}
				random_scope_character = {
					limit = { 
						has_modifier = head_of_armed_forces
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
					}
					add_character_role = character_role_reserved_powers_holder
				}
			}
			
			#law_HOS_communist_leader
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_HOS_communist_leader
					any_scope_character = {
						has_role_of_type = politician
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
						interest_group ?= {
							OR = { 
								is_in_government = yes
								is_interest_group_type = ig_intelligentsia
								is_interest_group_type = ig_armed_forces
							}
						}
					}
				}
				random_scope_character = {
					limit = { 
						has_role_of_type = politician
						is_ruler_of_own_country = no
						NOT = { has_variable = excluded_reserved_powers_holder_candidate }
						interest_group ?= {
							OR = { 
								is_in_government = yes
								is_interest_group_type = ig_intelligentsia
								is_interest_group_type = ig_armed_forces
							}
						}
					}
					add_character_role = character_role_reserved_powers_holder
				}
			}
			
			#Everyone else
			else = {
				random_interest_group = {
					limit = {
						OR = { 
							is_in_government = yes
							is_interest_group_type = ig_intelligentsia
							is_interest_group_type = ig_armed_forces
						}
					}
					save_scope_as = reserved_powers_holder_IG
				}
				create_character = {
					role = character_role_reserved_powers_holder
					interest_group = scope:reserved_powers_holder_IG
					on_created = { 
						
					}
				}
			}
			
		}
		random_scope_character = {
			limit = {
				has_role = character_role_reserved_powers_holder
			}
			save_scope_as = ceremonial_head_of_state
			interest_group = { save_scope_as = reserved_powers_holder_IG }
		}
		
		#Cannot fire head of state within 1 year
		set_variable = { name = ceremonial_head_of_state_fire_timer year = 1 }
		
	}
	
	invalid = {
	
		OR = {
			NOT = { has_law_or_variant = law_type:law_hybrid_chamber }
			NOT = { exists = scope:ceremonial_head_of_state }
			scope:ceremonial_head_of_state ?= { NOT = { has_role = character_role_reserved_powers_holder } }
			AND = {
				has_law_or_variant = law_type:law_HOS_president
				scope:ceremonial_head_of_state ?= { NOT = { has_role = character_role_head_of_state } }
			}
		}
	
	}
	
	on_invalid = {
		
		scope:ceremonial_head_of_state ?= {
			if = {
				limit = { has_role = character_role_reserved_powers_holder }
				remove_character_role = character_role_reserved_powers_holder
			}
		}
		remove_variable = ceremonial_head_of_state_veto_overruled
		
	}
	
	on_monthly_pulse = {
		
		#President will veto laws he does not approve of
		events = {
			reserve_powers_system.1
		}
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		#reserved_powers_system
	}
	
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = reserved_powers_holder_ig_support
				trigger = {
					scope:ceremonial_head_of_state ?= { has_role = character_role_reserved_powers_holder }
				}
			}
			triggered_desc = {
				desc = reserved_powers_holder_vacant
				trigger = {
					
				}
			}
		}
	}

	weight = 4

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
