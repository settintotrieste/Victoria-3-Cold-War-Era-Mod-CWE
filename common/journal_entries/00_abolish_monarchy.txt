# triggered by on_law_enactment_started
je_abolish_monarchy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_internal_affairs

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = { has_law_or_variant = law_type:law_HOS_monarch }
					NOT = {
						any_civil_war = {
							civil_war_progress > 0.75
							is_civil_war_type = revolution
						}
					}
					NOT = {
						any_interest_group = {
							law_stance = {
								law = law_type:law_HOS_monarch
								value > neutral
							}
							is_powerful = yes
						}
					}
				}
				change_variable = {
					name = abolishing_monarchy_var
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = { has_law_or_variant = law_type:law_HOS_monarch }
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.75
				is_civil_war_type = revolution
			}
		}
		NOT = {
			any_interest_group = {
				law_stance = {
					law = law_type:law_HOS_monarch
					value > neutral
				}
				is_powerful = yes
			}
		}
	}
	
	immediate = {
		ig:ig_landowners = {
			save_scope_as = landowners_ig
		}
	}

	on_complete = {
		set_variable = stamped_out_monarchy_var
		ig:ig_landowners = {
			add_modifier = {
				name = monarchist_nostalgia
				days = normal_modifier_time
			}
		}
	}

	invalid = {
		NOR = {
			is_enacting_law = law_type:law_HOS_president
			is_enacting_law = law_type:law_HOS_religious_leader
			is_enacting_law = law_type:law_HOS_nationalist_leader
			is_enacting_law = law_type:law_HOS_communist_leader
			has_law_or_variant = law_type:law_HOS_president
			has_law_or_variant = law_type:law_HOS_religious_leader
			has_law_or_variant = law_type:law_HOS_nationalist_leader
			has_law_or_variant = law_type:law_HOS_communist_leader
		}
	}

	current_value = {
		value = root.var:abolishing_monarchy_var
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes

	weight = 100
}
