je_judiciary = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_political_organs

	scripted_button = je_judiciary_appoint_judges_button
	#scripted_button = je_judiciary_expand_court_button

	possible = {
		
		#Judiciary has power
		NOT = { has_law_or_variant = law_type:law_kangaroo_law }
		
		#Not in a legal union
		custom_tooltip = {
			text = is_not_in_a_legal_union.tt
			NOT = {
				any_scope_diplomatic_pact = {
					is_diplomatic_action_type = legal_union
					second_country = ROOT
				}
			}
		}
		
	}

	immediate = {
		
		#Default is an even split
		set_variable = {
			name = judiciary_progressiveness
			value = 50
		}
		
	}
	
	invalid = {
	
		OR = {
		
			#Judiciary no longer has any power
			has_law_or_variant = law_type:law_kangaroo_law
			
			#Is in a legal union
			custom_tooltip = {
				text = is_in_a_legal_union.tt
				any_scope_diplomatic_pact = {
					is_diplomatic_action_type = legal_union
					second_country = ROOT
				}
			}
		
		}
		
	}
	
	on_invalid = {
		
		#Reset judicial progressiveness
		remove_variable = judiciary_progressiveness
		
		#Apply foreign Supreme Court
		if = {
			limit = {
				any_scope_diplomatic_pact = {
					is_diplomatic_action_type = legal_union
					second_country = ROOT
				}
			}
			add_journal_entry = { type = je_judiciary_foreign }
		}
		
	}
	
	on_monthly_pulse = {
		
		random_events = {
			2 = 0
			1 = judiciary_system.17
		}
		
	}
	
	on_yearly_pulse = {
	
		effect = {
		
			#Calculate Progressiveness
			if = {
				limit = { has_variable = judiciary_progressiveness }
				
				#Ruler
				if = {
					limit = { ruler ?= { character_is_autocratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = -1
					}
				}
				else_if = {
					limit = { ruler ?= { character_is_democratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = 1
					}
				}
				
				#Head of State
				if = {
					limit = { has_journal_entry = je_head_of_state_system any_scope_character = { has_role = character_role_head_of_state character_is_autocratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = -1
					}
				}
				else_if = {
					limit = { has_journal_entry = je_head_of_state_system any_scope_character = { has_role = character_role_head_of_state character_is_democratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = 1
					}
				}
				
				#Head of Government
				if = {
					limit = { has_journal_entry = je_cohabitation_system any_scope_character = { has_role = character_role_head_of_government character_is_autocratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = -1
					}
				}
				else_if = {
					limit = { has_journal_entry = je_cohabitation_system any_scope_character = { has_role = character_role_head_of_government character_is_democratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = 1
					}
				}
				
				#Reserve Powers Holder
				if = {
					limit = { has_journal_entry = je_reserved_powers any_scope_character = { has_role = character_role_reserved_powers_holder character_is_autocratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = -1
					}
				}
				else_if = {
					limit = { has_journal_entry = je_reserved_powers any_scope_character = { has_role = character_role_reserved_powers_holder character_is_democratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = 1
					}
				}
				
				#Head of the Civil Service
				if = {
					limit = { has_journal_entry = je_civil_service_system any_scope_character = { is_an_IG_leader = yes interest_group ?= { is_interest_group_type = ig_intelligentsia } character_is_autocratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = -1
					}
				}
				else_if = {
					limit = { has_journal_entry = je_civil_service_system any_scope_character = { is_an_IG_leader = yes interest_group ?= { is_interest_group_type = ig_intelligentsia } character_is_democratic = yes } }
					change_variable = {
						name = judiciary_progressiveness
						add = 1
					}
				}
				
				clamp_variable = {
					name = judiciary_progressiveness
					max = 100
					min = 0
				}
				
			}
			
		}
		
		random_events = {
			15 = 0
			1 = judiciary_system.1
			1 = judiciary_system.2
			1 = judiciary_system.3 
			1 = judiciary_system.4
			1 = judiciary_system.5
			1 = judiciary_system.6
			1 = judiciary_system.7
			1 = judiciary_system.8
			1 = judiciary_system.9
			1 = judiciary_system.10
			1 = judiciary_system.11
			1 = judiciary_system.12
			1 = judiciary_system.13
			1 = judiciary_system.14
			1 = judiciary_system.15
		}
		
	}

	weight = 3

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}

je_judiciary_foreign = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_political_organs

	possible = {
		
		#Judiciary has power
		NOT = { has_law_or_variant = law_type:law_kangaroo_law }
		
		#Is in a legal union
		custom_tooltip = {
			text = is_in_a_legal_union.tt
			any_scope_diplomatic_pact = {
				is_diplomatic_action_type = legal_union
				second_country = ROOT
			}
		}
		
	}

	immediate = {
		
		random_country = {
			limit = { has_diplomatic_pact = { who = ROOT type = legal_union is_initiator = yes } }
			save_scope_as = legal_union_foreign_supreme_court
		}
		
		#Set progressiveness
		if = {
			limit = { scope:legal_union_foreign_supreme_court ?= { has_variable = judiciary_progressiveness } }
			set_variable = {
				name = judiciary_progressiveness
				value = scope:legal_union_foreign_supreme_court.var:judiciary_progressiveness
			}
		}
		else = {
			set_variable = {
				name = judiciary_progressiveness
				value = 50
			}
		}
		
	}
	
	invalid = {
	
		OR = { 
		
			#Judiciary no longer has any power
			has_law_or_variant = law_type:law_kangaroo_law
		
			#Not in a legal union
			custom_tooltip = {
				text = is_not_in_a_legal_union.tt
				NOT = { 
					any_scope_diplomatic_pact = {
						is_diplomatic_action_type = legal_union
						second_country = ROOT
					}
				}
			}
			
		}
		
	}
	
	on_invalid = {
		
		#Reset judicial progressiveness
		remove_variable = judiciary_progressiveness
		
	}

	on_monthly_pulse = {
		
		random_events = {
			2 = 0
			1 = judiciary_system.17
		}
		
	}
	
	on_yearly_pulse = {
		
		effect = {
			
			#Set progressiveness
			if = {
				limit = { scope:legal_union_foreign_supreme_court ?= { has_variable = judiciary_progressiveness } }
				set_variable = {
					name = judiciary_progressiveness
					value = scope:legal_union_foreign_supreme_court.var:judiciary_progressiveness
				}
			}
		
		}
		
		random_events = {
			15 = 0
			1 = judiciary_system.1
			1 = judiciary_system.2
			1 = judiciary_system.3 
			1 = judiciary_system.4
			1 = judiciary_system.5
			1 = judiciary_system.6
			1 = judiciary_system.7
			1 = judiciary_system.8
			1 = judiciary_system.9
			1 = judiciary_system.10
			1 = judiciary_system.11
			1 = judiciary_system.12
			1 = judiciary_system.13
			1 = judiciary_system.14
			1 = judiciary_system.15
		}
		
	}

	weight = 3

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
