je_term_limits = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_political_organs

	scripted_button = je_create_collective_presidency_button

	is_shown_when_inactive = {
		
		#Not any other historical or generic system
		NOR = {
			has_journal_entry = je_term_limits_collective_HOS
			has_journal_entry = je_term_limits_monarchy
		}
		
		#Not for monarchs
		is_a_regime_monarchy = no
		
		#Ruler exists
		exists = ruler
		
	}

	possible = {
		
		#Not a special case
		NOT = {
			active_law:lawgroup_governance_principles ?= {
				has_amendment = amendment_type:amendment_rotating_presidency
			}
		}
		
		#There is a term limit
		has_law_or_variant = law_type:law_yes_term_limit
		
	}

	immediate = {
		
		#Mark term
		set_variable = {
			name = term_limit_var
			value = 0
		}
		
		#Mark current ruler
		ruler ?= {
			save_scope_as = the_current_president_scope
			interest_group ?= {
				save_scope_as = the_current_president_ig_scope
			}
		}
		
	}
	
	fail = {
	
		#President was changed along the way
		scope:the_current_president_scope ?= { is_ruler_of_own_country = no }
		
	}
	
	on_fail = {
		
		#VP becomes president
		if = {
			limit = {
				NOT = { exists = heir }
				any_scope_character = {
					has_role_of_type = politician
					interest_group = { 
						OR = {
							is_in_government = yes 
							is_interest_group_type = scope:the_current_president_ig_scope 
						} 
					}
					is_ruler_of_own_country = no
					count >= 1
				}
			}
			random_scope_character = {
				limit = { 
					has_role_of_type = politician
					interest_group = {
						OR = { 
							is_in_government = yes 
							is_interest_group_type = scope:the_current_president_ig_scope 
						}
					}
					is_ruler_of_own_country = no
				}
				set_character_as_ruler = yes
			}
		}
		else_if = {
			limit = {
				NOT = { exists = heir }
			}
			create_character = {
				interest_group = scope:the_current_president_ig_scope
				ruler = yes
			}
		}
		
		#Reset Term Limits
		remove_variable = term_limit_var
		
	}
	
	invalid = {
	
		OR = {
		
			active_law:lawgroup_governance_principles ?= {
				has_amendment = amendment_type:amendment_rotating_presidency
			}
			
			is_a_regime_monarchy = yes
			
			NOT = { has_law_or_variant = law_type:law_yes_term_limit }
			
			NOT = { exists = scope:the_current_president_scope }
			
			scope:the_current_president_scope ?= { NOT = { owner ?= ROOT } }
			
		}
	
	}
	
	on_invalid = {
		
		#Reset Term Limits
		remove_variable = term_limit_var
		
	}

	current_value = {
		value =	root.var:term_limit_var
	}

	goal_add_value = {
		add = 8
	}

	on_yearly_pulse = {
		
		#Add term ticker
		effect = {
		
			change_variable = {
				name = term_limit_var
				add = 1
			}
			
		}
		
	}

	complete = {
		
		#Term completed
		scope:journal_entry = {
			is_goal_complete = yes
		}
		
		#Elections are not suspended
		NOR = { has_modifier = state_of_emergency_issued active_law:lawgroup_distribution_of_power ?= { has_amendment = amendment_type:amendment_wartime_suspended_elections } has_modifier = wartime_measures_martial_law }
		
		#Only step down during election campaign
		trigger_if = {
			limit = { 
				country_has_voting_franchise = yes 
			}
			in_election_campaign = yes
		}
		
	}
	
	on_complete = {
	
		#President's term has ended so he shall retire
		custom_tooltip = {
			text = je_term_limits.tt
			scope:the_current_president_scope ?= { retire_character = yes }
		
			#Reset Term Limits
			remove_variable = term_limit_var
		}
		
	}

	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}

		is_a_regime_monarchy = yes

			interest_group ?= {
				save_scope_as = the_current_president_ig_scope
			}
		}
		
		#Generate Elective Monarchs
		heir ?= {
			if = {
				limit = { NOT = { has_role = character_role_monarch_elective } }
				add_character_role = character_role_monarch_elective
			}
		}
		generate_elective_monarchs_for_country = yes
		
	}
	
	fail = {
	
		#Monarch was changed along the way
		scope:the_current_president_scope ?= { is_ruler_of_own_country = no }
		
	}
	
	on_fail = {
		
		#Another Elected Monarch becomes Monarch
		if = {
			limit = {

					limit = { NOT = { has_role = character_role_monarch } }
					add_character_role = character_role_monarch
				}
				if = {
					limit = { has_role = character_role_monarch_elective }
					remove_character_role = character_role_monarch_elective
				}
			}
			scope:the_current_president_scope ?= {
				if = {
					limit = { NOT = { has_role = character_role_monarch_elective } }
					add_character_role = character_role_monarch_elective
				}
				if = {
					limit = { has_role = character_role_monarch }
					remove_character_role = character_role_monarch
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = heir }

			generate_elective_monarchs_for_country = yes
			
			#Select monarch

				}

					remove_character_role = character_role_monarch_elective
				}
			}
			scope:the_current_president_scope ?= { 
				if = {
					limit = { NOT = { has_role = character_role_monarch_elective } }
					add_character_role = character_role_monarch_elective
				}
				if = {
					limit = { has_role = character_role_monarch }
					remove_character_role = character_role_monarch
				}
			}
			
		}
		else_if = {
			limit = {
				exists = heir
			}
			heir ?= {
				set_character_as_ruler = yes
				if = {
					limit = { NOT = { has_role = character_role_monarch } }

				}
				if = {
					limit = { has_role = character_role_monarch_elective }
					remove_character_role = character_role_monarch_elective
				}
			}
			scope:the_current_president_scope ?= { 
				if = {
					limit = { NOT = { has_role = character_role_monarch_elective } }
					add_character_role = character_role_monarch_elective
				}
				if = {
					limit = { has_role = character_role_monarch }
