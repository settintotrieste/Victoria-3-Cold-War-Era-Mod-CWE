je_paramilitary_groups = {
	icon = "gfx/interface/icons/invention_icons/mandatory_service.dds"

	group = je_group_political_organs

	scripted_button = je_conduct_great_purge_button
	scripted_button = je_paramilitary_system_dismiss_military_chief_button

	is_shown_when_inactive = {
	
		#Has Secret Police
		has_law_or_variant = law_type:law_secret_police
		
	}

	possible = {
		
		exists = root.ruler.interest_group
		
		#Has Secret Police
		has_law_or_variant = law_type:law_secret_police
		
		#Has a strong military
		OR = {
			has_law_or_variant = law_type:law_professional_army
			has_law_or_variant = law_type:law_mass_conscription
		}
		
		#Politicised Military
		has_law_or_variant = law_type:law_politicised_armed_forces
		
	}

	immediate = {
		
		#Set interest group for Paramilitary Group
		ruler ?= { interest_group ?= { save_scope_as = paramilitary_IG } }
		if = {
			limit = { exists = scope:paramilitary_IG }
			random_list = {
				1 = {
					create_character = {
						is_agitator = yes
						ideology = ideology_autocratic
						interest_group = scope:paramilitary_IG
						on_created = {
							add_modifier = { name = head_of_paramilitary }
						}
					}
				}
				1 = {
					create_character = {
						is_agitator = yes
						ideology = ideology_power_hungry
						interest_group = scope:paramilitary_IG
						on_created = {
							add_modifier = { name = head_of_paramilitary }
						}
					}
				}
				1 = {
					create_character = {
						is_agitator = yes
						ideology = ideology_elitist
						interest_group = scope:paramilitary_IG
						on_created = {
							add_modifier = { name = head_of_paramilitary }
						}
					}
				}
			}
			random_scope_character = {
				limit = { has_modifier = head_of_paramilitary }
				save_scope_as = head_of_paramilitary
				add_character_role = general
			}
		}
		
		#Loyalty of Paramilitary Group
		if = {
			limit = { exists = scope:head_of_paramilitary }
			set_variable = { 
				name = paramilitary_groups_loyalty
				value = scope:head_of_paramilitary.loyalty
			}
			clamp_variable = {
				name = paramilitary_groups_loyalty
				max = 100
				min = 0
			}
		}
		else = {
			set_variable = { 
				name = paramilitary_groups_loyalty
				value = 0
			}
		}
		
	}

	invalid = {
	
		OR = {
		
			NAND = {
		
				has_law_or_variant = law_type:law_secret_police
				
				OR = {
					has_law_or_variant = law_type:law_professional_army
					has_law_or_variant = law_type:law_mass_conscription
				}
				
				has_law_or_variant = law_type:law_politicised_armed_forces
			
			}
			
			NOT = { exists = scope:head_of_paramilitary }
			
			scope:head_of_paramilitary ?= {
				OR = { 
					NOT = { owner ?= ROOT }
					NOT = { has_modifier = head_of_paramilitary }
				}
			}
			
		}
		
	}
	
	on_invalid = {
		
		if = {
			limit = { has_modifier = conducting_great_purge }
			remove_modifier = conducting_great_purge
		}
		
		remove_variable = paramilitary_groups_loyalty
		remove_variable = paramilitary_groups_loyalty_monthly
		
		#Clear modifiers
		scope:head_of_paramilitary ?= {
			if = {
				limit = { has_modifier = head_of_paramilitary }
				remove_modifier = head_of_paramilitary
			}
		}
		
		remove_variable = paramilitary_group_veto_overruled
		
	}
	
	on_monthly_pulse = {
	
		effect = {
			
			#Loyalty of Paramilitary Group
			if = {
				limit = { exists = scope:head_of_paramilitary }
				set_variable = { 
					name = paramilitary_groups_loyalty
					value = scope:head_of_paramilitary.loyalty
				}
				clamp_variable = {
					name = paramilitary_groups_loyalty
					max = 100
					min = 0
				}
			}
			else = {
				set_variable = { 
					name = paramilitary_groups_loyalty
					value = 0
				}
			}
			
		}
		
		#Paramilitary Group will veto laws it does not approve of
		events = {
			reserve_powers_system.2
		}
		
		#Once a year chance of a Palace Coup
		random_events = {
			11 = 0
			1 = revolution_system.5
		}
		
	}
	
	# zero or more static modifiers that will be applied to the Journal Entry when it activates, where they will propagate to the country
	modifiers_while_active = {
		paramilitary_groups_system
	}

	weight = 11

	should_be_pinned_by_default_uninvolved_or_context = no

	transferable = no

	progressbar = no
}
