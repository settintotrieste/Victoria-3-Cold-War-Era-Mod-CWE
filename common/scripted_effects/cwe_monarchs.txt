		set_character_as_ruler = yes
		if = {
			limit = { has_role = agitator }
			remove_character_role = agitator

	}
	if = {
		limit = { NOT = { has_modifier = monarch_character } }
		add_modifier = { name = monarch_character }
	}

}

clean_up_any_monarchs = {
	
	if = {
		limit = { ruler ?= { has_modifier = monarch_character } }
		
		#Constitutional Monarchy
		if = {
			limit = { is_a_monarchy = yes }
			ruler ?= { 
				remove_character_role = ruler
				remove_character_role = politician
				add_character_role = agitator
				interest_group ?= { save_scope_as = deposed_monarch_IG }
			}
			if = {
				limit = { exists = scope:deposed_monarch_IG }
				create_character = {
					ruler = yes
					ig_leader = yes
					noble = yes
					interest_group = scope:deposed_monarch_IG
				}
			}
			else = {
				get_ruler_for:parliamentary_elective = {
					set_character_as_ruler = yes
				}
			}
		}
		
		#Republics depose their monarch
		else = {
			ruler ?= { 
				if = {
					limit = { has_modifier = monarch_character }
					remove_modifier = monarch_character
				}
				exile_character_with_martyrdom = yes 
			}
		}
		
		heir ?= {
			disinherit_character = yes
		}
		
	}
	
	if = {
		limit = { 
			has_journal_entry = je_constitutional_monarchy
			any_scope_character = { 
				has_modifier = monarch_character 
				has_role = politician
				is_ruler = no
			}
		}
		random_scope_character = {
			limit = {
				has_modifier = monarch_character 
				has_role = politician
				is_ruler = no
			}
			remove_character_role = politician
			add_character_role = agitator
			interest_group ?= { save_scope_as = deposed_monarch_IG }
		}
		if = {
			limit = { exists = scope:deposed_monarch_IG }
			create_character = {
				ig_leader = yes
				noble = yes
				interest_group = scope:deposed_monarch_IG
			}
		}
	}
	
	set_variable = { 
		name = monarchy_deposed_recently
		months = 1
	}
	
}

	random_scope_character = {
		limit = {

	}

resolve_any_exiled_monarch = {

	if = {
		limit = { 
			OR = { 
				AND = { 
					is_ruler = no
					owner ?= { NOT = { owner = ROOT } }
				}
				is_in_exile_pool = yes 
				is_in_void = yes
			}
		}
		if = {
			limit = { is_in_void = yes }
			free_character_from_void = yes
		}
		transfer_character = root
		hidden_effect = {
			remove_modifier = in_exile_modifier
		}
	}

}

generate_viceroy_for_country = {

	#Generic Viceroy
	if = {
		limit = {
			any_scope_character = {
				has_modifier = viceroy_character
				is_ruler = no
			}
		}
		random_scope_character = {
			limit = {
				has_modifier = viceroy_character
				is_ruler = no
			}
			set_character_as_viceroy = yes
		}
	}
	else = {
		if = {
			limit = { is_sufficiently_independent = no }
			overlord ?= {
				random_primary_culture = { save_scope_as = culture_to_add }
			}
			if = {
				limit = { exists = scope:culture_to_add }
				create_character = {
					save_scope_as = ruler_to_be
					is_agitator = yes
					culture = scope:culture_to_add
					on_created = { 
						set_character_as_viceroy = yes
					}
				}
			}
			else = {
				create_character = {
					save_scope_as = ruler_to_be
					is_agitator = yes
					on_created = { 
						set_character_as_viceroy = yes
					}
				}
			}
		}
		else = {
			create_character = {
				save_scope_as = ruler_to_be
				is_agitator = yes
				on_created = { 
					set_character_as_viceroy = yes
				}
			}
		}
	}

}
