count_superstate_members = {

	#Determine members
	set_variable = { 
		name = superstate_members
		value = 0
	}
	if = {
		limit = {
			OR = {
				has_variable = superstate_free_trade_zone
				has_variable = superstate_customs_union
				has_variable = superstate_defence_union
				has_variable = superstate_political_union
			}
		}
		change_variable = { 
			name = superstate_members
			add = 1
		}
		every_country = {
			limit = { 
				shares_same_superstate = yes
				OR = {
					has_variable = superstate_free_trade_zone
					has_variable = superstate_customs_union
					has_variable = superstate_defence_union
					has_variable = superstate_political_union
				}
			}
			ROOT = {
				change_variable = { 
					name = superstate_members
					add = 1
				}
			}
		}
	}
	else = {
		change_variable = { 
			name = superstate_members
			add = 1
		}
		every_country = {
			limit = { 
				shares_same_superstate = yes
			}
			ROOT = {
				change_variable = { 
					name = superstate_members
					add = 1
				}
			}
		}
	}
	
	#Determine majority
	set_variable = { 
		name = superstate_members_majority
		value = root.var:superstate_members
	}
	change_variable = {
		name = superstate_members_majority
		divide = 2
	}

}

create_superstate_free_trade_zone = {

	every_country = {
		limit = { 
			has_variable = superstate_free_trade_zone
			shares_same_superstate = yes
			NOR = {
				has_diplomatic_pact = { who = ROOT type = trade_agreement }
				is_junior_in_customs_union = yes
			}
		}
		create_diplomatic_pact = {
			country = ROOT
			type = trade_agreement
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_customs_union = {
	
	every_country = {
		limit = { 
			has_variable = superstate_customs_union
			shares_same_superstate = yes
		}
		if = {
			limit = { 
				can_create_diplomatic_pact = { target = ROOT type = foreign_investment_agreement }
				NOT = { has_diplomatic_pact = { who = ROOT type = foreign_investment_agreement } } 
			}
			create_diplomatic_pact = {
				country = ROOT
				type = foreign_investment_agreement
			}
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_currency_union = {

	#Enter into Currency Union with others
	ordered_country = {
		limit = { 
			NOT = { THIS = ROOT }
			has_variable = superstate_currency_union
			shares_same_superstate = yes
			NOT = { has_diplomatic_pact = { who = ROOT type = currency_union } }
		}
		order_by = country_rank
		create_diplomatic_pact = {
			country = ROOT
			type = currency_union
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_defence_union = {

	if = {
		limit = { 
			is_in_power_bloc = yes
			power_bloc ?= { 
				has_identity = identity:identity_superstate 
				NOT = { has_principle = principle:principle_defensive_cooperation_3 }
			}
		}
		power_bloc ?= { 
			add_principle = principle_defensive_cooperation_3
		}
	}
	
	every_country = {
		limit = { 
			has_variable = superstate_defence_union
			shares_same_superstate = yes
			NOR = {
				has_diplomatic_pact = { who = ROOT type = defensive_pact }
				has_diplomatic_pact = { who = ROOT type = alliance }
			}
		}
		create_diplomatic_pact = {
			country = ROOT
			type = defensive_pact
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_confederation = {

	#Confederate THIS with Superstate
	if = {
		limit = { 
			is_power_bloc_leader = no
			is_in_power_bloc = yes
			power_bloc ?= { 
				has_identity = identity:identity_superstate
				power_bloc_leader = {
					shares_same_superstate = yes
				}
			}
			NOT = { is_subject_type = subject_type_confederal_state }
			is_sufficiently_independent = yes
		}
		power_bloc ?= { 
			power_bloc_leader ?= {
				create_diplomatic_pact = {
					country = ROOT
					type = confederal_state
				}
			}
		}
	}

}

	#THIS becomes Superstate

}
