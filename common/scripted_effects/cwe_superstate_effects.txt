count_superstate_members = {

	#Determine members
	set_variable = { 
		name = superstate_members
		value = 0
	}
	if = {
		limit = {
			OR = {
				has_variable = superstate_free_trade_zone
				has_variable = superstate_customs_union
				has_variable = superstate_defence_union
				has_variable = superstate_political_union
			}
		}
		change_variable = { 
			name = superstate_members
			add = 1
		}
		every_country = {
			limit = { 
				shares_same_superstate = yes
				OR = {
					has_variable = superstate_free_trade_zone
					has_variable = superstate_customs_union
					has_variable = superstate_defence_union
					has_variable = superstate_political_union
				}
			}
			ROOT = {
				change_variable = { 
					name = superstate_members
					add = 1
				}
			}
		}
	}
	else = {
		change_variable = { 
			name = superstate_members
			add = 1
		}
		every_country = {
			limit = { 
				shares_same_superstate = yes
			}
			ROOT = {
				change_variable = { 
					name = superstate_members
					add = 1
				}
			}
		}
	}
	
	#Determine majority
	set_variable = { 
		name = superstate_members_majority
		value = root.var:superstate_members
	}
	change_variable = {
		name = superstate_members_majority
		divide = 2
	}

}

		limit = { 
			has_variable = superstate_free_trade_zone
			shares_same_superstate = yes
			NOR = {
				any_scope_treaty = { binds = ROOT any_scope_article = { has_type = trade_privilege } }
				is_junior_in_customs_union = yes
			}
		}
		save_scope_as = THIS_scope
		create_treaty = {
			name = treaty_name_superstate_FTA
			first_country = scope:THIS_scope
			second_country = ROOT

			is_draft = no
			#binding_period = { years = 10 }

			articles_to_create = {
				{
					article = trade_privilege
					source_country = scope:THIS_scope
					target_country = ROOT
				}
				{
					article = trade_privilege
					source_country = ROOT
					target_country = scope:THIS_scope
				}
				{
					article = transit_rights
					source_country = scope:THIS_scope
					target_country = ROOT
				}
				{
					article = transit_rights
					source_country = ROOT
					target_country = scope:THIS_scope
				}
			}
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_customs_union = {
	
	every_country = {
		limit = { 
			has_variable = superstate_customs_union
			shares_same_superstate = yes
			NOT = { any_scope_treaty = { binds = ROOT any_scope_article = { has_type = foreign_investment_rights } } }
		}
		save_scope_as = THIS_scope
		create_treaty = {
			name = treaty_name_superstate_CU
			first_country = scope:THIS_scope
			second_country = ROOT

			is_draft = no
			#binding_period = { years = 10 }

			articles_to_create = {
				{
					article = foreign_investment_rights
					source_country = scope:THIS_scope
					target_country = ROOT
				}
				{
					article = foreign_investment_rights
					source_country = ROOT
					target_country = scope:THIS_scope
				}
			}
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}
	
	#Create superstate power bloc
	if = {
		limit = { 
			OR = { 
				is_in_power_bloc = no
				AND = { 
					OR = { 
						is_power_bloc_leader = yes
						is_in_power_bloc = yes
					}
					power_bloc ?= { 
						OR = { 
							power_bloc_leader = { NOT = { shares_same_superstate = yes } }
							NOT = { has_identity = identity:identity_superstate } 
						}
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					any_power_bloc = {
						has_identity = identity:identity_superstate
						power_bloc_leader = {
							shares_same_superstate = yes
						}
					}
				}
			}
			ordered_country = {
				limit = {
					shares_same_superstate = yes
					is_power_bloc_leader = no
				}
				order_by = country_rank
				save_scope_as = superstate_power_bloc
			}
			if = {
				limit = { exists = scope:superstate_power_bloc }
				scope:superstate_power_bloc = {
					if = { limit = { has_variable = XNA_ambitions } create_power_bloc = { name =  XNA map_color = { 66 94 193 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XCA_ambitions } create_power_bloc = { name =  XCA map_color = { 40 233 255 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XSA_ambitions } create_power_bloc = { name =  XSA map_color = { 0 36 144 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XEU_ambitions } create_power_bloc = { name =  XEU map_color = { 70 130 180 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XER_ambitions } create_power_bloc = { name =  XER map_color = { 177 73 165 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XAL_ambitions } create_power_bloc = { name =  XAL map_color = { 0 98 51 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XAU_ambitions } create_power_bloc = { name =  XAU map_color = { 120 98 57 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XEA_ambitions } create_power_bloc = { name =  XEA map_color = { 0 56 168 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XSE_ambitions } create_power_bloc = { name =  XSE map_color = { 255 53 35 } identity = identity_superstate } }
					else_if = { limit = { has_variable = XPI_ambitions } create_power_bloc = { name =  XPI map_color = { 1 41 115 } identity = identity_superstate } }
				}
				if = {
					limit = {  
						OR = {
							is_in_power_bloc = yes
							power_bloc ?= { 
								OR = { 
									power_bloc_leader = { NOT = { shares_same_superstate = yes } }
									NOT = { has_identity = identity:identity_superstate } 
								}
							}
						}
					}
					join_power_bloc = scope:superstate_power_bloc
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					is_in_power_bloc = no
					power_bloc ?= { 
						OR = { 
							power_bloc_leader = { NOT = { shares_same_superstate = yes } }
							NOT = { has_identity = identity:identity_superstate } 
						}
					}
				}
				any_power_bloc = {
					has_identity = identity:identity_superstate
					power_bloc_leader = {
						shares_same_superstate = yes
					}
				}
			}
			random_country = {
				limit = { 
					shares_same_superstate = yes
					is_in_power_bloc = yes
					is_power_bloc_leader = yes
				}
				save_scope_as = superstate_power_bloc
			}
			if = {
				limit = { exists = scope:superstate_power_bloc }
				join_power_bloc = scope:superstate_power_bloc
			}
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_defence_union = {

	if = {
		limit = { 
			is_in_power_bloc = yes
			power_bloc ?= { 
				has_identity = identity:identity_superstate 
				NOT = { has_principle = principle:principle_defensive_cooperation_3 }
			}
		}
		power_bloc ?= { 
			add_principle = principle_defensive_cooperation_3
		}
	}
	
	every_country = {
		limit = { 
			has_variable = superstate_defence_union
			shares_same_superstate = yes
			NOR = {
				any_scope_treaty = { binds = ROOT any_scope_article = { has_type = defensive_pact } }
				any_scope_treaty = { binds = ROOT any_scope_article = { has_type = alliance } }
			}
		}
		save_scope_as = target_country
		create_treaty = {
			name = treaty_name_superstate_alliance
			first_country = ROOT
			second_country = scope:target_country

			is_draft = no
			#binding_period = { years = 10 }

			articles_to_create = {
				{
					article = defensive_pact
				}
			}
		}
		change_relations = { 
			country = ROOT
			value = 25
		}
	}

}

create_superstate_confederation = {

	#Confederate THIS with Superstate
	if = {
		limit = { 
			is_power_bloc_leader = no
			is_in_power_bloc = yes
			power_bloc ?= { 
				has_identity = identity:identity_superstate
				power_bloc_leader = {
					shares_same_superstate = yes
				}
			}
			NOT = { is_subject_type = subject_type_confederal_state }
			is_sufficiently_independent = yes
		}
		power_bloc ?= { 
			power_bloc_leader ?= {
				create_diplomatic_pact = {
					country = ROOT
					type = confederal_state
				}
			}
		}
	}

}
