count_bretton_woods_members = {

	set_variable = { 
		name = bretton_woods_total_members
		value = 0
	}
	
	every_country = {
		limit = { 
			OR = { 
				is_involved_in_journal_entry = je_bretton_woods_system_leader
				is_involved_in_journal_entry = je_bretton_woods_system_member
			} 
		}
		ROOT = {
			change_variable = { 
				name = bretton_woods_total_members
				add = 1
			}
		}
	}

}
