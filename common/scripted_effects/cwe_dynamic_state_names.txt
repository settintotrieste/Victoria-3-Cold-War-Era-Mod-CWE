REPLACE_OR_CREATE:STATE_SANTA_CATARINA_state_name_assign = {
    if = {
        limit = {
            owner = {
                c:BRZ ?= this

					is_a_presidential_republic = yes
					is_a_parliamentary_republic = yes
                }
            }
        }

        set_hub_name = {
            type = port

        reset_hub_names = yes
    }
}

            name = HUB_NAME_STATE_EASTERN_THRACE_wood_russian

            type = mine
            name = HUB_NAME_STATE_EASTERN_THRACE_mine_russian

            type = port
            name = HUB_NAME_STATE_EASTERN_THRACE_port_russian

REPLACE_OR_CREATE:STATE_INGRIA_state_name_assign = {
	if = {
		limit = {
			owner = {
				country_has_primary_culture = cu:russian
				is_a_communist_government = yes
			}
		}
        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_INGRIA_city_russian_sov # St. Petersburg only becomes Leningrad if Lenin has been around to be a namesake
        }
	}
	else_if = {
		limit = {
			owner = {
				country_has_primary_culture = cu:russian
				is_a_monarchy = yes
				OR = {
					any_rival_country = {
						any_primary_culture = {
							has_discrimination_trait = heritage_germanic
						}
					}
					any_rivaling_country = {
						any_primary_culture = {
							has_discrimination_trait = heritage_germanic
						}					
					}
				}
			}
		}
        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_INGRIA_city_russian_imp # St. Petersburg only becomes Petrograd when Germanophobia becomes a thing
        }
	}
	else = { # switch back to defaults
		reset_state_name = yes
		reset_hub_names = yes
	}
}

REPLACE_OR_CREATE:evaluate_and_assign_state_hub_dynamic_names = {
	if = {
		limit = {
			has_game_rule = no_dynamic_naming
		}
		reset_state_name = yes
	    reset_hub_names = yes
	}
	else = {
		assign_state_hub_dynamic_names = yes
		cwe_assign_state_hub_dynamic_names = yes
	}
}

	if = { # Indochina Region

	}

}

cwe_STATE_ZAMBEZI_state_name_assign = {
    if = {
        limit = {
            owner = {
                country_has_primary_culture = cu:rhodesian
            }
        }

        set_state_name = STATE_ZAMBEZI_british
        set_hub_names = british
    }
    else_if = {
        limit = {
            owner = {
                OR = {
                    c:GBR ?= this
                    c:SAF ?= this
                    AND = {
                        exists = c:GBR
                        exists = top_overlord
                        top_overlord = c:GBR
                    }
                }
            }
        }

        set_state_name = STATE_ZAMBEZI_british
        set_hub_names = british
    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}

REPLACE_OR_CREATE:STATE_UPPER_SILESIA_state_name_assign = {
    if = {
        limit = {
            owner = {
                country_has_primary_culture = cu:polish
                NOT = {
                    any_primary_culture = {
                        has_discrimination_trait = language_germanophone
                    }
                }
            }
        }
        set_state_name = STATE_UPPER_SILESIA_polish
        set_hub_names = polish
    }
    else_if = {
        limit = {
            owner = {
                country_has_primary_culture = cu:austrian
                NOT = {
                    owns_entire_state_region = STATE_UPPER_SILESIA
                }
            }
        }
        set_state_name = STATE_UPPER_SILESIA_german_austrian
        reset_hub_names = yes # Hubs are German by default
    }
    else_if = {
        limit = {
            owner = {
                country_has_primary_culture = cu:german
                NOT = {
                    any_primary_culture = {
                        cu:polish ?= this
                    }
                }
            }
        }
        set_state_name = STATE_UPPER_SILESIA_german
        reset_hub_names = yes # Hubs are German by default
    }
    else_if = {
        limit = {
            owner = {
                any_primary_culture = {
                    cu:czech ?= this
                }
            }
        }
        if = {
            limit = {
                is_split_state = yes
            }
            set_state_name = STATE_UPPER_SILESIA_czech_split
        }
        else = {
            set_state_name = STATE_UPPER_SILESIA_czech
        }
        set_hub_names = czech
    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}
