calculate_election_momentum_fluctuation = {

	### Fluctuate polls ###
	
	#Always first set to 0 before calculating anything
	set_variable = {
		name = election_fluctuation_net_display
		value = 0
	}
	set_variable = {
		name = election_fluctuation_economy_display
		value = 0
	}
	set_variable = {
		name = election_fluctuation_politics_display
		value = 0
	}
	set_variable = {
		name = election_fluctuation_government_display
		value = 0
	}
	set_variable = {
		name = election_fluctuation_corruption_display
		value = 0
	}
	set_variable = {
		name = election_fluctuation_incumbency_display
		value = 0
	}
	
	#Country has bad economy = people hate gov (Max: -4%/month)
	if = {
		limit = { 
			has_healthy_economy = no
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.02
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = -2
		}
	}
	if = {
		limit = { 
			in_default = yes 
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.02
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = -2
		}
	}
	
	#Economic outlook (Max: -2%/month)
	if = {
		limit = { 
			has_modifier = economic_cycle_1
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.02
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = -2
		}
	}
	if = {
		limit = { 
			has_modifier = economic_cycle_2
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = -1
		}
	}
	if = {
		limit = { 
			has_modifier = economic_cycle_4
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = 0.01
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = 1
		}
	}
	if = {
		limit = { 
			has_modifier = economic_cycle_5
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = 0.02
		}
		change_variable = {
			name = election_fluctuation_economy_display
			add = 2
		}
	}
	
	#Country has turmoil = people hate gov (Max: -10%/month)
	if = {
		limit = { 
			country_turmoil > 0.1
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.2
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.3
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.4
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.5
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.6
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.7
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.8
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil > 0.9
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	if = {
		limit = { 
			country_turmoil = 1.0
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_politics_display
			add = -1
		}
	}
	
	#Cabinet infighting = more unpopularity (Max: -2%/month)
	if = {
		limit = { 
			has_variable = low_cabinet_unity
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.02
		}
		change_variable = {
			name = election_fluctuation_government_display
			add = -2
		}
	}
	if = {
		limit = { 
			has_variable = medium_cabinet_unity
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_government_display
			add = -1
		}
	}
	
	#Corruption (Max: -5%/month)
	if = {
		limit = { 
			OR = { 
				AND = { 
					has_law_or_variant = law_type:law_very_low_anti_corruption
					NOR = { 
						is_enacting_law = law_type:law_low_anti_corruption
						is_enacting_law = law_type:law_medium_anti_corruption
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_low_anti_corruption
					NOR = { 
						is_enacting_law = law_type:law_medium_anti_corruption
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_medium_anti_corruption
					NOR = { 
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_high_anti_corruption
					is_enacting_law = law_type:law_very_high_anti_corruption
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.05
		}
		change_variable = {
			name = election_fluctuation_corruption_display
			add = -5
		}
	}
	if = {
		limit = { 
			OR = { 
				AND = { 
					has_law_or_variant = law_type:law_very_low_anti_corruption
					OR = { 
						is_enacting_law = law_type:law_low_anti_corruption
						is_enacting_law = law_type:law_medium_anti_corruption
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_low_anti_corruption
					OR = { 
						is_enacting_law = law_type:law_medium_anti_corruption
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_medium_anti_corruption
					OR = { 
						is_enacting_law = law_type:law_high_anti_corruption
						is_enacting_law = law_type:law_very_high_anti_corruption
					}
				}
				AND = { 
					has_law_or_variant = law_type:law_high_anti_corruption
					is_enacting_law = law_type:law_very_high_anti_corruption
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = 0.05
		}
		change_variable = {
			name = election_fluctuation_corruption_display
			add = 5
		}
	}
	
	#Incumbency disadvantage (Max: -5%/month)
	if = {
		limit = { 
			#Dominant Party System does not penalise incumbents
			NOT = { has_law_or_variant = law_type:law_parties_allowed }
			any_active_party = {
				any_member = { 
					is_in_government = yes 
					ig_clout >= 0.2
					ig_clout < 0.3
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.01
		}
		change_variable = {
			name = election_fluctuation_incumbency_display
			add = -1
		}
	}
	else_if = {
		limit = { 
			any_active_party = {
				any_member = { 
					is_in_government = yes 
					ig_clout >= 0.3
					ig_clout < 0.4
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.02
		}
		change_variable = {
			name = election_fluctuation_incumbency_display
			add = -2
		}
	}
	else_if = {
		limit = { 
			any_active_party = {
				any_member = { 
					is_in_government = yes 
					ig_clout >= 0.4
					ig_clout < 0.5
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.03
		}
		change_variable = {
			name = election_fluctuation_incumbency_display
			add = -3
		}
	}
	else_if = {
		limit = { 
			any_active_party = {
				any_member = { 
					is_in_government = yes 
					ig_clout >= 0.5
					ig_clout < 0.6
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.04
		}
		change_variable = {
			name = election_fluctuation_incumbency_display
			add = -4
		}
	}
	else_if = {
		limit = { 
			any_active_party = {
				any_member = { 
					is_in_government = yes 
					ig_clout >= 0.6
				}
			}
		}
		every_active_party  = {
			limit = { 
				any_member = { is_in_government = yes } 
			}
			add_momentum = -0.05
		}
		change_variable = {
			name = election_fluctuation_incumbency_display
			add = -5
		}
	}
	
	#Calculate net momentum
	change_variable = {
		name = election_fluctuation_net_display
		add = var:election_fluctuation_economy_display
	}
	change_variable = {
		name = election_fluctuation_net_display
		add = var:election_fluctuation_politics_display
	}
	change_variable = {
		name = election_fluctuation_net_display
		add = var:election_fluctuation_government_display
	}
	change_variable = {
		name = election_fluctuation_net_display
		add = var:election_fluctuation_corruption_display
	}
	change_variable = {
		name = election_fluctuation_net_display
		add = var:election_fluctuation_incumbency_display
	}

}
