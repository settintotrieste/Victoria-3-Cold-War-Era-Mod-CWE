set_appropriate_character_ideology_from_overlord = {

	if = {
		limit = { owner.overlord.ruler = { has_ideology = ideology:ideology_autocratic } }
		set_ideology = ideology:ideology_autocratic
	}
	else_if = {
		limit = { owner.overlord.ruler = { has_ideology = ideology:ideology_power_hungry } }
		set_ideology = ideology:ideology_power_hungry
	}
	else_if = {
		limit = { owner.overlord.ruler = { has_ideology = ideology:ideology_elitist } }
		set_ideology = ideology:ideology_elitist
	}
	else_if = {
		limit = { owner.overlord.ruler = { has_ideology = ideology:ideology_democratic } }
		set_ideology = ideology:ideology_democratic
	}
	else_if = {
		limit = { owner.overlord.ruler = { has_ideology = ideology:ideology_liberal } }
		set_ideology = ideology:ideology_liberal
	}

}

generate_resident_general_for_country = {

	#Generate Resident Generals
	if = {
		limit = {
			NOT = {
				any_scope_character = {
					has_role = character_role_resident_general
				}
			}
		}
		every_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = dominate_politics is_initiator = yes }
				save_temporary_scope_as = overlord_country
				ROOT = { NOT = { is_subject_of = scope:overlord_country } }
			}
			save_scope_as = dominant_country
			random_primary_culture = { save_scope_as = dominant_country_culture }
			ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			create_character = {
				role = character_role_resident_general
				culture = scope:dominant_country_culture
				religion = scope:dominant_country.religion
				interest_group = scope:dominant_country_IG
				on_created = {
					#add_character_role = character_role_agitator
					set_variable = { name = new_resident_general days = 1 }
				}
			}
			random_scope_character = {
				limit = { has_variable = new_resident_general }
				remove_variable = new_resident_general
				set_home_country = scope:dominant_country
				set_home_state = scope:dominant_country.capital
				set_appropriate_character_ideology_from_overlord = yes
			}
		}
		if = {
			limit = { 
				OR = { 
					is_a_protectorate_type_subject = yes
					is_a_republican_colonial_subject_gov = yes
				}
			}
			overlord ?= { 
				save_scope_as = dominant_country
				random_primary_culture = { save_scope_as = dominant_country_culture }
				ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			}
			create_character = {
				role = character_role_resident_general
				culture = scope:dominant_country_culture
				religion = scope:dominant_country.religion
				interest_group = scope:dominant_country_IG
				on_created = {
					#add_character_role = character_role_agitator
					set_variable = { name = new_resident_general days = 1 }
				}
			}
			random_scope_character = {
				limit = { has_variable = new_resident_general }
				remove_variable = new_resident_general
				set_home_country = scope:dominant_country
				set_home_state = scope:dominant_country.capital
				set_appropriate_character_ideology_from_overlord = yes
			}
		}
	}
	else_if = {
		limit = { 
			any_scope_character = {
				has_role = character_role_resident_general
			}
		}
		every_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = dominate_politics is_initiator = yes }
				ROOT = { NOT = { is_subject_of = scope:overlord_country } }
				save_temporary_scope_as = dominant_country
				ROOT = {
					NOT = { 
						any_scope_character = {
							has_role = character_role_resident_general
							scope:dominant_country = this.home_country
						}
					}
				}
			}
			save_scope_as = dominant_country
			random_primary_culture = { save_scope_as = dominant_country_culture }
			ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			create_character = {
				role = character_role_resident_general
				culture = scope:dominant_country_culture
				religion = scope:dominant_country.religion
				interest_group = scope:dominant_country_IG
				on_created = {
					#add_character_role = character_role_agitator
					set_variable = { name = new_resident_general days = 1 }
				}
			}
			random_scope_character = {
				limit = { has_variable = new_resident_general }
				remove_variable = new_resident_general
				set_home_country = scope:dominant_country
				set_home_state = scope:dominant_country.capital
				set_appropriate_character_ideology_from_overlord = yes
			}
		}
		if = {
			limit = { 
				OR = { 
					is_a_protectorate_type_subject = yes
					is_a_republican_colonial_subject_gov = yes
				}
				NOT = { 
					any_scope_character = {
						has_role = character_role_resident_general
						root.overlord = this.home_country
					}
				}
			}
			overlord ?= { 
				save_scope_as = dominant_country
				random_primary_culture = { save_scope_as = dominant_country_culture }
				ruler ?= { interest_group = { save_scope_as = dominant_country_IG } }
			}
			create_character = {
				role = character_role_resident_general
				culture = scope:dominant_country_culture
				religion = scope:dominant_country.religion
				interest_group = scope:dominant_country_IG
				on_created = {
					#add_character_role = character_role_agitator
					set_variable = { name = new_resident_general days = 1 }
				}
			}
			random_scope_character = {
				limit = { has_variable = new_resident_general }
				remove_variable = new_resident_general
				set_home_country = scope:dominant_country
				set_home_state = scope:dominant_country.capital
				set_appropriate_character_ideology_from_overlord = yes
			}
		}
	}
	
}
