	activate_law = law_type:law_command_economy
	
	if = { 
		limit = { 
			game_date < 1951.1.1

		}
		activate_law = law_type:law_child_labor_allowed			#Compulsory Primary School
	}
	else_if = { 
		limit = { 
			game_date > 1951.1.1
