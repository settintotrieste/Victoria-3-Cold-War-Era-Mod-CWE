calculate_lower_house_composition = {

	if = {
		limit = { has_law_or_variant = law_type:law_propotional_representation }
		calculate_lower_house_composition_PR = yes
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_jefferson_system }
		calculate_lower_house_composition_jefferson = yes
	}
	else = { 
		calculate_lower_house_composition_FTP = yes 
	}

}
