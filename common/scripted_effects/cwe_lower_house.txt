calculate_lower_house_composition = {

	if = {
		limit = { has_law_or_variant = law_type:law_propotional_representation }
		calculate_lower_house_composition_PR = yes
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_jefferson_system }
		calculate_lower_house_composition_jefferson = yes
	}
	else = { 
		calculate_lower_house_composition_FTP = yes 
	}

}

calculate_lower_house_composition_FTP = {

	#Calculate pro-government MPs
	set_variable = {
		name = pro_government_MPs_var
		value = 0
	}
	
	#Add Pro-Government members
	if = { limit = { exists = ig:ig_industrialists } ig:ig_industrialists = { save_scope_as = ig_industrialists_scope_ig } }
	if = { limit = { exists = ig:ig_petty_bourgeoisie } ig:ig_petty_bourgeoisie = { save_scope_as = ig_petty_bourgeoisie_scope_ig } }
	if = { limit = { exists = ig:ig_trade_unions } ig:ig_trade_unions = { save_scope_as = ig_trade_unions_scope_ig } }
	if = { limit = { exists = ig:ig_landowners } ig:ig_landowners = { save_scope_as = ig_landowners_scope_ig } }
	if = { limit = { exists = ig:ig_devout } ig:ig_devout = { save_scope_as = ig_devout_scope_ig } }
	if = { limit = { exists = ig:ig_rural_folk } ig:ig_rural_folk = { save_scope_as = ig_rural_folk_scope_ig } }
	if = { limit = { exists = ig:ig_intelligentsia } ig:ig_intelligentsia = { save_scope_as = ig_intelligentsia_scope_ig } }
	if = { limit = { exists = ig:ig_armed_forces } ig:ig_armed_forces = { save_scope_as = ig_armed_forces_scope_ig } }
	
	#Calculate
	calculate_LH_pro_government_MPs_FTP = yes
	calculate_arrange_seats_LH = yes
	
	#Determine if a simple majority exists
	set_variable = {
		name = lower_house_majority_var
		value = root.var:pro_government_MPs_var
	}
	if = {
		limit = { root.var:lower_house_majority_var > 50 }
		set_variable = lower_house_has_simple_majority
	}
	if = {
		limit = { root.var:lower_house_majority_var > 66.7 }
		set_variable = lower_house_has_super_majority
	}
	if = {
		limit = { root.var:lower_house_majority_var <= 50 }
		remove_variable = lower_house_has_simple_majority
	}
	if = {
		limit = { root.var:lower_house_majority_var <= 66.7 }
		remove_variable = lower_house_has_super_majority
	}
	
	#For display only
	change_variable = { name = lower_house_010_var multiply = 100 }
	
	post_notification = lower_house_composition_updated

}
