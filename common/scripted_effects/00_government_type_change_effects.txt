change_to_parliamentary_elective = {
	clean_up_any_monarchs = yes
}

post_change_to_parliamentary_elective = {
	if = {
		limit = { is_foreign_governor_only_country = yes }
		trigger_event = { id = revolution_system.20 }
	}
}

change_to_hereditary = {

	#Presidents make themselves King
	if = {
		limit = { 
			OR = { 
				government_transfer_of_power = dictatorial 
				government_transfer_of_power = presidential_elective
			}
			is_a_colonial_subject_gov = no
			is_a_colonial_government = no
			shares_same_monarch = no
		}
		ruler ?= {
			set_character_as_monarch = yes
		}
	}
	
	#Prime ministers find a king
	else = {
		generate_monarch_for_country = yes
	}
	
}

post_change_to_hereditary = {
	if = {
		limit = {
			NOT = {
				any_scope_character = {
					has_role = character_role_monarch
				}
			}
		}
		generate_monarch_for_country = yes
	}
	else_if = {
		limit = {
			any_scope_character = {
				has_role = character_role_monarch
				is_ruler_of_own_country = no
			}
		}
		random_scope_character = {
			limit = {
				has_role = character_role_monarch
				is_ruler_of_own_country = no
			}
			set_character_as_monarch = yes
		}
	}
}

change_to_regency = {

	#Clean up
	ruler ?= {
		if = {
			limit = { has_role = character_role_monarch }
			remove_character_role = character_role_monarch
		}
	}
	heir ?= {
		if = {
			limit = { NOT = { has_role = character_role_monarch } }
			add_character_role = character_role_monarch
		}
	}

}

	#Clean up
