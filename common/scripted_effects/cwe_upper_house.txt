calculate_upper_house_composition = {

	if = {
		limit = { has_law_or_variant = law_type:law_propotional_representation }
		calculate_upper_house_composition_PR = yes
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_jefferson_system }
		calculate_upper_house_composition_jefferson = yes
	}
	else = { 
		calculate_upper_house_composition_FTP = yes 
	}

}

calculate_upper_house_composition_FTP = {

	set_variable = {
		name = upper_house_size_var
		value = 0
	}

		limit = { 
			is_incorporated = yes 
			NOT = { state_region = s:STATE_DISTRICT_OF_COLUMBIA }

	}
	
	#Calculate proportion
	if = {
		limit = { 
			has_variable = upper_house_size_var 
			root.var:upper_house_size_var > 0
		}
		set_variable = {
			name = upper_house_increment_FTP_upper_var
			value = 200
		}
		change_variable = {
			name = upper_house_increment_FTP_upper_var
			divide = root.var:upper_house_size_var
		}
		set_variable = {
			name = upper_house_increment_FTP_lower_var
			value = 100
		}
		change_variable = {
			name = upper_house_increment_FTP_lower_var

		}
	}

	#Calculate pro-government senators
	set_variable = {
		name = pro_government_senators_var
		value = 0
	}
	
	#1/3 of Upper House will always be pro-government if Legislature is appointed
	if = {
		limit = { has_law_or_variant = law_type:law_technocratic_legislature }

					limit = { has_variable = upper_house_size_var }
					change_variable = {
						name = upper_house_size_var
						add = 1
					}
				}
				if = {
					limit = { has_variable = pro_government_senators_var }

		limit = { 
			has_variable = upper_house_upper_limit_var
			has_variable = upper_house_lower_limit_var
		}

			limit = {

			change_variable = { name = upper_house_lower_limit_var multiply = 0.90 }

		}

	}

	}
	if = {
		limit = { root.var:upper_house_majority_var <= 50 }
		remove_variable = upper_house_has_simple_majority
	}

	}
	if = {
		limit = { root.var:upper_house_majority_var <= 66.7 }
		remove_variable = upper_house_has_super_majority
	}
	
	post_notification = upper_house_composition_updated
