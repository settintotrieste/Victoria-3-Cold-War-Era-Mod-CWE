enforce_disarm_country_effect = {

	activate_law = law_type:law_peasant_levies
	if = {
		limit = { has_law = law_type:law_offensive_war }
		activate_law = law_type:law_defensive_war
	}

}

enforce_disarm_wmd_country_effect = {

	#Enact denuclearisation law which will auto terminate any nuclear program
	activate_law = law_type:law_wmd_banned
	
	#THIS cannot station nukes on its soil for 10 years
	save_scope_as = target_country_indy_scope
	custom_tooltip = {
		text = demand_nuclear_weapons_removal_2.tt
		remove_variable = no_nuclear_weapons_agreement
		set_variable = { name = no_nuclear_weapons_agreement years = 10 }
	}
	
	#Enforce no WMD for 10 years
	set_variable = { name = no_WMD_agreement years = 10 }

}

opportunity_to_steal_nuclear_wmd_effect = {

	#THIS has the opportunity to take TARGET's nuclear weapons
	if = {
		limit = {
			NOT = { has_modifier = nuclear_capability }
			NOT = { has_law_or_variant = law_type:law_wmd_banned }
			scope:target_country = { 
				has_modifier = nuclear_capability
			}
		}
		scope:target_country = { set_variable = { name = TARGET_nuclear_weapons months = 1 } }
		trigger_event = { id = nuclear_warfare_system.8 }
	}
	else = {
		change_infamy = -10
	}

}
