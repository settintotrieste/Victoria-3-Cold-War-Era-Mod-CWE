release_vassal_TAN = { 
	random_scope_state = { 
		limit = { state_region = s:STATE_TANGANYIKA } 
		save_scope_as = released_vassals_capital 
	}
	create_country = { 
		tag = TAN 
		origin = ROOT
		state = scope:released_vassals_capital
		on_created = { set_release_vassal_effects = yes }
	}
	if = {
		limit = { exists = c:TAN }
		every_scope_state = { 
			limit = { is_homeland_of_country_cultures = c:TAN capable_of_release = yes } 
			set_state_owner = c:TAN 
		}
		create_diplomatic_pact = {
			country = c:TAN
			type = semi_autonomous_colony
		}
	}
}
