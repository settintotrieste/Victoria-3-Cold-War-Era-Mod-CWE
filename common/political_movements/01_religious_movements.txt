movement_religious_majority = {
	category = movement_category_religious

	ideology = ideology_agitator_theocrat
	character_ideologies = {
		ideology_autocratic
		ideology_power_hungry
	}

	creation_trigger = {
		NOT = {
			has_law_or_variant = law_type:law_state_atheism
		}
	}
	creation_weight = {
		value = 100
	}
	on_created = {}

	disband_trigger = {
		scope:political_movement ?= {
			religion = {
				NOT = { is_state_religion = root }
			}
		}
	}
	
	can_pressure_interest_group = {
		law_stance = {
			law = law_type:law_HOS_religious_leader
			value >= neutral
		}
	}

	religion_selection_trigger = {
		is_state_religion = scope:country
	}
	religion_selection_weight = {
		value = 100
	}

	character_support_trigger = {
		religion = scope:religion
		OR = {
			is_interest_group_type = ig_devout
			has_ideology = ideology:ideology_agitator_theocrat
			has_ideology = ideology:ideology_traditionalist
			has_ideology = ideology:ideology_agitator_anti_human_rights
		}
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				scope:interest_group ?= {
					is_interest_group_type = ig_devout
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_agitator_theocrat
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_autocratic
			}
			multiply = {
				value = 2
			}
		}
	}

	pop_support_trigger = {
		religion = scope:religion
	}

	pop_support_factors = {
		movement_support_low_urbanization
		movement_support_low_literacy
		movement_support_clergymen
		movement_support_aristocrats
		movement_support_peasants
		movement_support_corporatism_tech
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		add = {
			desc = "URBAN_STATE" # Increases support in rural states, decreases in urban
			value = 0.5
			subtract = state.state_urbanization_rate
			multiply = 10
		}

		add = {
			value = 10
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 45
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 15
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 10
				desc = "POP_PEASANTS"
			}
		}

		add = { # Weakened by increasing literacy rates
			desc = "POP_LITERACY"
			this = {
				value = literacy_rate
				multiply = -2
			}
		}

		if = { # Discriminated pops are less inclined to support the state church
			limit = {
				pop_acceptance < acceptance_status_5
			}
			multiply = {
				desc = "POP_ACCEPTANCE"
				value = pop_acceptance
				divide = 100
				add = 0.25
			}
		}
		
	}

	revolution = {
		weight = {
			value = 100
		}

		state_weight = {
			value = 1

			if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.75
					}
				}
				add = 75
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.5
					}
				}
				add = 50
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.25
					}
				}
				add = 25
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.1
					}
				}
				add = 10
			}
		}
	}

	secession = {
		possible = {
			always = no
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1
		}

		state_weight = {
			value = 0

			if = {
				limit = {
					NOT = { is_homeland_of_country_cultures = ROOT }
				}
				add = -50
			}

			if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.75
					}
				}
				add = 250
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.5
					}
				}
				add = 150
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.25
					}
				}
				add = 50
			}
			else_if = {
				limit = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.1
					}
				}
				add = 10
			}
		}
	}

	additional_radicalism_factors = {
		if = {
			limit = {
				owner.ruler ?= {
					NOT = { religion = root.religion }
				}
			}

			add = {
				value = 0.1
				desc = "RULER_WRONG_RELIGION"
			}
		}

		add = {
			value = "owner.average_sol"
			subtract = "owner.average_sol_for_religion(scope:religion)"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_RELIGION"
		}
	}
}

movement_religious_minority = {
	category = movement_category_religious

	ideology = ideology_agitator_theocrat
	character_ideologies = {
		ideology_autocratic
		ideology_power_hungry
		ideology_elitist
		ideology_democratic
		ideology_liberal
	}

	creation_trigger = {
		
	}
	creation_weight = {
		value = 200
	}
	on_created = {}

	disband_trigger = {
		scope:political_movement ?= {
			religion = {
				is_state_religion = root
			}
			root = {
				religion_percent_country = {
					target = scope:political_movement.religion
					value <= 0.08
				}
			}
		}
	}

	on_disbanded = {
		
	}

	can_pressure_interest_group = {
		law_stance = {
			law = law_type:law_HOS_religious_leader
			value >= neutral
		}
	}

	religion_selection_trigger = {
		NOT = { is_state_religion = scope:country }
		NOT = { rel:atheist = this }
		scope:country = {
			religion_percent_country = {
				target = root
				value >= 0.1
			}
		}
	}
	religion_selection_weight = {
		value = 100
	}

	character_support_trigger = {
		religion = scope:religion
		OR = {
			AND = {
				owner ?= {
					has_law_or_variant = law_type:law_state_atheism
				}
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_agitator_theocrat
				}
			}
			has_ideology = ideology:ideology_agitator_theocrat
			has_ideology = ideology:ideology_agitator_anti_human_rights
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				scope:interest_group ?= {
					is_interest_group_type = ig_devout
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						owner ?= {
							has_law_or_variant = law_type:law_state_atheism
						}
						OR = {
							has_ideology = ideology:ideology_traditionalist
							has_ideology = ideology:ideology_agitator_theocrat
						}
					}
					has_ideology = ideology:ideology_agitator_theocrat
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				NOT = {
					scope:interest_group ?= {
						is_interest_group_type = ig_devout
					}
				}
				OR = {
					has_ideology = ideology:ideology_progressive
					has_ideology = ideology:ideology_communist
				}
			}
			multiply = 0.05
		}
	}

	pop_support_trigger = {
		religion = scope:religion
	}

	pop_support_factors = {
		movement_support_poverty
		movement_support_low_literacy
		movement_support_religious_discrimination
		movement_support_clergymen
		movement_support_aristocrats
		movement_support_peasants
		movement_support_corporatism_tech
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			value = 10
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				standard_of_living < 15
			}

			add = {
				value =	standard_of_living
				subtract = 15
				multiply = -5
				desc = "POP_POVERTY"
			}
		}

		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 45
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 10
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 10
				desc = "POP_PEASANTS"
			}
		}

		add = { # Weakened by increasing literacy rates
			desc = "POP_LITERACY"
			this = {
				value = literacy_rate
				multiply = -2
			}
		}

		if = {
			limit = {
				state = {
					state_religious_acceptance = {
						target = PREV.religion
						value < religious_acceptance_high
					}
				}
			}
			multiply = {
				desc = "POP_RELIGIOUS_DISCRIMINATION"
				value = 100
				subtract = pop_acceptance
				divide = 25 # Maxes out at 4x at 0% acceptance, starts being a liability at 75% acceptance
			}
		}

		# Favor cultural movement if necessary
		if = {
			limit = {
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						this.culture ?= root.culture
					}
				}
			}
			if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_1 } }
				}
				multiply = 0.1
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_2 } }
				}
				multiply = 0.25
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_3 } }
				}
				multiply = 0.5
			}
		}
	}

	secession = {
		possible = {
			always = no
			owner ?= {
				any_scope_state = {
					religion_percent_state = {
						target = root.religion
						value >= 0.10
					}
				}
			}
		}

		weight = {
			value = 1
		}

		state_weight = {
			value = 0
			
			if = {
				limit = {
					NOT = { is_homeland_of_country_cultures = ROOT }
				}
				add = -50
			}

			if = {
				limit = {
					is_capital = no
				}

				if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.75
						}
					}
					add = 250
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.5
						}
					}
					add = 150
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.25
						}
					}
					add = 50
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.1
						}
					}
					add = 10
				}
			}
		}

		target_fraction_of_states = {
			value = political_movement_identity_support
			add = scope:clout
			multiply = 3
			min = 0.15
			max = 0.75
		}
	}

	additional_radicalism_factors = {
		add = {
			value = "owner.average_sol"
			subtract = "owner.average_sol_for_religion(scope:religion)"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_RELIGION"
		}

		subtract = {
			value = 0.5
			subtract = political_movement_identity_support
			min = 0
			desc = "LOW_MOVEMENT_IDENTITY_SUPPORT"
		}
	}
}
