##### country-specific and misc bloc names #####

SINOSPHERE = {
	trigger = {
		OR = {
			scope:selected_identity = identity:identity_sovereign_empire
			scope:selected_identity = identity:identity_military_treaty_organization
			scope:selected_identity = identity:identity_ideological_union
		}
		OR = {
			c:PRC ?= this
			c:ROC ?= this
			c:CHI ?= this
		}
	}
}

one_belt_one_road = {
	trigger = {
		scope:selected_identity = identity:identity_trade_league
		OR = {
			c:PRC ?= this
			c:ROC ?= this
			c:CHI ?= this
		}
	}
}

NON_ALIGNED_MOVEMENT = {
	trigger = {
		OR = {

			scope:selected_identity = identity:identity_ideological_union
		}
		has_modifier = non_aligned_movement_member
	}
}

### Overrides ###

pact_of_calcutta = {
	trigger = {
	    scope:selected_identity = identity:identity_ideological_union
	    OR = {
	    	any_scope_state = { state_region = s:STATE_WEST_BENGAL }
	    	any_scope_ally = {
	    		any_scope_state = { state_region = s:STATE_WEST_BENGAL }
	    	}
	    }
	}
}

league_of_india = {
	trigger = {
	    any_scope_state = {
	    	OR = {
				region = sr:region_south_india
				region = sr:region_north_india
			}
			is_capital = yes
	    }
	}
}

order_of_the_colossus = { # or simply the Colossal Order
	trigger = {
		any_scope_state = {
			state_region = s:STATE_UUSIMAA # Tampere
			is_incorporated = yes
		}
	}
}

unholy_alliance = {
	trigger = {
	    scope:selected_identity = identity:identity_military_treaty_organization

	    OR = {
	    	AND = {
	    		c:FRA ?= this
	    		any_scope_ally = { 
	    			c:GER ?= this 
	    			is_subject = no
	    		}
	    	}
	    	AND = {
	    		c:GER ?= this
	    		any_scope_ally = { 
	    			c:FRA ?= this 
	    			is_subject = no
	    		}
	    	}
	    }
	}
}

waterloo_pact = {
	trigger = {
		scope:selected_identity = identity:identity_military_treaty_organization
	    OR = {
	    	c:BEL ?= this
	    	any_scope_ally = {
	    		c:BEL ?= this
	    	}
	    	any_scope_ally = {
	    		any_scope_state = { state_region = s:STATE_WALLONIA }
	    	}
	    }
		any_rival_country = {
			c:FRA ?= this
		}
	}
}

strasbourg_congress = {
	trigger = {
		c:FRA ?= this
	    any_scope_state = { 
		    state_region = s:STATE_ALSACE_LORRAINE
	    	is_incorporated = yes
	    }
	}
}

				c:FRA ?= this
				c:ITA ?= this
				c:SPA ?= this
				c:POR ?= this

			}
		}
	}
}

cross_channel_alliance = {
	trigger = {
		OR = {
			c:GBR ?= this
			c:FRA ?= this
		}
		any_scope_ally = {
			OR = {
				c:GBR ?= this
				c:FRA ?= this
			}
		}
	}
}

communard_league = {
	trigger = {
		c:FRA ?= this
		coa_communist_trigger = yes
	}
}

entente = {
	trigger = {
	    scope:selected_identity = identity:identity_military_treaty_organization
	    c:FRA ?= this
	    any_scope_ally = {
	    	count >= 2
	    	OR = {
		    	c:GBR ?= this
		    	c:RUS ?= this
		    	c:ITA ?= this
		    	c:USA ?= this
		    	c:JAP ?= this
	    	}
	    }
	}
}
