namespace = day_1_events

day_1_events.1 = {
	type = country_event
	hidden = yes
	immediate = {
		
		#Set correct literacy
		force_set_literacy_rate = yes
		
		#Set correct starting AI strategy
		every_country = {
			limit = {
				OR = { 
					AND = { 
						literacy_rate >= 0.30 
						OR = { 
							has_law_or_variant = law_type:law_manufacturing_focus
							has_law_or_variant = law_type:law_no_economic_specialisation
						}
					}
					c:WGR ?= this
					c:EGR ?= this
					c:JAP ?= this
					c:USA ?= this
					c:GBR ?= this
					c:ITA ?= this
					c:SOV ?= this
				}
			}
			set_strategy = ai_strategy_high_tech_expansion
		}
		
		#Initialise parliaments
		every_country = {
			if = {
				limit = { has_journal_entry = je_upper_house }
				set_variable = election_held_recently_UH_var
				calculate_upper_house_composition = yes
			}
			if = {
				limit = { has_journal_entry = je_lower_house }
				set_variable = election_held_recently_LH_var
				calculate_lower_house_composition = yes
			}
		}
		
	}
}
