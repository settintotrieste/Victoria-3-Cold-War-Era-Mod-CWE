namespace = politicians_system

#Randomly retire rulers
politicians_system.1 = {
	type = country_event
	hidden = yes

	trigger = {
		
		#Not at war
		is_at_war = no
		
		#Has an elderly ruler, politician or agitator
		OR = {
			
			#Ruler (dictators retire by dying)
			AND = {
				NOT = { is_a_dictatorship = yes }
				ruler ?= {
					NOT = { has_variable = is_ruler_for_life }
					NOT = { has_modifier = monarch_character }
					OR = { 
						AND = { 
							is_historical = yes
							age >= 90
						}
						AND = { 
							is_historical = no
							age >= 80
						}
						AND = {
							age >= 70
							OR = {
								has_trait = psychological_affliction
								has_trait = cancer
								has_trait = senile
								has_trait = wounded
							}
							is_historical = no
						}
					}
				}
			}
			
			#IG Leaders & Agitators
			any_scope_character = {
				is_ruler = no
				NOT = { has_modifier = monarch_character }
				OR = { 
					has_role = politician
					has_role = agitator
				}
				OR = { 
					AND = { 
						is_historical = yes
						age >= 90
					}
					AND = { 
						is_historical = no
						age >= 80
					}
					AND = {
						age >= 70
						OR = {
							has_trait = psychological_affliction
							has_trait = cancer
							has_trait = senile
							has_trait = wounded
						}
						is_historical = no
					}
				}
			}
		
		}
		
	}

	immediate = {
		
		#Retire the elderly
		random_scope_character = {
		
			limit = {
			
				OR = {
					
					#Rulers
					AND = {
						is_ruler = yes
						NOT = { has_variable = is_ruler_for_life }
						NOT = { has_modifier = monarch_character }
						owner = { NOT = { is_a_dictatorship = yes } }
						OR = { 
							AND = { 
								is_historical = yes
								age >= 90
							}
							AND = { 
								is_historical = no
								age >= 80
							}
							AND = {
								age >= 70
								OR = {
									has_trait = psychological_affliction
									has_trait = cancer
									has_trait = senile
									has_trait = wounded
								}
								is_historical = no
							}
						}
					}
					
					#Politicians & Agitators
					AND = { 
						is_ruler = no
						NOT = { has_modifier = monarch_character }
						OR = { 
							has_role = politician
							has_role = agitator
						}
						OR = { 
							AND = { 
								is_historical = yes
								age >= 90
							}
							AND = { 
								is_historical = no
								age >= 80
							}
							AND = {
								age >= 70
								OR = {
									has_trait = psychological_affliction
									has_trait = cancer
									has_trait = senile
									has_trait = wounded
								}
								is_historical = no
							}
						}
					}
					
				}
				
			}
			
			save_scope_as = retiring_politician_scope
			
		}
		
		if = {
			limit = { exists = scope:retiring_politician_scope }
			
			#Notify player of retirement
			post_notification = ig_leader_has_retired
			
			#Retire character
			scope:retiring_politician_scope ?= {
				kill_character = {
					hidden = yes
					value = yes
				} 
			}
		}
		
	}

}

#Auto select rulers for parliamentary elective
politicians_system.3 = {
	type = country_event
	hidden = yes

	trigger = {
		
		#Bypass Parliamentary Elective System
		is_a_parliamentary_elective_system = yes
		
		#Ruler's IG is not a government IG
		ruler ?= { 
			interest_group ?= { 
				is_in_government = no 
			} 
		}
		
		#There are multiple gov IGs to select a ruler from
		any_interest_group = {
			is_in_government = yes
			count < 2
		}
		
	}

	immediate = {
		
		#Appoint ruler
		random_scope_character = {
			limit = { 
				has_role = politician
				interest_group = {
					is_in_government = yes
					AND = { is_in_government = yes ig_government_power_share > 0.5 } 
				}
			}
			set_character_as_ruler = yes
		}
		
	}

}
