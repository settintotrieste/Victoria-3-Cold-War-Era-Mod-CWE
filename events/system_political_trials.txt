namespace = political_trials_system

#Accused is convicted
political_trials_system.1 = {
    type = country_event

    title = won_a_political_trials.t 					# Title of the event; ADD LOCALISATION
    desc = won_a_political_trials.d 					# Description of the event; ADD LOCALISATION
    flavor = won_a_political_trials.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "middleeast_courtroom_upheaval" }

    icon = "gfx/interface/icons/event_icons/event_scales.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {											# What triggers the event, if using other events or journal entries leave empty
		
		#Triggered by JE
		
    }
	
	immediate = {
		
		random_scope_character = {
			limit = { has_variable = convicted_character_flag }
			remove_variable = convicted_character_flag
			
			save_scope_as = convicted_character_scope
			interest_group = { save_scope_as = convicted_character_ig_scope }
		}
		
	}
	
	cancellation_trigger = {
		NOT = { exists = scope:convicted_character_scope }
	}
    
    option = {
        name = won_a_political_trials.option.1					# Imprison character
		default_option = yes
		
		#Imprison Character
		custom_tooltip = {
			text = imprison_character_for_10_years.tt
		}
		if = {
			limit = { 
				has_law_or_variant = law_type:law_slavery_political 
			}
			scope:convicted_character_scope ?= {
				if = {
					limit = { NOT = { has_trait = sickly } }
					add_trait = sickly
				}
			}
		}
		
		#Domestic reaction
		if = {
			limit = {
				scope:convicted_character_scope ?= { 
					OR = {
						
						## Character has committed a crime ##
						
						#Corruption scandal
						has_modifier = ignored_corruption
						has_modifier = corrupt_connections_exposed
						has_modifier = corrupt_connections_exposed_minor
						has_modifier = corruption_scandal_character_modifier
						has_modifier = dealing_with_corruption
						has_trait = grifter
						has_modifier = framed_for_crimes
						
						#Traitor
						has_modifier = treasonous_traitor
						
						#Abuse of Power
						has_modifier = seizing_power
						has_trait = war_criminal
						
					}
				}
			}
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = small_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
		}
		else_if = {
			limit = {
				scope:convicted_character_scope ?= { 
					OR = {
						
						# Personal Misconduct
						AND = {
							OR = {
								has_trait = cocaine_addiction
								has_trait = opium_addiction
							}
							owner ?= {
								OR = {
									has_law_or_variant = law_type:law_no_minority_issues
									has_law_or_variant = law_type:law_some_minority_issues
								}
							}
						}
						AND = {
							OR = { 
								has_trait = alcoholic
								has_trait = hedonist
							}
							owner ?= { 
								has_law_or_variant = law_type:law_no_minority_issues
							}
						}
						AND = {
							has_trait = homosexual_character
							owner ?= { has_law_or_variant = law_type:law_no_gender_issues }
						}
						
					}
				}
			}
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = medium_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
			create_incident = {
				value = 1
				country = ROOT
			}
		}
		else = {
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = large_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
			create_incident = {
				value = 3
				country = ROOT
			}
		}
		
		#Rehabilitate Character
		scope:convicted_character_scope ?= {
			effect_rehabilitate_convicted_character = yes
		}
		
		ai_chance = {
			base = 75
		}
    }
	
	option = {
        name = won_a_political_trials.option.2					# Execute Character
		trigger = {
			is_a_dictatorship = yes
			has_law_or_variant = law_type:law_draconian_policing
		}
		
		#Execute character
		if = {
			limit = { exists = scope:convicted_character_scope }
			scope:convicted_character_scope ?= {
				kill_character = yes
			}
		}
		
		#Domestic reaction
		if = {
			limit = { exists = scope:convicted_character_ig_scope }
			add_radicals = {
				value = very_large_radicals
				interest_group = scope:convicted_character_ig_scope
			}
		}
		
		#International reaction
		create_incident = {
			value = 5
			country = ROOT
		}
		
		ai_chance = {
			base = 0
			modifier = { 
				trigger = {
					ruler ?= {
						has_ideology = ideology:ideology_autocratic
					}
				}
				add = 25
			}
		}
    }
	
	option = {
        name = won_a_political_trials.option.3					# Release
		trigger = {
			scope:convicted_character_scope ?= {
				interest_group = { interest_group = { is_same_interest_group_type = root.ruler.interest_group } }
			}
		}
		
		#Release character
		if = {
			limit = { exists = scope:convicted_character_scope }
			scope:convicted_character_scope ?= {
				replace_character_roles = { add = { character_role_politician } remove_all = yes }
				remove_modifier = convicted_character
				save_scope_as = released_prisoner
				post_notification = political_prisoner_released
			}
		}
		
		#Government reaction
		if = {
			limit = { 
				is_a_dictatorship = yes 
				scope:convicted_character_scope ?= { interest_group = { NOT = { is_same_interest_group_type = root.ruler.interest_group } } }
			}
			add_radicals = {
				value = medium_radicals
				interest_group = ruler.interest_group
			}
		}
		
		#Rehabilitate Character
		scope:convicted_character_scope ?= {
			effect_rehabilitate_convicted_character = yes
		}
		
		ai_chance = {
			base = 0
			modifier = { 
				trigger = {
					ruler ?= {
						has_ideology = ideology:ideology_autocratic
					}
				}
				add = 25
			}
		}
    }

}

    event_image = { texture = "middleeast_courtroom_upheaval" }

    icon = "gfx/interface/icons/event_icons/event_scales.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {											# What triggers the event, if using other events or journal entries leave empty
		
		#Triggered by JE
		
    }
	
	immediate = {
		
		random_scope_character = {
			limit = { has_variable = acquitted_character_flag }
			remove_variable = acquitted_character_flag
			
			save_scope_as = convicted_character_scope
			interest_group = { save_scope_as = convicted_character_ig_scope }
		}
		
	}
	
	cancellation_trigger = {
		NOT = { exists = scope:convicted_character_scope }
	}
    
    option = {
        name = lost_a_political_trials.option.1					# Let him be free
		default_option = yes
		
		#Domestic effect
		if = {
			limit = { scope:convicted_character_scope ?= { is_ruler_of_own_country = no } }
			ruler ?= {
				hidden_effect = { remove_modifier = embarassed_by_witch_hunt }
				add_modifier = { name = embarassed_by_witch_hunt years = 5 is_decaying = yes }
			}
		}
		if = {
			limit = { exists = scope:convicted_character_ig_scope }
			add_radicals = {
				value = small_radicals
				interest_group = scope:convicted_character_ig_scope
			}
		}
		
    }

}

#Character is under investigation
political_trials_system.3 = {
    type = country_event

    title = investigation_into_character_start.t 					# Title of the event; ADD LOCALISATION
    desc = investigation_into_character_start.d 					# Description of the event; ADD LOCALISATION
    flavor = investigation_into_character_start.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "middleeast_police_breaking_door" }

    icon = "gfx/interface/icons/event_icons/event_scales.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {														# What triggers the event, if using other events or journal entries leave empty
		
		#No investigation ongoing
		NOT = { has_journal_entry = je_political_investigations_system }
		
		#There is a character which is eligible to be investigated
		any_scope_character = {
			
			#Eligible characters only
			NAND = {
				interest_group ?= { is_in_government = yes }
				owner ?= { has_law_or_variant = law_type:law_politicised_bureaucrats }
			}
			character_has_investigation_immunity = no
			
			#Eligibility
			NOT = { has_modifier = accused_of_crimes }
			NOT = { has_variable = investigation_found_crimes }
			NOT = { has_variable = acquitted_of_investigation }
			
			NOT = { has_modifier = indicted_character }
			NOT = { has_modifier = convicted_character }
			NOT = { has_role = character_role_prisoner }
			NOT = { has_modifier = acquitted_character }
			
			#Has cause
			OR = {
				
				#Corruption scandal
				has_modifier = ignored_corruption
				has_modifier = corrupt_connections_exposed
				has_modifier = corrupt_connections_exposed_minor
				has_modifier = corruption_scandal_character_modifier
				has_modifier = dealing_with_corruption
				has_trait = grifter
				has_modifier = framed_for_crimes
				
				#Traitor
				has_modifier = treasonous_traitor
				
				#Abuse of Power
				has_modifier = seizing_power
				has_trait = war_criminal
				
			}
			
		}
		
    }
	
	immediate = {
		
		random_scope_character = {
			limit = {
				
				#Government affiliated characters only
				NAND = { 
					interest_group ?= { is_in_government = yes }
					owner ?= { has_law_or_variant = law_type:law_politicised_bureaucrats }
				}
				character_has_investigation_immunity = no
				
				#Eligibility
				NOT = { has_modifier = accused_of_crimes }
				NOT = { has_variable = investigation_found_crimes }
				NOT = { has_variable = acquitted_of_investigation }
				
				NOT = { has_modifier = indicted_character }
				NOT = { has_modifier = convicted_character }
				NOT = { has_role = character_role_prisoner }
				NOT = { has_modifier = acquitted_character }
				
				#Has cause
				OR = {
					
					#Corruption scandal
					has_modifier = ignored_corruption
					has_modifier = corrupt_connections_exposed
					has_modifier = corrupt_connections_exposed_minor
					has_modifier = corruption_scandal_character_modifier
					has_modifier = dealing_with_corruption
					has_trait = grifter
					has_modifier = framed_for_crimes
					
					#Traitor
					has_modifier = treasonous_traitor
					
					#Abuse of Power
					has_modifier = seizing_power
					has_trait = war_criminal
					
				}
				
			}
			save_scope_as = investigated_character_scope
		}
		
	}
	
	cancellation_trigger = {
		NOT = { exists = scope:investigated_character_scope }
	}
    
    option = {
        name = investigation_into_character_start.option.1			# Oh no
		default_option = yes
		
		#Begin investigation
		scope:investigated_character_scope ?= {
			add_modifier = {
				name = accused_of_crimes
				years = 6
			}
			owner ?= { hidden_effect = { add_journal_entry = { type = je_political_investigations_system } } }
		}
		
    }

}
