namespace = political_trials_system

#Accused is convicted
political_trials_system.1 = {
    type = country_event

    title = won_a_political_trials.t 					# Title of the event; ADD LOCALISATION
    desc = won_a_political_trials.d 					# Description of the event; ADD LOCALISATION
    flavor = won_a_political_trials.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "middleeast_courtroom_upheaval" }

    icon = "gfx/interface/icons/event_icons/event_scales.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {											# What triggers the event, if using other events or journal entries leave empty
		
		#Triggered by JE
		
    }
	
	immediate = {
		
		random_scope_character = {
			limit = { has_variable = convicted_character_flag }
			remove_variable = convicted_character_flag
			
			save_scope_as = convicted_character_scope
			interest_group = { save_scope_as = convicted_character_ig_scope }
		}
		
	}
	
	cancellation_trigger = {
		NOT = { exists = scope:convicted_character_scope }
	}
    
    option = {
        name = won_a_political_trials.option.1					# Imprison character
		default_option = yes
		
		#Imprison Character
		custom_tooltip = {
			text = imprison_character_for_10_years.tt
		}
		if = {
			limit = { 
				has_law_or_variant = law_type:law_slavery_political 
			}
			scope:convicted_character_scope ?= {
				if = {
					limit = { NOT = { has_trait = sickly } }
					add_trait = sickly
				}
			}
		}
		
		#Domestic reaction
		if = {
			limit = {
				scope:convicted_character_scope ?= { 
					OR = {
						
						## Character has committed a crime ##
						
						#Corruption scandal
						has_modifier = ignored_corruption
						has_modifier = corrupt_connections_exposed
						has_modifier = corrupt_connections_exposed_minor
						has_modifier = corruption_scandal_character_modifier
						has_modifier = dealing_with_corruption
						has_trait = grifter
						has_trait = framed_for_crimes
						
						#Traitor
						has_modifier = treasonous_traitor
						
						#Abuse of Power
						has_modifier = seizing_power
						has_trait = war_criminal
						
					}
				}
			}
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = small_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
		}
		else_if = {
			limit = {
				scope:convicted_character_scope ?= { 
					OR = {
						
						# Personal Misconduct
						AND = {
							OR = {
								has_trait = cocaine_addiction
								has_trait = opium_addiction
							}
							owner ?= {
								OR = {
									has_law_or_variant = law_type:law_no_minority_issues
									has_law_or_variant = law_type:law_some_minority_issues
								}
							}
						}
						AND = {
							OR = { 
								has_trait = alcoholic
								has_trait = hedonist
							}
							owner ?= { 
								has_law_or_variant = law_type:law_no_minority_issues
							}
						}
						AND = {
							has_trait = homosexual_character
							owner ?= { has_law_or_variant = law_type:law_no_gender_issues }
						}
						
					}
				}
			}
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = medium_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
			create_incident = {
				value = 1
				country = ROOT
			}
		}
		else = {
			if = {
				limit = { exists = scope:convicted_character_ig_scope }
				add_radicals = {
					value = large_radicals
					interest_group = scope:convicted_character_ig_scope
				}
			}
			create_incident = {
				value = 3
				country = ROOT
			}
		}
		
		ai_chance = {
			base = 75
		}
    }
	
	option = {
        name = won_a_political_trials.option.2					# Execute Character
		trigger = {
			is_a_dictatorship = yes
			has_law_or_variant = law_type:law_draconian_policing
		}
		
		#Execute character
		if = {
			limit = { exists = scope:convicted_character_scope }
			scope:convicted_character_scope ?= {
				kill_character = yes
			}
		}
		
		#Domestic reaction
		if = {
			limit = { exists = scope:convicted_character_ig_scope }
			add_radicals = {
				value = very_large_radicals
				interest_group = scope:convicted_character_ig_scope
			}
		}
		
		#International reaction
		create_incident = {
			value = 5
			country = ROOT
		}
		
		ai_chance = {
			base = 0
			modifier = { 
				trigger = {
					ruler ?= {
						has_ideology = ideology:ideology_autocratic
					}
				}
				add = 25
			}
		}
    }
	
	option = {
        name = won_a_political_trials.option.3					# Release
		trigger = {
			scope:convicted_character_scope ?= {
				interest_group = { interest_group = { is_same_interest_group_type = root.ruler.interest_group } }
			}
		}
		
		#Release character
		if = {
			limit = { exists = scope:convicted_character_scope }
			scope:convicted_character_scope ?= {
				replace_character_roles = { add = { character_role_politician } remove_all = yes }
				remove_modifier = convicted_character
				save_scope_as = released_prisoner
				post_notification = political_prisoner_released
			}
		}
		
		#Government reaction
		if = {
			limit = { 
				is_a_dictatorship = yes 
				scope:convicted_character_scope ?= { interest_group = { NOT = { is_same_interest_group_type = root.ruler.interest_group } } }
			}
			add_radicals = {
				value = medium_radicals
				interest_group = ruler.interest_group
			}
		}
		
		ai_chance = {
			base = 0
			modifier = { 
				trigger = {
					ruler ?= {
						has_ideology = ideology:ideology_autocratic
					}
				}
				add = 25
			}
		}
    }

}

		#Domestic effect

			hidden_effect = { remove_modifier = embarassed_by_witch_hunt }

		}
		if = {
			limit = { exists = scope:convicted_character_ig_scope }
			add_radicals = {
				value = small_radicals
				interest_group = scope:convicted_character_ig_scope
			}
		}
		
    }

}

#Assassination of Character - Retired
political_trials_system.3 = {
    type = country_event

    title = {
		first_valid = {
			triggered_desc = {
				desc = character_assassination.YES.t
				trigger = { has_variable = character_assassinated_YES }
			}
			triggered_desc = {
				desc = character_assassination.NO.t
				trigger = { NOT = { has_variable = character_assassinated_YES } }
			}
		}
	}
    desc = {
		first_valid = {
			triggered_desc = {
				desc = character_assassination.YES.d
				trigger = { has_variable = character_assassinated_YES }
			}
			triggered_desc = {
				desc = character_assassination.NO.d
				trigger = { NOT = { has_variable = character_assassinated_YES } }
			}
		}
	}
    flavor = character_assassination.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "southamerica_public_figure_assassination" }

    icon = "gfx/interface/icons/event_icons/event_skull.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {											# What triggers the event, if using other events or journal entries leave empty
		
		#Triggered by JE
		
    }
	
	immediate = {
		
		#Reset flags
		remove_variable = character_assassinated_YES
		
		random_scope_character = {
			limit = { has_variable = state_assassination_target }
			remove_variable = state_assassination_target
			
			save_scope_as = assassination_character_scope
			interest_group = { save_scope_as = assassination_character_ig_scope }
		}
		
		#Chance for assassination
		random_list = {

			10 = {
				
			}

			1 = {

				#Character Assassinated
				set_variable = character_assassinated_YES
				
				#More likely to have a successful assassination if Secret Police is strong
				modifier = {
					if = { limit = { institution_investment_level = { institution = institution_home_affairs value >= 1 } } }
					add = 1
				}
				modifier = {
					if = { limit = { institution_investment_level = { institution = institution_home_affairs value >= 2 } } }
					add = 2
				}
				modifier = {
					if = { limit = { institution_investment_level = { institution = institution_home_affairs value >= 3 } } }
					add = 3
				}
				modifier = {
					if = { limit = { institution_investment_level = { institution = institution_home_affairs value >= 4 } } }
					add = 4
				}
				modifier = {
					if = { limit = { institution_investment_level = { institution = institution_home_affairs value >= 5 } } }
					add = 5
				}
				
			}
			
		}
		
	}
	
	cancellation_trigger = {
		NOT = { exists = scope:assassination_character_scope }
	}
    
    option = {
        name = character_assassination.option
		default_option = yes
		
		#Reset flags
		remove_variable = character_assassinated_YES
		
		#Assassinate Character
		if = {
			limit = { has_variable = character_assassinated_YES }
			scope:assassination_character_scope = {
				kill_character = yes
			}
			add_radicals = {
				value = very_large_radicals
				interest_group = scope:assassination_character_ig_scope
			}
		}
		else = {
		
			#Character has a chance of exile
			random_list = {
				1 = {
					scope:assassination_character_scope = {
						exile_character_with_martyrdom = yes
					}
				}
				1 = {
					
				}
			}
			
			add_radicals = {
				value = large_radicals
				interest_group = scope:assassination_character_ig_scope
			}
		
		}
		
		#Infamy
		change_infamy = 5
		
    }

}
