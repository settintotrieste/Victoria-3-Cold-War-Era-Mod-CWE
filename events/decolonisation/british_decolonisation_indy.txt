#Canada
british_decolonisation_indy.1 = {
    type = country_event

    title = british_canada.indy.t 					# Title of the event; ADD LOCALISATION
    desc = british_canada.indy.d 					# Description of the event; ADD LOCALISATION
    #flavor = british_canada.indy.f 				# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		exists = c:GBR this = c:GBR											# Change TAG of the overlord ****
		game_date >= 1982.4.17									# START DATE: Date colony became self governing ****
		game_date < 1992.4.17									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_democratic_monarchy = yes							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#Colony conditions
		exists = c:CAN
		c:CAN = { 												# Change TAG of the colony ****
			is_direct_subject_of = ROOT
			is_subject_type = subject_type_dominion
		}
		
		# Event has not been fired before
		NOT = { has_variable = british_canada_indy }			# CHANGE ****
		
    }
    
    option = {
        name = british_decolonisation_indy.option.1				# Grant them independence
        highlighted_option = yes
		
		# Colony becomes independent
		c:CAN = { 												# Change TAG of the colony ****
			make_independent = yes
		}
		
		# Overlord create client state pact with colony
		create_diplomatic_pact = {
			country = c:CAN										# Change TAG ****
			type = personal_union
		}
		
		# Fire only once
		set_variable = british_canada_indy						# Change variable ****
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = british_decolonisation_indy.option.2					# REFUSE
		default_option = yes
		
		c:CAN = {
		
			# Natives are angry!
			every_scope_state = {
				add_radicals_in_state = {
					value = large_radicals
				}
				add_modifier = { name = nationalist_agitation months = 60 is_decaying = yes } 
			}
			
			#Colony is angry
			change_relations = {
				country = ROOT
				value = -100
			}
		}
		
		# Fire only once
		set_variable = british_canada_indy						# Change variable ****
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }

}

#Cyprus
british_decolonisation_indy.4 = {
    type = country_event

    title = british_cyprus.indy.t 					# Title of the event; ADD LOCALISATION
    desc = british_cyprus.indy.d 					# Description of the event; ADD LOCALISATION
    #flavor = british_cyprus.indy.f 				# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		exists = c:GBR this = c:GBR											# Change TAG of the overlord ****
		game_date >= 1960.8.16									# START DATE: Date colony became self governing ****
		game_date < 1970.8.16									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_democratic_monarchy = yes							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#Colony conditions
		NOR = { 												# Colony to be released must not already exist on the map
			exists = c:CYP										# Change TAG of the colony ****
			exists = c:WCY										# Change TAG of the colony ****
		}
		any_scope_state = { 									# Overlord owns the states of the colony to be released
			state_region = { is_homeland = cu:cypriot }			# Change homeland identity to match colony to be released ****
			#is_incorporated = no								# State is a colony
			#owner = THIS										# Overlord must own the state to be released
		}
		
		# Event has not been fired before
		NOT = { has_variable = british_cyprus_indy }			# CHANGE ****
		
    }
	
	immediate = {
	
		# Generate list of provinces to be released				# DO NOT TOUCH THIS SECTION
		every_scope_state = {
			limit = {
				state_region = { is_homeland = cu:cypriot }	# Change homeland identity to match colony to be released ****
				#is_incorporated = no							# State is a colony
				#contains_capital_of = c:CYP					# State is the capital of colony ****
			}
			save_scope_as = colonial_state
		}
		
	}
    
    option = {
        name = british_decolonisation.option.1				#Grant them self-government
        default_option = yes highlighted_option = yes
		
		#Release country
		create_country = {
			tag = WCY										# Change TAG of country being released ****
			origin = THIS									# TAG where the created country takes tech from
			state = scope:colonial_state
			
			on_created = {									# Set government laws & rulers here
				
				# SET laws for colony ****
				effect_starting_politics_presidential_republic = yes
				effect_starting_politics_regime = yes
				effect_starting_politics_constitutional_unitary_state = yes
				effect_starting_politics_war_policy_anti_military = yes
				effect_starting_politics_economic_policy_interventionism = yes
				effect_starting_politics_trade_policy_protectionism = yes
				effect_starting_politics_anti_nuclear_policy = yes
				
				# Set diplomatic recognition
				set_country_type = unrecognized
				
				# CREATE leader for colony ****
				create_character = {
					first_name = "Michael_Mouskos_first_name"
					last_name = "Michael_Mouskos_last_name"
					ruler = yes ig_leader = yes historical = yes
					birth_date = 1913.8.13
					interest_group = ig_devout
					ideology = ideology_elitist
					culture = cu:cypriot
					religion = rel:orthodox
					traits = {
						charismatic
						tactful
						pious
					}
				}
				
			}
		}
		
		if = {
			limit = { exists = c:WCY }
			every_scope_state = {
				limit = {
					state_region = { is_homeland = cu:cypriot }		# Change homeland identity to match colony to be released ****
					#is_incorporated = no							# State is a colony
				}
				set_state_owner = c:WCY								# Change TAG to colony ****
			}
			
			# Create Sovereign Bases
			create_diplomatic_pact = {
				country = c:WCY										# Change TAG ****
				type = military_cooperation_agreement
			}
			set_variable = sovereign_military_bases_CYP
			c:WCY = { set_variable = sovereign_military_bases_CYP }
		}
		
		# Fire only once
		set_variable = british_cyprus_indy						# Change variable ****
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = british_decolonisation.option.2					# REFUSE
		
		# Natives are angry!
		every_scope_state = {
			limit = {
		 		state_region = { is_homeland = cu:cypriot }	# Change homeland identity to match colony to be released ****
		 	}
			add_radicals_in_state = {
				culture = cu:cypriot							# Change culture ****
				value = large_radicals
			}
			add_modifier = { name = nationalist_agitation months = 60 is_decaying = yes } 
		}
		
		# Fire only once
		set_variable = british_cyprus_indy						# Change variable ****
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }

}
