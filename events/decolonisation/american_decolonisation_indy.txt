american_decolonisation_indy.1 = {
    type = country_event

    title = american_west_germany.indy.t 					# Title of the event; ADD LOCALISATION
    desc = american_west_germany.indy.d 					# Description of the event; ADD LOCALISATION
    #flavor = american_west_germany.indy.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		exists = c:USA this = c:USA											# Change TAG of the overlord ****
		game_date >= 1955.5.5									# START DATE: Date colony became self governing ****
		game_date < 1965.5.5									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_republican_democracy = yes							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#Colony conditions
		exists = c:WGR
		c:WGR = { 												# Change TAG of the colony ****
			is_direct_subject_of = ROOT
			is_not_sufficiently_independent = yes
		}
		
		# Event has not been fired before
		NOT = { has_variable = american_west_germany_indy }			# CHANGE ****
		
    }
	
	immediate = { 
		
		# Fire only once
		set_variable = american_west_germany_indy						# Change variable ****
		
	}
    
    option = {
        name = american_decolonisation_indy.option.1				# Grant them independence
        highlighted_option = yes
		
		# Colony becomes independent
		c:WGR = { 												# Change TAG of the colony ****
			make_independent = yes
			activate_law = law_type:law_defensive_war
			activate_law = law_type:law_national_militia
		}
		
		# Overlord create client state pact with colony
		create_diplomatic_pact = {
			country = c:WGR										# Change TAG ****
			type = satellite
		}
		
		# Fire only once
		set_variable = american_west_germany_indy						# Change variable ****
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = american_decolonisation_indy.option.2					# REFUSE
		default_option = yes
		
		c:WGR = {
		
			# Natives are angry!
			every_scope_state = {
				add_radicals_in_state = {
					value = large_radicals
				}
				add_modifier = { name = nationalist_agitation months = 60 is_decaying = yes } 
			}
			
			#Colony is angry
			change_relations = {
				country = ROOT
				value = -100
			}
		}
		
		# Add Infamy because doing this is against international law
		change_infamy = 10
		
		# Fire only once
		set_variable = american_west_germany_indy						# Change variable ****
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }

}

		#Colony conditions
		OR = { 
			AND = { 
				exists = c:WGR
				c:WGR ?= { 											# Change TAG of the colony ****
					is_direct_subject_of = ROOT
					is_subject_type = subject_type_satellite
				}
			}
			AND = { 
				exists = c:GER
				c:GER ?= { 											# Change TAG of the colony ****
					is_direct_subject_of = ROOT
					is_subject_type = subject_type_satellite
				}
			}
			AND = {
				exists = c:JAP
				c:JAP ?= { 											# Change TAG of the colony ****
					is_direct_subject_of = ROOT
					is_subject_type = subject_type_satellite
				}
			}
		}

    }
	
	immediate = { 
		
		# Fire only once

	}
    
    option = {
        name = historical_gov_changes.option.1						# Grant them independence
        highlighted_option = yes
		
		# Open Markets
		if = { 														# Change TAG of the colony ****
			limit = { exists = c:WGR c:WGR = { is_direct_subject_of = ROOT } }
			c:WGR = {
				save_scope_as = other_party
				create_treaty = {
					first_country = ROOT
					second_country = scope:other_party

					is_draft = no
					binding_period = { years = 10 } # 10 years

					articles_to_create = {
						{
							article = trade_privilege
							source_country = scope:other_party
							target_country = ROOT
						}
						{
							article = foreign_investment_rights
							source_country = ROOT
							target_country = scope:other_party
						}
					}
				}
				effect_starting_politics_fiat_standard = yes
			}
		}
		if = { 														# Change TAG of the colony ****
			limit = { exists = c:GER c:GER = { is_direct_subject_of = ROOT } }
			c:GER = {
				save_scope_as = other_party
				create_treaty = {
					first_country = ROOT
					second_country = scope:other_party

					is_draft = no
					binding_period = { years = 10 } # 10 years

					articles_to_create = {
						{
							article = trade_privilege
							source_country = scope:other_party
							target_country = ROOT
						}
						{
							article = foreign_investment_rights
							source_country = ROOT
							target_country = scope:other_party
						}
					}
				}
				effect_starting_politics_fiat_standard = yes
			}
		}
		
		# Open Markets
		if = { 														# Change TAG of the colony ****
			limit = { exists = c:JAP c:JAP = { is_direct_subject_of = ROOT } }
			c:JAP = {
				save_scope_as = other_party
				create_treaty = {
					first_country = ROOT
					second_country = scope:other_party

					is_draft = no
					binding_period = { years = 10 } # 10 years

					articles_to_create = {
						{
							article = trade_privilege
							source_country = scope:other_party
							target_country = ROOT
						}
						{
							article = foreign_investment_rights
							source_country = ROOT
							target_country = scope:other_party
						}
					}
				}
				effect_starting_politics_fiat_standard = yes
			}
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2						# REFUSE
		default_option = yes
		
		#Minor discontent in USA
		add_radicals = {
			value = large_radicals
			strata = lower
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }
