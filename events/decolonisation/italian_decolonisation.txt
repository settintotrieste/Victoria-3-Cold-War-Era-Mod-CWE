namespace = italian_decolonisation

#Somalia
italian_decolonisation.1 = {
    type = country_event
	placement = c:SOM
	
    title = italian_somalia.sgc.t 								# Title of the event; ADD LOCALISATION
    desc = italian_somalia.sgc.d 								# Description of the event; ADD LOCALISATION
    #flavor = italian_somalia.sgc.f 							# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:ITA ?= this
		game_date >= 1956.2.1									# START DATE: Date colony became self governing ****
		game_date < 1966.2.1									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_democracy = yes									# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#Colony conditions
		exists = c:SOM
		c:SOM ?= { 												# Change TAG of the colony ****
			is_direct_subject_of = ROOT
			is_a_democracy = no
			is_not_sufficiently_independent = yes
		}
		
		# Event has not been fired before
		NOT = { has_variable = italian_somalia_sgc }			# CHANGE ****
		
    }
	
	immediate = {
		
		# Fire only once
		set_variable = italian_somalia_sgc						# Change variable ****
		
	}
    
    option = {
        name = historical_yes_no.option.1						# Grant them independence
        highlighted_option = yes
		
		# Create democracy
		c:SOM ?= {
			effect_starting_politics_democratic = yes
			
			#Remove Italian Governor-General
			ruler ?= { retire_character = yes }
			
			#Create new leader
			create_character = {
				first_name = "Abdullahi_Issa_first_name"
				last_name = "Abdullahi_Issa_last_name"
				ruler = yes ig_leader = yes historical = yes
				birth_date = 1922.12.1
				interest_group = ig_rural_folk
				ideology = ideology_democratic
				culture = cu:somali
				religion = rel:sunni
				traits = {
					charismatic
					ambitious
					tactful
				}
				on_created = { set_variable = Abdullahi_Issa }
			}
			
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_yes_no.option.2					# REFUSE
		default_option = yes
		
		#Colonial unrest
		c:SOM ?= {
		
			# Natives are angry!
			add_radicals = {
				value = large_radicals
			}
			every_scope_state = {
				add_modifier = { name = nationalist_agitation months = 60 is_decaying = yes } 
			}
			
			#Colony is angry
			change_relations = {
				country = ROOT
				value = -100
			}
			
		}
		
		# Add Infamy because doing this is against international law
		create_incident = {
			value = 10
			country = ROOT
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}
