namespace = treaty_ports_system

# Lease of Treaty Port Terminated
treaty_ports_system.1 = {
    type = country_event
	placement = scope:treaty_port_lease_state

    title = treaty_port_lease_terminated.t 							# Title of the event; ADD LOCALISATION
    desc = treaty_port_lease_terminated.d 							# Description of the event; ADD LOCALISATION
    flavor = treaty_port_lease_terminated.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "unspecific_signed_contract" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {														# What triggers the event, if using other events or journal entries leave empty
		
		#Is triggered by event
		
    }
	
	immediate = {
	
		# Markers
		random_country = {
			limit = { has_variable = treaty_port_lease_landlord }
			save_scope_as = treaty_port_lease_landlord
			remove_variable = treaty_port_lease_landlord
		}
		if = {
			limit = { any_scope_state = { has_variable = treaty_port_lease_state } }
			random_scope_state = {
				limit = { has_variable = treaty_port_lease_state }
				save_scope_as = treaty_port_lease_state
				remove_variable = treaty_port_lease_state
			}
		}
		else = {
			random_country = {
				limit = {
					is_subject_of = ROOT
					is_sufficiently_independent = no
					any_scope_state = { has_variable = treaty_port_lease_state }
				}
				save_scope_as = treaty_port_lease_state_subject
				random_scope_state = {
					limit = { has_variable = treaty_port_lease_state }
					save_scope_as = treaty_port_lease_state
					remove_variable = treaty_port_lease_state
				}
			}
		}
		
	}
    
	cancellation_trigger = {
		NOR = {
			exists = scope:treaty_port_lease_landlord
			exists = scope:treaty_port_lease_state
		}
	}
	
	option = {
        name = treaty_port_lease_terminated.1						# Return state
		highlighted_option = yes
		
		#Return state unconditionally
		if = {
			limit = { scope:treaty_port_lease_state ?= { owner = scope:treaty_port_lease_state_subject } }
			scope:treaty_port_lease_landlord ?= {
				annex_with_incorporation = scope:treaty_port_lease_state_subject
			}
		}
		else = {
			scope:treaty_port_lease_state ?= {
				set_state_owner = scope:treaty_port_lease_landlord
			}
		}
		
		#Gain relations
		change_relations = { 
			country = scope:treaty_port_lease_landlord 
			value = 50 
		}
		
		ai_chance = { 
			base = 80
		}
    }
	
	option = {
        name = treaty_port_lease_terminated.2						# SAR
		trigger = {
			scope:treaty_port_lease_landlord ?= {
				is_sufficiently_independent = yes
				NOT = { is_subject_of = ROOT }
			}
			is_sufficiently_independent = yes
			NOT = { is_subject_of = scope:treaty_port_lease_landlord }
			NOR = {
				AND = { has_law_or_variant = law_type:law_HOS_monarch scope:treaty_port_lease_landlord = { has_law_or_variant = law_type:law_HOS_monarch } }
				AND = { has_law_or_variant = law_type:law_HOS_president scope:treaty_port_lease_landlord = { has_law_or_variant = law_type:law_HOS_president } }
				AND = { has_law_or_variant = law_type:law_HOS_religious_leader scope:treaty_port_lease_landlord = { has_law_or_variant = law_type:law_HOS_religious_leader } }
				AND = { has_law_or_variant = law_type:law_HOS_nationalist_leader scope:treaty_port_lease_landlord = { has_law_or_variant = law_type:law_HOS_nationalist_leader } }
				AND = { has_law_or_variant = law_type:law_HOS_communist_leader scope:treaty_port_lease_landlord = { has_law_or_variant = law_type:law_HOS_communist_leader } }
			}
		}
		
		#Mark again
		scope:treaty_port_lease_state = {
			set_variable = { name = treaty_port_lease_state days = 1 }
		}
		scope:treaty_port_lease_landlord = {
			set_variable = { name = treaty_port_lease_landlord days = 1 }
		}
		ROOT = {
			set_variable = { name = treaty_port_lease_tenant days = 1 }
		}
		
		#Response
		scope:treaty_port_lease_landlord ?= {
			trigger_event = { id = treaty_ports_system.3 }
		}
		
		ai_chance = { 
			base = 15
		}
    }
	
    option = {
        name = treaty_port_lease_terminated.3						# Refuse
		default_option = yes
		trigger = {
			scope:treaty_port_lease_landlord ?= {
				is_sufficiently_independent = yes
				NOT = { is_subject_of = ROOT }
			}
			is_sufficiently_independent = yes
			NOT = { is_subject_of = scope:treaty_port_lease_landlord }
		}
		
		# Add Infamy because doing this is against international law
		create_incident = {
			value = 10
			country = scope:treaty_port_lease_landlord
		}
		
		#Mark again
		scope:treaty_port_lease_state = {
			set_variable = { name = treaty_port_lease_state days = 1 }
		}
		scope:treaty_port_lease_landlord = {
			set_variable = { name = treaty_port_lease_landlord days = 1 }
		}
		ROOT = {
			set_variable = { name = treaty_port_lease_tenant days = 1 }
		}
		
		#Response
		scope:treaty_port_lease_landlord ?= {
			trigger_event = { id = treaty_ports_system.2 }
		}
		
		ai_chance = { 
			base = 5
		}
    }

}

# Lease of Treaty Port Terminated - War
treaty_ports_system.2 = {
    type = country_event
	placement = scope:treaty_port_lease_state

    title = treaty_port_lease_terminated_war.t 							# Title of the event; ADD LOCALISATION
    desc = treaty_port_lease_terminated_war.d 							# Description of the event; ADD LOCALISATION
    flavor = treaty_port_lease_terminated_war.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "unspecific_signed_contract" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 			# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 		# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {															# What triggers the event, if using other events or journal entries leave empty
		
		#Is triggered by event
		
    }
	
	immediate = {
	
		# Mark
		random_country = {
			limit = { has_variable = treaty_port_lease_tenant }
			save_scope_as = treaty_port_lease_tenant
			remove_variable = treaty_port_lease_tenant
			if = {
				limit = { any_scope_state = { has_variable = treaty_port_lease_state } }
				random_scope_state = {
					limit = { has_variable = treaty_port_lease_state }
					save_scope_as = treaty_port_lease_state
					remove_variable = treaty_port_lease_state
				}
			}
			else = {
				random_country = {
					limit = {
						is_subject_of = scope:treaty_port_lease_tenant
						is_sufficiently_independent = no
						any_scope_state = { has_variable = treaty_port_lease_state }
					}
					save_scope_as = treaty_port_lease_state_subject
					random_scope_state = {
						limit = { has_variable = treaty_port_lease_state }
						save_scope_as = treaty_port_lease_state
						remove_variable = treaty_port_lease_state
					}
				}
			}
		}
		
	}
    
	cancellation_trigger = {
		NOR = {
			exists = scope:treaty_port_lease_tenant
			exists = scope:treaty_port_lease_state
		}
	}
	
	option = {
        name = treaty_port_lease_war.1								# War
		highlighted_option = yes
		
		#War to reclaim state
		if = {
			limit = { scope:treaty_port_lease_state = { owner = { is_subject_of = scope:treaty_port_lease_tenant } } }
			create_diplomatic_play = {
				name = second_indo_pakistan_war
				type = dp_annex_war
				target_country = scope:treaty_port_lease_state.owner
				
				war = no
			}
		}
		else = {
			create_diplomatic_play = {
				name = second_indo_pakistan_war
				type = dp_return_state
				target_state = scope:treaty_port_lease_state
				
				war = no
			}
		}
		
		ai_chance = { 
			base = 90
		}
    }
	
    option = {
        name = treaty_port_lease_war.2								# Leave it be
		default_option = yes
		
		#Give up claim
		scope:treaty_port_lease_state ?= {
			state_region = {
				remove_claim = ROOT
			}
		}
		
		#Domestic effects
		add_modifier = { 
			name = domestic_unrest
			years = 5
			is_decaying = yes 
		}
		add_modifier = { 
			name = national_humiliation 
			years = 5
			is_decaying = yes 
		}
		
		ai_chance = {
			base = 10
		}
    }

}

treaty_ports_system.3 = {
    type = country_event
	placement = scope:treaty_port_lease_state

    title = treaty_port_lease_terminated_SAR.t 							# Title of the event; ADD LOCALISATION
    desc = treaty_port_lease_terminated_SAR.d 							# Description of the event; ADD LOCALISATION
    flavor = treaty_port_lease_terminated_SAR.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { texture = "unspecific_signed_contract" }

    icon = "gfx/interface/icons/event_icons/waving_flag.dds" 			# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 		# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {															# What triggers the event, if using other events or journal entries leave empty
		
		#Is triggered by event
		
    }
	
	immediate = {
	
		# Mark
		random_country = {
			limit = { has_variable = treaty_port_lease_tenant }
			save_scope_as = treaty_port_lease_tenant
			remove_variable = treaty_port_lease_tenant
			if = {
				limit = { any_scope_state = { has_variable = treaty_port_lease_state } }
				random_scope_state = {
					limit = { has_variable = treaty_port_lease_state }
					save_scope_as = treaty_port_lease_state
					remove_variable = treaty_port_lease_state
					state_region = { save_scope_as = state_to_release }
				}
			}
			else = {
				random_country = {
					limit = {
						is_subject_of = scope:treaty_port_lease_tenant
						is_sufficiently_independent = no
						any_scope_state = { has_variable = treaty_port_lease_state }
					}
					save_scope_as = treaty_port_lease_state_subject
					random_scope_state = {
						limit = { has_variable = treaty_port_lease_state }
						save_scope_as = treaty_port_lease_state
						remove_variable = treaty_port_lease_state
					}
				}
			}
		}
		
	}
    
	cancellation_trigger = {
		NOR = {
			exists = scope:treaty_port_lease_tenant
			exists = scope:treaty_port_lease_state
		}
	}
	
	option = {
        name = treaty_port_lease_SAR.1								# SAR
		highlighted_option = yes
		
		#Release 1 SAR
		if = {
			limit = { scope:treaty_port_lease_state ?= { owner = scope:treaty_port_lease_tenant } }
			save_scope_as = treaty_port_lease_landlord
			scope:treaty_port_lease_state ?= {
				state_region = { save_scope_as = state_to_release }
			}
			scope:treaty_port_lease_landlord ?= {
				random_primary_culture = {
					save_scope_as = scope:culture_to_release
				}
			}
			custom_tooltip = {
				text = special_administrative_region_2.tt
				scope:treaty_port_lease_tenant ?= {
					random_scope_state = {
						limit = { state_region = scope:state_to_release }
						set_variable = { name = state_to_cede months = 3 }
					}
					create_dynamic_country = {
						origin = ROOT
						country_type = unrecognized
						tier = city_state
						state = scope:state_to_release
						cede_state_trigger = {
							has_variable = state_to_cede
						}
						on_created = {
						
							every_scope_state = {
								remove_variable = state_to_cede
							}
						
							set_release_vassal_effects = yes
							
							copy_laws = scope:treaty_port_lease_tenant
							
						}
					}
				}
				
				#SAR diplomatic pacts
				scope:treaty_port_lease_landlord ?= {
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state.owner
						type = special_administrative_region
					}
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state.owner
						type = currency_union
					}
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state.owner
						type = legal_union
					}
				}
			}
		}
		if = {
			limit = { exists = scope:treaty_port_lease_state_subject }
			hidden_effect = {
				scope:treaty_port_lease_state_subject ?= { make_independent = yes }
			}
			custom_tooltip = {
				text = terminate_treaty_port_lease_transfer_subject.tt
				scope:treaty_port_lease_landlord ?= {
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state_subject
						type = special_administrative_region
					}
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state_subject
						type = currency_union
					}
					create_diplomatic_pact = {
						country = scope:treaty_port_lease_state_subject
						type = legal_union
					}
				}
			}
		}
		
		ai_chance = { 
			base = 90
		}
    }
	
    option = {
        name = treaty_port_lease_SAR.2								# War
		default_option = yes
		
		#War to reclaim state
		if = {
			limit = { scope:treaty_port_lease_state = { owner = { is_subject_of = scope:treaty_port_lease_tenant } } }
			create_diplomatic_play = {
				name = second_indo_pakistan_war
				type = dp_annex_war
				target_country = scope:treaty_port_lease_state.owner
				
				war = no
			}
		}
		else = {
			create_diplomatic_play = {
				name = second_indo_pakistan_war
				type = dp_return_state
				target_state = scope:treaty_port_lease_state
				
				war = no
			}
		}
		
		ai_chance = {
			base = 10
		}
    }
