namespace = country_tag_clean_up_system

#IAR changes to Rhineland upon independence
country_tag_clean_up_system.1 = {
	type = country_event
	hidden = yes

	trigger = {
		c:IAR ?= this
		NOT = { exists = c:RHE }
		
		is_sufficiently_independent = yes
		is_at_war = no
	}

	immediate = {
		change_tag = RHE
	}

}
