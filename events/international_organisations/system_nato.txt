namespace = nato_system

#Set NATO leader
nato_system.1 = {
	type = country_event
	hidden = yes

	trigger = {
		
		OR = {
		
			#No NATO leader and there is an eligible one
			AND = { 
			
				NOT = { any_country = { has_modifier = nato_leader } }
				
				any_country = { 
					has_modifier = nato_member 
					OR = { 
						country_is_in_europe = yes
						country_is_in_north_america = yes
					}
					is_a_democracy = yes
					country_rank = rank_value:super_power
				}
				
			}
			
			#NATO leader is wrong
			AND = { 
				has_modifier = nato_leader 
				OR = {
					NOT = { is_sufficiently_independent = yes }
					NOT = { is_not_neutral = yes }
					NOT = { country_rank = rank_value:super_power }
				}
			}
			
			#Overlapping modifiers
			any_country = { has_modifier = nato_leader has_modifier = nato_member }
			
		}
		
	}

	immediate = {
		
		#Clear NATO leader
		every_country = {
			limit = { has_modifier = nato_leader }
			remove_modifier = nato_leader
			if = { 
				limit = { NOT = { has_modifier = nato_member } }
				add_modifier = { name = nato_member }
			}
		}
	
		#Set NATO leader
		if = {
			limit = { 
				NOT = { any_country = { has_modifier = nato_leader } }
				
				any_country = { 
					has_modifier = nato_member 
					OR = { 
						country_is_in_europe = yes
						country_is_in_north_america = yes
					}
					is_a_democracy = yes
					country_rank = rank_value:super_power
				}
			}
			random_country = {
				limit = {
					has_modifier = nato_member 
					OR = { 
						country_is_in_europe = yes
						country_is_in_north_america = yes
					}
					is_a_democracy = yes
					country_rank = rank_value:super_power
				}
				add_modifier = { name = nato_leader }
			}
		}
		else_if = {
			limit = {
				has_modifier = nato_leader 
				OR = {
					NOT = { is_sufficiently_independent = yes }
					NOT = { is_not_neutral = yes }
					NOT = { country_rank = rank_value:super_power }
				}
			}
			random_country = {
				limit = {
					has_modifier = nato_member 
					OR = { 
						country_is_in_europe = yes
						country_is_in_north_america = yes
					}
					is_a_democracy = yes
					country_rank = rank_value:super_power
				}
				add_modifier = { name = nato_leader }
			}
		}
		
		#Clean up memberships
		if = { 
			limit = { has_modifier = nato_leader has_modifier = nato_member }
			remove_modifier = nato_member
		}
		
	}

}

#Set NATO relations
nato_system.2 = {
	type = country_event
	hidden = yes

	trigger = {
		
		OR = { has_modifier = nato_member has_modifier = nato_leader }
		
		any_country = { 
			OR = { has_modifier = nato_member has_modifier = nato_leader }
			NOR = { any_scope_treaty = { binds = ROOT any_scope_article = { has_type = defensive_pact } } any_scope_treaty = { binds = ROOT any_scope_article = { has_type = alliance } } }
		}
		
	}

	immediate = {
		
		#Force defensive pact on NATO members
		every_country = { 
			limit = { 
				OR = { has_modifier = nato_member has_modifier = nato_leader }
				NOR = { any_scope_treaty = { binds = ROOT any_scope_article = { has_type = defensive_pact } } any_scope_treaty = { binds = ROOT any_scope_article = { has_type = alliance } } }
			}
			save_scope_as = target_country
			create_treaty = {
				name = treaty_name_NATO_alliance
				first_country = ROOT
				second_country = scope:target_country

				is_draft = no
				#binding_period = { years = 10 }

				articles_to_create = {
					{
						article = defensive_pact
					}
				}
			}
		}
		
	}

}

#Dissolve NATO
nato_system.3 = {
	type = country_event
	hidden = yes

	trigger = {
		
		OR = { 
		
			#THIS is the sole member of NATO
			AND = { 
				OR = { has_modifier = nato_member has_modifier = nato_leader }
				NOT = { any_country = { has_modifier = nato_member } }
			}
			
			#No NATO leader
			AND = { 
				NOT = { has_modifier = nato_leader }
				NOT = { any_country = { has_modifier = nato_leader } }
			}
			
		}
		
	}

	immediate = {
		
		#Everyone leaves NATO
		remove_modifier = nato_member
		remove_modifier = nato_leader
		
		every_country = {
			remove_modifier = nato_member
			remove_modifier = nato_leader
		}
		
	}

}

#Kick ineligible countries out of NATO
nato_system.4 = {
	type = country_event
	title = nato_expulsion.t 					# Title of the event; ADD LOCALISATION
    desc = nato_expulsion.d 					# Description of the event; ADD LOCALISATION
    flavor = nato_expulsion.f 					# Flavor for event; ADD LOCALISATION, IF REQUIRED

	event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/diplo_icons/nato_member.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met

	trigger = {
		
		OR = { 
			has_modifier = nato_member
			has_modifier = nato_leader
			has_variable = nato_application_var
		}
		
		OR = { 
			NOT = { is_sufficiently_independent = yes }
			NOT = { is_not_neutral = yes }
			is_a_communist_government = yes
		}
		
	}

	immediate = {
	
		#THIS leaves NATO
		remove_modifier = nato_member
		remove_modifier = nato_leader
		
		#Terminate any applications
		remove_variable = nato_application_var
		
		#Block reapplication for 5 years
		set_variable = {
			name = left_nato_var
			months = 60
		}
		
	}
	
	option = {
        name = nato_application_rejection.1				# Unfortunate
		
    }

}

#Option for NATO leader to accept or reject application for membership
nato_system.5 = {
	title = nato_application_choice.t 					# Title of the event; ADD LOCALISATION
    desc = nato_application_choice.d 					# Description of the event; ADD LOCALISATION
    flavor = nato_application_choice.f 				# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met

	trigger = {
		
		has_modifier = nato_leader
		
		any_country = {
			has_variable = nato_application_var
			
			is_sufficiently_independent = yes
			is_not_neutral = yes
			is_at_war = no
		}
		
	}

	immediate = {
	
		random_country = {
			limit = { has_variable = nato_application_var }
			save_scope_as = nato_applicant
		}
		
		ROOT = {
			save_scope_as = nato_leader
		}
		
	}

	option = {
        name = nato_application_choice.option.1			# Accept application from PREV
        highlighted_option = yes
		
		#Accept FROM into NATO
		scope:nato_applicant = {
			#FROM joins NATO
			add_modifier = { name = nato_member }
			remove_variable = nato_application_var
		}
		
		#Create defensive alliance with FROM
		every_country = {
			limit = { 
				OR = { has_modifier = nato_member has_modifier = nato_leader }
				NOR = { any_scope_treaty = { binds = ROOT any_scope_article = { has_type = defensive_pact } } any_scope_treaty = { binds = ROOT any_scope_article = { has_type = alliance } } }
			}
			save_scope_as = target_country
			create_treaty = {
				name = treaty_name_NATO_alliance
				first_country = ROOT
				second_country = scope:target_country

				is_draft = no
				#binding_period = { years = 10 }

				articles_to_create = {
					{
						article = defensive_pact
					}
				}
			}
			change_relations = {
				country = scope:nato_applicant
				value = 50
			}
		}
		
		#Inform applicant of successful outcome
		scope:nato_applicant = { trigger_event = { id = nato_system.7 } }
		
		#FROM is happy with THIS
		change_relations = {
			country = scope:nato_applicant
			value = 50
		}
		
		#Hostile countries become more hostile
		every_country = {
			limit = { 
				has_attitude = {
					who = ROOT
					attitude <= antagonistic
				}
			}
			change_relations = {
				country = root
				value = -25
			}
		}
		
		ai_chance = { 
			base = 99
		}
		
    }
	
	option = {
        name = nato_application_choice.option.2 # Refuse application from PREV
		default_option = yes
		
		#Inform applicant of unsuccessful outcome
		scope:nato_applicant = { 
			trigger_event = { id = nato_system.6 } 
			remove_variable = nato_application_var
		}
		
		#FROM is angry!
		change_relations = {
			country = scope:nato_applicant
			value = -50
		}
		
		#Hostile countries become less hostile
		every_country = {
			limit = { 
				has_attitude = {
					who = ROOT
					attitude <= antagonistic
				}
			}
			change_relations = {
				country = root
				value = 25
			}
		}
		
		ai_chance = { 
			base = 1
		}
		
    }

}

#Notification of NATO call for help
nato_system.8 = {
	placement = scope:nato_victim
	
	title = nato_join_war.t 							# Title of the event; ADD LOCALISATION
    desc = nato_join_war.d 								# Description of the event; ADD LOCALISATION
    flavor = nato_join_war.f 							# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/diplo_icons/nato_member.dds" 		# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met

	immediate = {
		
		random_country = {
			limit = { has_variable = requested_nato_intervention }
			save_scope_as = nato_victim
		}
		
	}

	option = {
        name = nato_join_war.1							# Fortunate
		
		#Join NATO war
		custom_tooltip = {
			text = join_nato_war.tt
			every_diplomatic_play = {
				limit = {
					is_war = yes
					any_scope_play_involved = {
						this = scope:nato_victim
					}
					initiator = {
						NOT = { this = scope:nato_victim }
						NOR = { 
							has_modifier = nato_member
							has_modifier = nato_leader
						}
					}
					target = {
						this = scope:nato_victim
						OR = { 
							has_modifier = nato_member
							has_modifier = nato_leader
						}
					}
					OR = {
						has_play_goal = open_market
						has_play_goal = regime_change
						has_play_goal = conquer_state
						has_play_goal = return_state
						has_play_goal = liberate_subject
						has_play_goal = liberate_country
						has_play_goal = take_treaty_port
						has_play_goal = make_tributary
						has_play_goal = make_protectorate
						has_play_goal = transfer_subject
						has_play_goal = annex_country
						has_play_goal = humiliation
						has_play_goal = unification
						has_play_goal = unification_leadership
					}
				}
				add_target_backers = { ROOT }
			}
		}
		
    }
