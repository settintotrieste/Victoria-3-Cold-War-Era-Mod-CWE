#Underwrite all insufficiently independent subjects
colonial_empire_system.2 = {
	placement = scope:colonial_empire_debtor
	
	title = colonial_empire_bankroll.t 								# Title of the event; ADD LOCALISATION
    desc = colonial_empire_bankroll.d 								# Description of the event; ADD LOCALISATION
    flavor = colonial_empire_bankroll.f 							# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/diplo_icons/colonial_empire_member.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 			# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met

	trigger = {
		
		any_subject_or_below = {
			is_direct_subject_of = ROOT
			is_sufficiently_independent = no
			scaled_debt > 0.5 net_fixed_income < 0
			OR = { 
				NOT = { 
					any_scope_diplomatic_pact = {
						is_diplomatic_action_type = bankroll_subject
						first_country = ROOT
					}
				}
				in_default = yes
			}
		}
		
	}

	immediate = {
		
		random_country = {
			limit = { 
				is_direct_subject_of = ROOT
				is_sufficiently_independent = no
				scaled_debt > 0.5 net_fixed_income < 0
				OR = { 
					NOT = { 
						any_scope_diplomatic_pact = {
							is_diplomatic_action_type = bankroll_subject
							first_country = ROOT
						}
					}
					in_default = yes
				}
			}
			save_scope_as = colonial_empire_debtor
		}
		
	}

	option = {
        name = colonial_empire_bankroll.1							# Fortunate
		
		if = {
			limit = { exists = scope:colonial_empire_debtor }
			if = {
				limit = { NOT = { has_diplomatic_pact = { who = scope:colonial_empire_debtor type = bankroll_subject is_initiator = yes } } }
				create_diplomatic_pact = {
					country = scope:colonial_empire_debtor
					type = bankroll_subject
				}
			}
			if = {
				limit = { scope:colonial_empire_debtor = { in_default = yes } }
				take_on_scaled_debt = {
					who = scope:colonial_empire_debtor
					value = scope:colonial_empire_debtor.scaled_debt
				}
				save_scope_as = colonial_empire_creditor
				post_notification = requested_overlord_takes_subject_debt_notification
			}
		}
		
    }

}
