#Create superstate power bloc
superstate_system.2 = {
	type = country_event
	hidden = yes

	trigger = {
		
		#Triggered by event
		
	}

	immediate = {
		
		#Create superstate power bloc
		if = {
			limit = { 
				is_in_power_bloc = no
			}
			if = {
				limit = {
					has_variable = superstate_customs_union
					NOT = {
						any_power_bloc = {
							has_identity = identity:identity_superstate
							power_bloc_leader = {
								shares_same_superstate = yes
							}
						}
					}
				}
				ordered_country = {
					limit = {
						has_variable = superstate_customs_union
						shares_same_superstate = yes
						is_power_bloc_leader = no
						is_in_power_bloc = no
					}
					order_by = prestige
					save_scope_as = superstate_power_bloc
				}
				if = {
					limit = { 
						exists = scope:superstate_power_bloc
						scope:superstate_power_bloc ?= {
							is_in_power_bloc = no
						}
					}
					scope:superstate_power_bloc = {
						if = { limit = { has_variable = XNA_ambitions } create_power_bloc = { name = XNA map_color = { 66 94 193 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XCA_ambitions } create_power_bloc = { name = XCA map_color = { 40 233 255 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XSA_ambitions } create_power_bloc = { name = XSA map_color = { 0 36 144 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XEU_ambitions } create_power_bloc = { name = XEU map_color = { 70 130 180 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XER_ambitions } create_power_bloc = { name = XER map_color = { 177 73 165 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XAL_ambitions } create_power_bloc = { name = XAL map_color = { 0 98 51 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XAU_ambitions } create_power_bloc = { name = XAU map_color = { 120 98 57 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XEA_ambitions } create_power_bloc = { name = XEA map_color = { 0 56 168 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XSE_ambitions } create_power_bloc = { name = XSE map_color = { 255 53 35 } identity = identity_superstate  } }
						else_if = { limit = { has_variable = XPI_ambitions } create_power_bloc = { name = XPI map_color = { 1 41 115 } identity = identity_superstate  } }
						power_bloc ?= { 
							if = {
								limit = { NOT = { has_principle = principle:principle_foreign_investment_3 } }
								add_principle = principle_foreign_investment_3
							}
							if = {
								limit = { NOT = { has_principle = principle:principle_freedom_of_movement_3 } }
								add_principle = principle_freedom_of_movement_3
							}
							if = {
								limit = { NOT = { has_principle = principle:principle_internal_trade_3 } }
								add_principle = principle_internal_trade_3
							}
						}
					}
					if = {
						limit = {  
							OR = {
								is_in_power_bloc = yes
								power_bloc ?= { 
									OR = { 
										power_bloc_leader = { NOT = { shares_same_superstate = yes } }
										NOT = { has_identity = identity:identity_superstate } 
									}
								}
							}
						}
						join_power_bloc = scope:superstate_power_bloc
					}
				}
			}
			if = {
				limit = {
					has_variable = superstate_customs_union
					any_power_bloc = {
						has_identity = identity:identity_superstate
						power_bloc_leader = {
							shares_same_superstate = yes
						}
					}
				}
				random_country = {
					limit = { 
						shares_same_superstate = yes
						is_in_power_bloc = yes
						is_power_bloc_leader = yes
						power_bloc ?= { has_identity = identity:identity_superstate }
					}
					save_scope_as = superstate_power_bloc
				}
				if = {
					limit = { exists = scope:superstate_power_bloc }
					join_power_bloc = scope:superstate_power_bloc
				}
			}
		}
		
	}

}
