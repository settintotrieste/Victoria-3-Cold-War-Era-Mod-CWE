namespace = historical_restore_monarchy

historical_restore_monarchy.1 = {
	type = country_event
	hidden = yes
	
	trigger = {															# What triggers the event, if using other events or journal entries leave empty
		
		#King is the ruler
		is_a_regime_monarchy = yes										# Correct gov types
		
		#Too many kings
		any_scope_character = {
			has_modifier = monarch_character
			count > 1
		}
		
    }
	
	immediate = {
	
		#Prefer historical characters; remove all excess monarchs
		if = {
			limit = { 
				any_scope_character = {
					has_modifier = monarch_character
					is_historical = yes
					is_ruler = no
				}
			}
			random_scope_character = {
				limit = { 
					has_modifier = monarch_character 
					is_historical = yes
					is_ruler = no
				}
				set_character_as_monarch = yes
			}
		}
		every_scope_character = {
			limit = { 
				has_modifier = monarch_character 
				is_ruler = no
			}
			kill_character = { hidden = yes }
		}
		
	}

}
