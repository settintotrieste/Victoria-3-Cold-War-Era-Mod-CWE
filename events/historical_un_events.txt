#Universal Declaration of Human Rights
historical_un_events.1 = {
    type = country_event

    title = 1948_Universal_Declaration_of_Human_Rights.t 			# Title of the event; ADD LOCALISATION
    desc = 1948_Universal_Declaration_of_Human_Rights.d 			# Description of the event; ADD LOCALISATION
    #flavor = 1948_Universal_Declaration_of_Human_Rights.f 			# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_signed_contract.bk2" }

    icon = "gfx/interface/icons/event_icons/united_nations.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {														# What triggers the event, if using other events or journal entries leave empty
		
		is_player = yes
		
		has_global_variable = 1946_start_date
		
		game_date >= 1948.12.10										# START DATE: Date colony became self governing ****
		game_date < 1958.12.10										# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		# Event has not been fired before
		NOT = { has_variable = 1948_Universal_Declaration_of_Human_Rights_hue }	# CHANGE ****
		
    }
    
    immediate = {
		
		# Fire only once
		set_variable = 1948_Universal_Declaration_of_Human_Rights_hue			# Change variable ****
		
	}

	option = {
		name = historical_un_events.option.1
		
		#Every Colonial Country gains Fervor
		if = {
			limit = { NOT = { has_global_variable = 1948_Universal_Declaration_of_Human_Rights_hue } }
			set_global_variable = 1948_Universal_Declaration_of_Human_Rights_hue
			hidden_effect = {
				every_country = {
					save_scope_as = THIS_scope
					every_scope_state = {
						limit = { is_incorporated = no }
						every_scope_culture = {
							limit = { NOT = { is_primary_culture_of = scope:THIS_scope } }
							add_fervor = 10
						}
					}
				}
			}
		}
		
	}

}
