# 1948 coup d'état
VNZ_gov_changes.1 = {
    type = country_event

    title = VNZ_1948_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = VNZ_1948_revolution.d 							# Description of the event; ADD LOCALISATION
    #flavor = VNZ_1948_revolution.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		has_global_variable = 1946_start_date
		
		#Overlord conditions
		c:VNZ ?= this
		game_date >= 1948.11.24									# START DATE: Date colony became self governing ****
		game_date < 1958.11.24									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_republican_democracy = yes
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		# Event has not been fired before
		NOT = { has_variable = VNZ_1946_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = VNZ_1946_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Become independent
		if = {
			limit = { is_subject = yes }
			make_independent = yes
		}
		
		#Enact new regime
		effect_starting_politics_presidential_republic = yes
		effect_starting_politics_regime = yes
		effect_increase_corruption = yes
		
		#Carlos Delgado Chalbaud
		create_character = {
			first_name = "carlos_delgado_chalbaud_first_name"
			last_name = "carlos_delgado_chalbaud_last_name"
			ruler = yes ig_leader = yes historical = yes
			birth_date = 1909.1.20
			interest_group = ig_armed_forces
			ideology = ideology_autocratic
			culture = cu:venezuelan
			traits = {
				ambitious
				innovative
			}
			on_created = { set_variable = carlos_delgado_chalbaud }
		}
		
		#Major discontent
		add_radicals = {
			value = large_radicals
			strata = lower
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals
			strata = upper
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}
