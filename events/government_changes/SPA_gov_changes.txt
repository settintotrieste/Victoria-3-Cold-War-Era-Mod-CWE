# Law of Succession to the Headship of the State
SPA_gov_changes.1 = {
    type = country_event

    title = SPA_1947_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = SPA_1947_revolution.d 							# Description of the event; ADD LOCALISATION
    #flavor = SPA_1947_revolution.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#1946 start date only
		has_global_variable = 1946_start_date
		
		#Overlord conditions
		c:SPA ?= this
		game_date >= 1947.6.7									# START DATE: Date colony became self governing ****
		game_date < 1957.6.7									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_nationalist_government = yes						# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		ruler ?= { has_variable = Francisco_Franco_Bahamonde }
		
		# Event has not been fired before
		NOT = { has_variable = SPA_1947_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = SPA_1947_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Boost to government
		add_modifier = { 
			name = strengthened_government_authority
			years = 5
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#No restoration of Juan Carlos event
		set_variable = SPA_francoist_monarchy_rejected
		
		#Major discontent
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_landowners
		}
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_devout
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { historical_railroading_active = yes } add = -100 } 
		}
		
    }

}

SPA_gov_changes.3 = {
    type = country_event

    title = SPA_1975_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = SPA_1975_revolution.d 							# Description of the event; ADD LOCALISATION

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:SPA ?= this
		game_date >= 1975.11.20									# START DATE: Date colony became self governing ****

		ruler ?= { has_variable = Francisco_Franco_Bahamonde }
		
		NOT = { has_variable = SPA_francoist_monarchy_rejected }
		
		# Event has not been fired before
		NOT = { has_variable = SPA_1975_hgc }					# CHANGE ****
		
		#No AH
		NOT = {
			any_scope_character = { has_variable = Luis_Carrero_Blanco }
		}
		
    }
	
	immediate = {
	
		#Kill ruler
		random_scope_character = {
			limit = { has_variable = Francisco_Franco_Bahamonde }
			kill_character = yes
		}
	
		# Fire only once
		set_variable = SPA_1975_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Enact new regime
		effect_starting_politics_semi_constitutional_monarchy = yes
		
		#Juan Carlos I
		create_character = {
			first_name = "juan_carlos_i_first_name"
			last_name = "juan_carlos_i_last_name"
			birth_date = 1938.1.5
			ruler = yes ig_leader = yes historical = yes
			interest_group = ig_landowners
			ideology = ideology_elitist
			culture = cu:spanish
			religion = rel:catholic
			traits = {
				reserved
				pious
				romantic
			}
			on_created = { set_variable = juan_carlos_i }
		}
		
		#Carlos Arias Navarro
		if = {
			limit = { 
				NOT = { 
					any_scope_character = {
						has_variable = Carlos_Arias_Navarro
					}
				}
			}
			create_character = {
				first_name = "Carlos_Arias_Navarro_first_name"
				last_name = "Carlos_Arias_Navarro_last_name"
				ig_leader = yes historical = yes
				birth_date = 1908.12.11
				interest_group = ig_armed_forces
				ideology = ideology_power_hungry
				culture = cu:spanish
				religion = rel:catholic
				traits = {
					cruel
					imperious
					wrathful
				}
				on_created = { set_variable = Carlos_Arias_Navarro }
			}
		}
		
		#For transitional flag
		set_variable = legacy_francoist_spain
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Carlos Arias Navarro
		if = {
			limit = { 
				NOT = { 
					any_scope_character = {
						has_variable = Carlos_Arias_Navarro
					}
				}
			}
			create_character = {
				first_name = "Carlos_Arias_Navarro_first_name"
