		effect_starting_politics_trade_policy_free_trade = yes

		create_character = {
			first_name = "Fulbert_Youlou_first_name"

			birth_date = 1917.7.19

			religion = rel:catholic
			traits = {
				pious
				innovative
				cruel
			}
			on_created = { set_variable = Fulbert_Youlou }
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
