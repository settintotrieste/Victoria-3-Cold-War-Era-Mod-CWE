namespace = QAT_gov_changes

# Abdication of Ali bin Abdullah Al Thani 
QAT_gov_changes.1 = {
    type = country_event

    title = QAT_1960_revolution.t 								# Title of the event; ADD LOCALISATION
    desc = QAT_1960_revolution.d 								# Description of the event; ADD LOCALISATION
    #flavor = QAT_1960_revolution.f 							# Flavor for event; ADD LOCALISATION, IF REQUIRED

	event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:QAT ?= this
		game_date >= 1960.10.24									# START DATE: Date colony became self governing ****
		game_date < 1970.10.24									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_regime_monarchy = yes								# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		ruler ?= { has_variable = Ali_of_Qatar }
		
		# Event has not been fired before
		NOT = { has_variable = QAT_1960_revolution_hgc }		# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = QAT_1960_revolution_hgc					# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Leader dies
		ruler ?= { retire_character_scripted_effect = yes }
		if = {
			limit = { any_scope_character = { has_variable = Ahmad_bin_Ali_Al_Thani } }
			random_scope_character = {
				limit = { has_variable = Ahmad_bin_Ali_Al_Thani }
				set_character_as_ruler = yes
			}
		}
		else = {
			create_character = {
				first_name = "Ahmad_bin_Ali_Al_Thani_first_name"
				last_name = "Ahmad_bin_Ali_Al_Thani_last_name"
				ruler = yes ig_leader = yes historical = yes
				birth_date = 1922.1.1
				interest_group = ig_industrialists
				ideology = ideology_autocratic
				religion = rel:sunni
				traits = {
					tactful
					pious
					innovative
				}
				on_created = {
					set_variable = Ahmad_bin_Ali_Al_Thani
				}
			}
		}
		
		#New Heir
		create_character = {
			first_name = "Khalifa_bin_Hamad_Al_Thani_first_name"
			last_name = "Khalifa_bin_Hamad_Al_Thani_last_name"
			heir = yes historical = yes
			birth_date = 1932.9.17
			interest_group = ig_landowners
			ideology = ideology_elitist
			religion = rel:sunni
			traits = {
				experienced_diplomat
				ambitious
				expensive_tastes
			}
			on_created = {
				set_variable = Khalifa_bin_Hamad_Al_Thani
			}
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }

}
