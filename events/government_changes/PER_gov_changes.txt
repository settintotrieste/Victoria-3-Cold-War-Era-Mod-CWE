		game_date < 1959.5.8									# END DATE: All historical events must not fire 10 years after the START DATE ****

		NOT = { has_law_or_variant = law_type:law_bicameralism }
		
		# Event has not been fired before
		NOT = { has_variable = PER_1949_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = PER_1949_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Enact new regime
		activate_law = law_type:law_bicameralism
		activate_law = law_type:law_censorship
		activate_law = law_type:law_draconian_policing
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { historical_railroading_active = yes } add = -100 } 
		}
		
    }
