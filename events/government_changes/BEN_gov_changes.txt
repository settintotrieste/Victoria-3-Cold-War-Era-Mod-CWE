namespace = BEN_gov_changes

# The People's Republic of Benin
BEN_gov_changes.1 = {
    type = country_event

    title = BEN_1972_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = BEN_1972_revolution.d 							# Description of the event; ADD LOCALISATION
    #flavor = BEN_1972_revolution.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:BEN ?= this
		game_date >= 1972.10.26									# START DATE: Date colony became self governing ****
		game_date < 1982.10.26									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_communist_government = no							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		# Event has not been fired before
		NOT = { has_variable = BEN_1972_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = BEN_1972_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Enact new regime
		effect_starting_politics_communist_republic = yes
		effect_starting_politics_dictatorship = yes
		effect_starting_politics_economic_policy_planned_economy = yes
		
		#Mathieu Kérékou
		create_character = {
			first_name = "Mathieu_Kerekou_first_name"
			last_name = "Mathieu_Kerekou_last_name"
			ruler = yes ig_leader = yes historical = yes
			birth_date = 1933.9.2
			interest_group = ig_trade_unions
			ideology = ideology_autocratic
			culture = cu:beninese
			religion = rel:catholic
			traits = {
				cruel
				ambitious
				persistent
			}
			on_created = { set_variable = Mathieu_Kerekou }
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals
			strata = lower
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}
