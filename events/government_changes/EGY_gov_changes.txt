# Egyptian Revolution of 1952
EGY_gov_changes.1 = {
    type = country_event

    title = EGY_1952_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = EGY_1952_revolution.d 								# Description of the event; ADD LOCALISATION
    #flavor = EGY_1952_revolution.f 							# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:EGY ?= this
		game_date >= 1952.7.23									# START DATE: Date colony became self governing ****
		game_date < 1962.7.23									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_monarchy = yes										# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		ruler ?= { has_variable = farouk_i }
		
		# Event has not been fired before
		NOT = { has_variable = EGY_1952_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = EGY_1952_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Become independent
		if = {
			limit = { is_subject = yes }
			make_independent = yes
		}
		
		#Leader flees
		ruler ?= { exile_character = yes }
		set_variable = farouk_i_abdicated
		
		#Enact new regime
		effect_starting_politics_constitutional_monarchy = yes
		effect_starting_politics_dictatorship = yes
		
		#Create new leader
		if = {
			limit = { any_scope_character = { has_variable = Mohamed_Naguib } }
			random_scope_character = { 
				limit = { has_variable = Mohamed_Naguib }
				set_character_as_ruler = yes
			}
		}
		else = {
			create_character = {
				first_name = "Mohamed_Naguib_first_name"
				last_name = "Mohamed_Naguib_last_name"
				ruler = yes ig_leader = yes historical = yes
				is_general = yes
				commander_rank = commander_rank_5
				birth_date = 1901.1.19
				interest_group = ig_industrialists
				ideology = ideology_elitist
				culture = cu:egyptian
				religion = rel:sunni
				traits = {
					honorable
					popular_commander
					tactful
				}
				on_created = { set_variable = Mohamed_Naguib }
			}
		}
		create_character = {
			first_name = "Gamal_Abdel_Nasser_first_name"
			last_name = "Gamal_Abdel_Nasser_last_name"
			ig_leader = yes historical = yes
			is_general = yes
			commander_rank = commander_rank_5
			birth_date = 1918.1.15
			interest_group = ig_armed_forces
			ideology = ideology_power_hungry
			culture = cu:egyptian
			religion = rel:sunni
			traits = {
				charismatic
				ambitious
				imposing
			}
			on_created = { set_variable = Gamal_Abdel_Nasser }
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { has_global_variable = historical_railroading_active } add = -100 } 
		}
		
    }

}
