namespace = MGL_gov_changes

# Yumjaagiin Tsedenbal
MGL_gov_changes.1 = {
    type = country_event

    title = MGL_1952_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = MGL_1952_revolution.d 							# Description of the event; ADD LOCALISATION
    #flavor = MGL_1952_revolution.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:MGL ?= this
		game_date >= 1952.1.26									# START DATE: Date colony became self governing ****
		game_date < 1962.1.26									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_communist_government = yes
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#Father
		ruler ?= { has_variable = Khorloogiin_Choibalsan }
		
		# Event has not been fired before
		NOT = { has_variable = MGL_1952_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = MGL_1952_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Death
		random_scope_character = {
			limit = { has_variable = Khorloogiin_Choibalsan }
			kill_character = yes
		}
		
		#Install new ruler
		if = {
			limit = { any_scope_character = { has_variable = Yumjaagin_Tsedenbal } }
			random_scope_character = {
				limit = { has_variable = Yumjaagin_Tsedenbal }
				set_character_as_ruler = yes
			}
		}
		else = {
			create_character = {
				first_name = "yumjaagin_tsedenbal_first_name"
				last_name = "yumjaagin_tsedenbal_last_name"
				ruler = yes ig_leader = yes historical = yes
				birth_date = 1916.9.17
				interest_group = ig_rural_folk
				ideology = ideology_autocratic
				culture = cu:mongol
				religion = rel:atheist
				traits = {
					ambitious
					tactful
					meticulous
				}
				on_created = { set_variable = Yumjaagin_Tsedenbal }
			}
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Death
		random_scope_character = {
			limit = { has_variable = Khorloogiin_Choibalsan }
			kill_character = yes
		}
		
		#Major discontent
		add_radicals = {
			value = large_radicals
			interest_group = ig:ig_trade_unions
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}
