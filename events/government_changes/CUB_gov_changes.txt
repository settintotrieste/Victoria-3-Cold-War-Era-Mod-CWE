# Cuban Revolution of 1959
CUB_gov_changes.1 = {
    type = country_event

    title = CUB_1959_revolution.t 							# Title of the event; ADD LOCALISATION
    desc = CUB_1959_revolution.d 							# Description of the event; ADD LOCALISATION
    #flavor = CUB_1959_revolution.f 						# Flavor for event; ADD LOCALISATION, IF REQUIRED

    event_image = { video = "gfx/event_pictures/unspecific_politicians_arguing.bk2" }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds" 	# Icon that shows on the map or the outliner
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear" 	# Sound effect if needed

    duration = 1 # Duration to fire if triggers are met
	
	trigger = {													# What triggers the event, if using other events or journal entries leave empty
		
		#Overlord conditions
		c:CUB ?= this
		game_date >= 1959.1.1									# START DATE: Date colony became self governing ****
		game_date < 1969.1.1									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_presidential_regime = yes							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		#right_wing_ruling_coalition = yes						# Correct Ruling Coalition
		
		# Event has not been fired before
		NOT = { has_variable = CUB_1959_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = CUB_1959_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Becomes independent
		make_independent = yes
		
		#Enact new regime
		effect_starting_politics_communist_republic = yes
		effect_starting_politics_dictatorship = yes
		effect_communist_ruling_coalition = yes
		
		#Create new leader
		create_character = {
			first_name = "Fidel_Castro_first_name"
			last_name = "Fidel_Castro_last_name"
			ruler = yes ig_leader = yes historical = yes
			birth_date = 1926.8.13
			interest_group = ig_trade_unions
			ideology = ideology_autocratic
			culture = cu:cuban
			religion = rel:atheist
			traits = {
				charismatic
				ambitious
				imposing
			}
			on_created = { set_variable = Fidel_Castro }
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = very_large_radicals
			strata = lower
		}
		add_modifier = { 
			name = stirrings_of_revolution
			months = 60 
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}

		game_date < 2016.6.31									# END DATE: All historical events must not fire 10 years after the START DATE ****
		
		is_a_communist_government = yes							# Correct gov types
		is_sufficiently_independent = yes						# is_vassal = no		
		is_at_war = no											# war = no
		
		ruler ?= { has_variable = Fidel_Castro }
		
		# Event has not been fired before
		NOT = { has_variable = CUB_2006_hgc }					# CHANGE ****
		
    }
	
	immediate = {
	
		# Fire only once
		set_variable = CUB_2006_hgc								# Change variable ****
		
	}
    
    option = {
        name = historical_gov_changes.option.1					# Accept
        highlighted_option = yes
		
		#Create Dynastic Republic
		random_scope_character = {
			limit = { has_variable = Fidel_Castro }
			set_variable = dynastic_ruler_enabled
		}
		
		#Raúl Castro
		create_character = {
			first_name = "Raul_Castro_first_name"
			last_name = "Raul_Castro_last_name"
			heir = yes historical = yes
			birth_date = 1931.6.3
			interest_group = ig_trade_unions
			ideology = ideology_power_hungry
			religion = rel:atheist
			traits = {
				tactful
				reserved
				innovative
			}
			on_created = { 
				set_variable = Raul_Castro
				set_variable = dynastic_ruler_enabled 
			}
		}
		
		ai_chance = { 
			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_modifier = { 
			name = party_in_chaos_modifier
			months = 12
			is_decaying = yes 
		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { high_historical_ai_railroading = yes } add = -100 } modifier = { trigger = { medium_historical_ai_railroading = yes } add = 0 } modifier = { trigger = { low_historical_ai_railroading = yes } add = 20 } modifier = { trigger = { no_historical_ai_railroading = yes } add = 80 } 
		}
		
    }

}
