			interest_group = ig_rural_folk

			}

			base = 90 
		}
		
    }
	
	option = {
        name = historical_gov_changes.option.2				# Reject
		default_option = yes
		
		#Major discontent
		add_radicals = {
			value = large_radicals

		}
		
		ai_chance = { 
			base = 10 modifier = { trigger = { historical_railroading_active = yes } add = -100 }
