# Completion event 1
communism.1 = {
	type = country_event
	placement = root
	title = communism.1.t
	desc = communism.1.desc
	flavor = communism.1.f

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		# triggered by communsim JE 1
	}

	immediate = {
		random_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_HOS_communist_leader
					value > neutral
				}
			}
			set_variable = communist_ig_var
			save_scope_as = unions
			leader = {
				save_scope_as = union_leader
			}
		}
		set_variable = {
			name = je_communism_2_var
			value = 0
		}
		add_journal_entry = {
			type = je_communism_2
		}
	}

	option = { # wage controls
		name = communism.1.a
		trigger = {
			NOT = { has_law_or_variant = law_type:law_medium_social_security }
			NOT = { has_law_or_variant = law_type:law_high_social_security }
		}
		activate_law = law_type:law_medium_social_security
		if = {
			limit = {
				any_interest_group = {
					has_variable = communist_ig_var
				}
			}
			custom_tooltip = communist_unions.tt
		}
	}
	option = { # Regulatory bodies
		name = communism.1.c
		trigger = {
			NOT = { has_law_or_variant = law_type:law_restricted_trade_unions }
			NOT = { has_law_or_variant = law_type:law_full_trade_unions }
		}
		activate_law = law_type:law_restricted_trade_unions
		if = {
			limit = {
				any_interest_group = {
					has_variable = communist_ig_var
				}
			}
			custom_tooltip = communist_unions.tt
		}
	}
	option = {
		name = communism.1.d
		default_option = yes
		if = {
			limit = {
				any_interest_group = {
					has_variable = communist_ig_var
				}
			}
			custom_tooltip = communist_unions.tt
		}
		every_interest_group = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_communist
					has_ideology = ideology:ideology_far_left_economy
				}
			}
			add_modifier = {
				name = communist_government_modifier
				days = normal_modifier_time
			}
		}
		add_loyalists = {
			value = medium_radicals
			strata = upper
		}
	}
}

# Completion event 2, choose ideology
communism.2 = {
	type = country_event
	placement = root
	title = communism.2.t
	desc = communism.2.d
	flavor = communism.2.f

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		# triggered by communsim JE 2
	}

	immediate = {
		ig:ig_trade_unions = {
			save_scope_as = unions
			leader = {
				save_scope_as = union_leader
			}
		}
	}
	option = { # vanguardism
		name = communism.2.a
		
		ig:ig_trade_unions = {
			set_variable = chose_vanguardism
		}
		custom_tooltip = chose_vanguardism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_autocratic }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_autocratic
			}
		}
		add_modifier = {
			name = path_to_vanguardism
			days = normal_modifier_time
		}
		if = {
			limit = {
				any_interest_group = {
					OR = {
						has_ideology = ideology:ideology_communist
						has_ideology = ideology:ideology_far_left_economy
					}
					count >= 3
				}
				custom_tooltip = {
					text = vanguard_custom_tooltip
					NOT = {
						any_interest_group = {
							is_powerful = yes
							NOR = {
								has_ideology = ideology:ideology_communist
								has_ideology = ideology:ideology_far_left_economy
							}
						}
					}
				}
			}
			trigger_event = { id = communism.5 }
		}
		else = {
			add_journal_entry = {
				type = je_vanguard
			}
		}
		add_journal_entry = {
			type = je_5_year_plan
		}
	}
	option = { # communism
		name = communism.2.b
		default_option = yes
		ig:ig_trade_unions = {
			set_variable = chose_communism
		}
		custom_tooltip = chose_communism.tt
		if = {
			limit = {
				scope:union_leader = {
					NOT = { has_ideology = ideology:ideology_power_hungry }
				}
			}
			scope:union_leader = {
				set_ideology = ideology:ideology_power_hungry
			}
		}
		add_modifier = {
			name = path_to_communism
			days = normal_modifier_time
		}
		if = {
			limit = {
				root.bureaucracy > 0
				approaching_bureaucracy_shortage = no
				has_law_or_variant = law_type:law_public_schools
				institution_investment_level = {
					institution = institution_schools
					value = 5
				}
				has_law_or_variant = law_type:law_public_health_insurance
				institution_investment_level = {
					institution = institution_health_system
					value = 5
				}
				has_law_or_variant = law_type:law_high_social_security
				institution_investment_level = {
					institution = institution_social_security
					value = 5
				}
				has_law_or_variant = law_type:law_full_trade_unions
				institution_investment_level = {
					institution = institution_workplace_safety
					value = 5
				}
			}
			trigger_event = {
				id = communism.6
			}
		}
		else = {
			add_journal_entry = {
				type = je_communist_welfare
			}
		}
		add_journal_entry = {
			type = je_5_year_plan
		}
	}
	
}

# Worker coops, happy workers
communism.9 = {
	type = country_event
	placement = scope:coop_state
	title = communism.9.t
	desc = communism.9.d
	flavor = communism.9.f

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	event_image = {
		video = "africa_prosperous_farm"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		has_law_or_variant = law_type:law_HOS_communist_leader
		any_scope_state = {
			is_incorporated = yes
			NOT = { has_modifier = coop_loyalty_modifier }
			NOT = { has_modifier = coop_migration_modifier }
			any_scope_pop = {
				pop_acceptance >= acceptance_status_4
				standard_of_living > 15
				is_employed = yes
				workplace ?= {
					OR = {
					    has_active_production_method = pm_worker_cooperative_building_agriculture
					    has_active_production_method = pm_worker_cooperative_building_logging
					    has_active_production_method = pm_worker_cooperative_building_manufacturing
						has_active_production_method = pm_worker_cooperative_building_mining
						has_active_production_method = pm_worker_cooperative_building_services
                    }
					cash_reserves_ratio > 0.25
					weekly_profit > 5
					occupancy >= 0.95
					level >= 5
				}
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				NOT = { has_modifier = coop_loyalty_modifier }
				NOT = { has_modifier = coop_migration_modifier }
				any_scope_pop = {
					pop_acceptance >= acceptance_status_4
					standard_of_living > 15
					is_employed = yes
					workplace ?= {
						OR = {
                            has_active_production_method = pm_worker_cooperative_building_agriculture
							has_active_production_method = pm_worker_cooperative_building_logging
							has_active_production_method = pm_worker_cooperative_building_manufacturing
							has_active_production_method = pm_worker_cooperative_building_mining
							has_active_production_method = pm_worker_cooperative_building_services
                        }
						cash_reserves_ratio > 0.25
						weekly_profit > 5
						occupancy >= 0.95
						level >= 5
					}
				}
			}
			save_scope_as = coop_state
			random_scope_pop = {
				limit = {
					pop_acceptance >= acceptance_status_4
					standard_of_living > 15
					is_employed = yes
					workplace ?= {
						OR = {
                            has_active_production_method = pm_worker_cooperative_building_agriculture
							has_active_production_method = pm_worker_cooperative_building_logging
							has_active_production_method = pm_worker_cooperative_building_manufacturing
							has_active_production_method = pm_worker_cooperative_building_mining
							has_active_production_method = pm_worker_cooperative_building_services
                        }
						cash_reserves_ratio > 0.25
						weekly_profit > 5
						occupancy >= 0.95
						level >= 5
					}
				}
				save_scope_as = coop_pop
				workplace = {
					save_scope_as = coop_building
				}
			}
		}
	}

	option = {
		name = communism.9.a
		default_option = yes
		scope:coop_state = {
			add_modifier = {
				name = coop_loyalty_modifier
				days = normal_modifier_time
			}
		}
	}

	option = {
		name = communism.9.b
		scope:coop_state = {
			add_modifier = {
				name = coop_migration_modifier
				days = normal_modifier_time
			}
		}
	}
}
