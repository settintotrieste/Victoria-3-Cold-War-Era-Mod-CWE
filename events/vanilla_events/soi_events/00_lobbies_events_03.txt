# Impressive Tax Reform
lobby_events.40 = {
	type = country_event
	placement = root
	title = lobby_events.40.t
	desc = lobby_events.40.d
	flavor = lobby_events.40.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	#dlc = dlc010

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		#has_dlc_feature = lobbies
		is_revolutionary = no
		OR = {
			AND = {
				is_enacting_law = law_type:law_taxation_4
				any_interest_group = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_4
						value > neutral
					}
				}
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_4
					}
				} 
			}
			AND = {
				is_enacting_law = law_type:law_taxation_5
				any_interest_group = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_5
						value > neutral
					}
				}
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_5
					}
				}
			}
		}	
	}

	immediate = {
		if = {
			limit = {
				is_enacting_law = law_type:law_taxation_4
				any_interest_group = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_4
						value > neutral
					}
				}
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_4
					}
				}				
			}
			random_interest_group = {
				limit = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_4
						value > neutral
					}
				}
				save_scope_as = relevant_ig
			}
			random_political_lobby = {
				limit = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_4
					}
				}
				target = {
					save_scope_as = relevant_lobby_country
				}
				save_scope_as = relevant_lobby	
			}
		}
		else = {
			limit = {
				is_enacting_law = law_type:law_taxation_5
				any_interest_group = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_5
						value > neutral
					}
				}
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_5
					}
				}				
			}
			random_interest_group = {
				limit = {
					is_member_of_any_lobby = no
					law_stance = {
						law = law_type:law_taxation_5
						value > neutral
					}
				}
				save_scope_as = relevant_ig
			}
			random_political_lobby = {
				limit = {
					is_political_lobby_type = lobby_pro_country
					target = { 
						has_law_or_variant = law_type:law_taxation_5
					}
				}
				target = {
					save_scope_as = relevant_lobby_country
				}
				save_scope_as = relevant_lobby	
			}
		}
		currently_enacting_law = {
    		save_scope_as = relevant_law_scope
    	}
	}

	cancellation_trigger = {
		OR = {
			NOT = { currently_enacting_law = scope:relevant_law_scope }
			NOT = { exists = scope:relevant_lobby }
			NOT = { exists = scope:relevant_lobby_country }
		}		
	}

	option = {
		name = lobby_events.40.a
		default_option = yes
		scope:relevant_lobby = {
			add_lobby_member = scope:relevant_ig
			change_appeasement = {
			   amount = 1
			   factor = appeasement_lobby_pleased
			}
		}
		add_enactment_modifier = {
			name = ig_happy_with_law_enactment
		}
	}
	option = {
		name = lobby_events.40.b
		scope:relevant_lobby = {
			add_lobby_member = scope:relevant_ig
		}
		scope:relevant_ig = {
			add_modifier = {
				name = ig_join_pol_str_positive
				days = short_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = {
			value = small_radicals
			interest_group = scope:relevant_ig
		}
	}
}

lobby_events.33 = {
	type = country_event
	placement = root
	title = lobby_events.33.t
	desc = lobby_events.33.d
	flavor = lobby_events.33.f

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	#dlc = dlc010

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		#has_dlc_feature = lobbies
		is_revolutionary = no
		any_political_lobby = {
			OR = {
				is_political_lobby_type = lobby_pro_country
				is_political_lobby_type = lobby_anti_country
			}
			any_lobby_member = {
				is_marginal = no
				count >= 3
			}
		}
	}

	immediate = {
		random_political_lobby = {
			limit = {
				OR = {
					is_political_lobby_type = lobby_pro_country
					is_political_lobby_type = lobby_anti_country
				}
				any_lobby_member = {
					is_marginal = no
					count >= 3
				}
			}
			save_scope_as = relevant_lobby
			random_lobby_member = {
				limit = {
					is_marginal = no
				}
				save_scope_as = relevant_ig
			}
			random_lobby_member = { # another ig in the same lobby
				limit = {
					is_marginal = no
					NOT = { this = scope:relevant_ig }	
				}
				save_scope_as = other_ig
			}
		}
	}

	cancellation_trigger = {
		OR = {
			NOT = { exists = scope:relevant_lobby }
			NOT = { exists = scope:relevant_lobby_country }
		}	
	}

	option = {
		name = lobby_events.33.a
		default_option = yes
		scope:relevant_lobby = {
			remove_lobby_member = scope:relevant_ig
		}
		scope:other_ig = {
			add_modifier = {
				name = ig_join_pol_str_positive
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}
	option = { # appears if relevant_ig is not in government
		name = lobby_events.33.b
		trigger = {
			scope:relevant_ig = {
				is_in_government = no
			}
		}
		scope:relevant_ig = {
			add_modifier = {
				name = lobby_events_ig_blocked 
				days = short_modifier_time
			}
		}
	}
	option = { # appears if relevant_ig is in government
		name = lobby_events.33.c
		trigger = {
			scope:relevant_ig = {
				is_in_government = yes
			}
		}
		scope:relevant_ig = {
			add_modifier = {
				name = lobby_events_ig_blocked 
				days = normal_modifier_time
			}
		}
		add_radicals = {
			value = medium_radicals
			interest_group = scope:relevant_ig
		}
	}
}
