# state scope
strike_state_add_modifier_effect_strong = {
	if = {
		limit = {
			owner = {
				has_variable = industrial_strike
			}
		}
		add_modifier = {
			name = industrial_strike_modifier
			multiplier = 2
			months = normal_modifier_time
		}
	}
	else_if = {
		limit = {
			owner = {
				has_variable = mining_strike
			}
		}
		add_modifier = {
			name = mining_strike_modifier
			multiplier = 2
			months = normal_modifier_time
		}
	}
	else_if = {
		limit = {
			owner = {
				has_variable = railway_strike
			}
		}
		add_modifier = {
			name = railway_strike_modifier
			multiplier = 2
			months = normal_modifier_time
		}
	}
	if = { 
		limit = {
			any_scope_pop = {
				is_pop_type = laborers
			}
		}
		add_radicals = {
			value = medium_radicals
			pop_type = laborers
		}
	}
	if = { 
		limit = {
			any_scope_pop = {
				is_pop_type = machinists
			}
		}
		add_radicals = {
			value = medium_radicals
			pop_type = machinists
		}
	}
}
