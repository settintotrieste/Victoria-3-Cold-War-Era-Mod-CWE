	texture = "gfx/interface/icons/ig_icons/minority_groups_middle.dds"

					multiply = 0.0025
					add = 1
					max = 1.25

				}
			}

				value = 0.1
			}

	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
