law_monarchy = {
	group = lawgroup_governance_principles
	
	icon = "gfx/interface/icons/law_icons/monarchy.dds"
	
	progressiveness = 0

	disallowing_laws = {
		law_anarchy
	}

	modifier = {
		country_legitimacy_headofstate_add = 20
		country_legitimacy_govt_total_clout_add = 0.20
		country_legitimacy_govt_size_mult = -0.30
		
		interest_group_ig_landowners_pol_str_mult = 0.25
	}
	
	possible_political_movements = {
		law_presidential_republic
		law_semi_parliamentary_republic
		law_parliamentary_republic
		law_council_republic
	}
	
	pop_support = {
		value = 0
		
		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = { 
					is_pop_type = peasants
					owner = { has_law = law_type:law_monarchy }
				}
				value = 1.0
			}
		}		
	}
	
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology_royalist
		}
	}
}
